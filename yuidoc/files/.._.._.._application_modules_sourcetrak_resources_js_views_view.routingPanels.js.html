/**
 * Routing View
 *
 * @module Sourcetrak
 * @class Views.Wizard
 */
Ibp.Sourcetrak.Views.RoutingPanels = Ibp.Backbone.View.extend({
    name: 'RoutingPanels',
    template: Ibp.Sourcetrak.Templates.routing_panels,

    /**
     * Pattern Panel DOM object
     *
     * @type jQuery Object
     */
    $patternPanel: null,

    /**
     * Pattern Panel CSS selector
     *
     * @type String
     */
    patternPanelEl: '.js-pattern-panel',

    /**
     * Schedule Panel DOM object
     *
     * @type jQuery Object
     */
    $schedulePanel: null,

    /**
     * Schedule Panel CSS selector
     *
     * @type String
     */
    schedulePanelEl: '.js-schedule-panel',

    /**
     * Route Panel DOM object
     *
     * @type jQuery Object
     */
    $routePanel: null,

    /**
     * Route Panel CSS selector
     *
     * @type String
     */
    routePanelEl: '.js-route-panel',

    /**
     * Pattern Panel View
     *
     * @type Ibp.Backbone.View Object
     */
    patternPanelView: null,

    /**
     * Schedule Panel View
     *
     * @type Ibp.Backbone.View Object
     */
    schedulePanelView: null,

    /**
     * Route Panel View
     *
     * @type Ibp.Backbone.View Object
     */
    routePanelView: null,

    /**
     * Container DOM object
     *
     * @type jQuery Object
     */
    $container: null,

    /**
     * Model to hold view state
     *
     * @property state
     * @type Model
     */
    state: null,

    /**
     * Define the channels to subscribe
     *
     * @type Object
     */
    subscriptions: {
        'enablePatterns': 'handleEnablePatterns',
        'enableSchedules': 'handleEnableSchedules',
        'routingState': 'handleRoutingStateChange'
    },

    /**
     * Init state and render
     *
     * @method init
     */
    init: function (options) {
        var firstMode = this.model.getRouting().getRouteByPattern(0).getConfig().at(0).get('mode_id');

        this.$container = options.$container;

        // set initial state
        this.state = new Ibp.Carbon.Models.State({
            patterns: this.model.arePatternsEnabled(),
            schedules: this.model.areSchedulesEnabled(),
            activePatternId: 0,
            activeScheduleId: firstMode > 0 ? Ibp.Sourcetrak.Data.Schedules.getScheduleByMode(firstMode).id : 0,
            activeModeId: firstMode
        });

        this.render();

        this.listenTo(this.model, 'change:routing', this.handelRoutingChange);
        this.listenTo(this.state, 'change', this.applyColumnClasses);
    },

    /**
     * Render
     *
     * @method render
     * @return this view
     */
    render: function () {
        var routingState = {
            activePatternId: this.state.get('activePatternId'),
            activeScheduleId: this.state.get('activeScheduleId'),
            activeModeId: this.state.get('activeModeId')
        };

        this.unrenderPanels();

        this.$el = $(this.template());

        this.$patternPanel  = this.$(this.patternPanelEl);
        this.$schedulePanel = this.$(this.schedulePanelEl);
        this.$routePanel    = this.$(this.routePanelEl);

        this.$container.html(this.$el);

        this.applyColumnClasses();

        this.patternPanelView = new Ibp.Sourcetrak.Views.PatternPanel({
            el: this.$patternPanel,
            model: this.model,
            routingState: routingState
        });

        this.schedulePanelView = new Ibp.Sourcetrak.Views.SchedulePanel({
            el: this.$schedulePanel,
            model: this.model,
            routingState: routingState
        });

        this.routePanelView = new Ibp.Sourcetrak.Views.RoutePanel({
            el: this.$routePanel,
            model: this.model,
            routingState: routingState
        });

        return this;
    },

    /**
     * handlePatternChange
     *
     * @method handlePatternChange
     */
    handleRoutingStateChange: function (routingState) {
        this.state.set(routingState);
    },

    /**
     * Handle Enable Patterns Subscription Event
     *
     * @method handleEnablePatterns
     */
    handleEnablePatterns: function (enablePatterns) {
        var state = null,
            firstMode = null;

        if (!enablePatterns) {
            this.model.getRouting().deleteAllPatterns();

            firstMode = this.model.getRouting().getRouteByPattern(0).getConfig().at(0).get('mode_id');

            state = {
                activePatternId: 0,
                activeScheduleId: firstMode > 0 ? Ibp.Sourcetrak.Data.Schedules.getScheduleByMode(firstMode).id : 0,
                activeModeId: firstMode
            };

            this.state.set(state);

            this.publish('routingState', state);
        }
        else {
            this.patternPanelView.render();
        }

        this.state.set({'patterns': enablePatterns});
    },

    /**
     * Handle Enable Schedules Subscription Event
     *
     * @method handleEnablePatterns
     */
    handleEnableSchedules: function (enableSchedules) {
        var state = null;

        if (!enableSchedules) {
            this.model.getRouting().deleteAllSchedules();

            state = {
                activePatternId: this.state.get('activePatternId'),
                activeScheduleId: 0,
                activeModeId: 0
            };

            this.state.set(state);

            this.publish('routingState', state);
        }
        else {
            this.schedulePanelView.render();
        }

        this.state.set({'schedules': enableSchedules});
    },

    /**
     * Handle Routing Changes
     *
     * @method handelRoutingChange
     */
    handelRoutingChange: function () {
        this.updateColumnStates();
    },

    /**
     * Apply Column Classes
     *
     * @method applyColumnClasses
     */
    applyColumnClasses: function() {
        var state  = this.state.toJSON(),
            columns = 1 + (state.patterns ? 1 : 0) + (state.schedules ? 1 : 0),
            columnNumber = {
                pattern:  state.patterns ? 1 : 0,
                schedule: 0,
                route:    1
            };

        columnNumber.schedule = (state.schedules ? 1 : 0) + (state.schedules ? columnNumber.pattern : 0);
        columnNumber.route    = columnNumber.pattern + (state.schedules ? 1 : 0) + columnNumber.route;

        this.$el.attr('data-columns', columns);

        this.$patternPanel.attr('data-column-number', columnNumber.pattern);
        this.$schedulePanel.attr('data-column-number', columnNumber.schedule);
        this.$routePanel.attr('data-column-number', columnNumber.route);
    },

    /**
     * Update Column States
     *
     * @method updateColumnStates
     */
    updateColumnStates: function() {
        var state = {
            patterns: (this.model.arePatternsEnabled() || this.state.get('patterns')),
            schedules: (this.model.areSchedulesEnabled() || this.state.get('schedules'))
        };

        this.state.set(state);
    },

    /**
     * UnRender Panels
     *
     * @method unrenderPanels
     */
    unrenderPanels: function () {
        if (this.patternPanelView !== null) {
            this.patternPanelView.unrender();
        }

        if (this.schedulePanelView !== null) {
            this.schedulePanelView.unrender();
        }

        if (this.routePanelView !== null) {
            this.routePanelView.unrender();
        }
    },

    /**
     * onUnrender
     *
     * @method onUnrender
     */
    onUnrender: function () {
        this.undelegateSubscriptions();

        this.unrenderPanels();
    }
});

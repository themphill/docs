/**
 * St Snippet Modal
 * Modal that displays the ST Snippet with Copy to Clipboard button
 *
 * This class defines init, render, and unrender in a way that is dramatically better than the current
 * modal implementation in Ibp.Carbon.Views.Modal. That base class should be refactored like this one.
 * Views that would inherit from the refactored base class would only needs to implement the
 * onInit(), onRender(), and onUnrender() hooks to add additional functionality.
 * The modal's parent view would then call show() instead of render().
 * The modal's template must include the root modal element, instead of creating it in init(), as in the template here.
 *
 * @module SourceTrak
 * @class Views.SnippetModal
 * @extends Ibp.Carbon.Views.Modal
 */
 Ibp.Sourcetrak.Views.ConfirmDeleteModal = Ibp.Carbon.Views.Modal.extend({
    name: 'snippetModal',
    template: Ibp.Sourcetrak.Templates.modal_delete_domain_set,
    isDestroying: false,

    /**
     * DOM Event handlers
     *
     * @property events
     */
    events: {
        'click #modal-confirm': 'confirm',
        'click #modal-cancel': 'cancel',
        'hidden': 'hidden'
    },

    /**
     * Refactored init function from Ibp.Carbon.Views.Modal
     * Stores
     *
     * @method init
     * @params options {Object} Config object
     * @params options.template {Function} Template function. Required if not defined in the view.
     * @params options.templateData {Object} Data to pass to template function. Defaults to empty object.
     * @params options.confirmHandler {Function} 'confirm' button click handler. Defaults to a no-op.
     */
    init : function (options) {
        this.template = options.template || this.template;
        this.templateData = options.templateData || {};
        this.confirmHandler = options.handler || function () {};

        this.onInit(options);

        return this.render();
    },

    /**
     * Cancel function is used by both cancel click and hide events. Passes the DOM event and passes true for the
     * navigate param for unrender.
     *
     * @method cancel
     * @params e {Object} DOM Event
     * @return this view
     */
    cancel: function () {
        if (!this.isDestroying) {
            this.isDestroying = true;
            this.publish('modal:unrender');
        }
    },

    hidden: function (e) {
        // make sure we're only listening to hide events from this modal
        if (e && e.target.id !== this.$el.attr('id')) {
            return;
        }

        this.cancel();
    },

    /**
     * Refactored render function from Ibp.Carbon.Views.Modal
     * Handles removing existing modal, rendering this modal, and calling its hook function
     *
     * @method render
     * @return this view
     */
    render: function () {
        // create modal fragment
        this.$el = $(this.template(this.templateData));

        // add this modal to the DOM
        $('body').append(this.$el);

        this.onRender();

        return this;
    },

    /**
     * Refactored unrender function from Ibp.Carbon.Views.Modal
     * On this modal's hide or cancel button click event, removes this view
     *
     * @method unrender
     */
    unrender: function (e) {
        this.onUnrender();

        this.undelegateEvents();
        this.stopListening();
        this.hide();
        this.$el.remove();
    },

    /**
     * Sets the template data
     *
     * @method onInit
     */
    onInit: function (options) {
        this.templateData = {
            domainSetName: options.model.attributes.name
        };
    },

    confirm: function () {
        this.model.collection.trigger('delete', this.model);

        this.publish('navigate', '/');
    }
});

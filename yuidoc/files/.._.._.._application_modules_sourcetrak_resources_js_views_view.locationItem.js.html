/**
 * Domain Set Single Location Item View
 * View container for a single Domain Set Location
 *
 * @module SourceTrak
 * @class Views.LocationItem
 * @extends Ibp.Backbone.View
 */
Ibp.Sourcetrak.Views.LocationItem = Ibp.Backbone.View.extend({
    mixins: ['Autosave'],
    name: 'LocationItem',
    template: Ibp.Sourcetrak.Templates.location_item,

    /**
     * String selector for the DOM element containing the entire location item.
     *
     * @property locationWrapperEl
     * @type String
     */
    locationWrapperEl: '.js-location-wrapper',

    /**
     * Holds a jQuery reference to the DOM element containing the entire location item.
     *
     * @property $locationWrapper
     * @type jQuery Object
     */
    $locationWrapper: null,

    /**
     * String selector for the DOM element containing the expanded location node.
     *
     * @property expandedLocationContainerEl
     * @type String
     */
    expandedLocationContainerEl: '.js-expanded-location-wrapper',

    /**
     * Holds a jQuery reference to the expandedLocationContainerEl DOM element.
     *
     * @property $expandedLocationContainer
     * @type jQuery Object
     */
    $expandedLocationContainer: null,

    /**
     * String selector for the title element that will trigger the
     * full location view to slide in and out of view/toggle.
     *
     * @property toggleTriggerEl
     * @type String
     */
    toggleTriggerEl: '.js-location-title',

    /**
     * Holds a jQuery reference to the title element that will trigger the
     * full location view to slide in and out of view/toggle.
     *
     * @property $toggleTrigger
     * @type jQuery Object
     */
    $toggleTrigger: null,

    /**
     * String selector for this view's replacement number editable
     * label DOM element.
     *
     * @property replacementNumberLabelEl
     * @type String
     */
    replacementNumberLabelEl: '.js-replacement-number',

    /**
     * Holds a jQuery reference to the replacementNumberLabelEl.
     *
     * @property $replacementNumberLabel
     * @type jQuery Object
     */
    $replacementNumberLabel: null,

    /**
     * String selector for the locationNameLabelEl editable
     * label DOM element.
     *
     * @property locationNameLabelEl
     * @type String
     */
    locationNameLabelEl: '.js-location-name',

    /**
     * Holds a jQuery reference to the the locationNameLabelEl.
     *
     * @property $locationName
     * @type jQuery Object
     */
    $locationNameLabel: null,

    /**
    * Stores the editable label views for this location view.
    *
    * @property editableLabels
    * @type Object
    */
    // editableLabels: null,
    views: null,

    /**
     * Backbone Model to hold view state
     *
     * @property state
     * @type Model
     */
    state: null,

    /**
     * Max number of characters before truncation of name property occurs
     *
     * @property NAME_MAX_CHARS
     * @type Number
     */
    NAME_MAX_CHARS: 72,

    /**
     * Max number of characters before truncation of routing summary property occurs
     *
     * @property ROUTING_MAX_CHARS
     * @type Number
     */
    ROUTING_MAX_CHARS: 52,

    /**
     * Max number of characters before truncation of area codes string occurs
     *
     * @property AREA_CODES_MAX_CHARS
     * @type Number
     */
    AREA_CODES_MAX_CHARS: 21,

    /**
     * Max number of characters for the name of the location when viewed as a
     * title in the collapsed headline area (different max chars from the name as
     * it appears in the expanded location view).
     *
     * @property TITLENAME_MAX_CHARS
     * @type Number
     */
    TITLENAME_MAX_CHARS: 104,

    events: {
        'click .js-location-header': 'handleToggleClick',
        'click .js-edit-location': 'renderEditLocation',
        'click .js-delete-location': 'renderDeleteModal'
    },

    /**
     * Init
     *
     * @method init
     */
    init: function () {
        this.views = [];

        this.state = new Ibp.Carbon.Models.State({
            isExpanded: false,
            focusName: false,
            isAnimating: false
        });

        this.listenTo(this.state, 'change:isExpanded', this.handleStateChange);

        this.listenTo(this.model, 'sync', this.render);

        this.render();
    },

    /**
     * Render
     * Actually attaching this view to the DOM is handled by Views.LocationsList
     *
     * @method render
     */
    render: function () {
        var data = {
                name: this.model.get('name'),
                areaCodes: this.model.getAreaCodesString(),
                numberQty: this.model.getTotalNumbers(),
                routingSummary: this.model.getRoutingSummary(),
                isExpanded: this.state.get('isExpanded')
            };

        data.truncated = {
            name: Ibp.Utils.truncate(data.name, this.NAME_MAX_CHARS),
            title: Ibp.Utils.truncate(data.name, this.TITLENAME_MAX_CHARS),
            areaCodes: Ibp.Utils.truncate(data.areaCodes, this.AREA_CODES_MAX_CHARS),
            routingSummary: Ibp.Utils.truncate(data.routingSummary, this.ROUTING_MAX_CHARS)
        };

        this.$el.html(this.template(data));

        // Create view references to dom elements that now exist
        this.$toggleTrigger = this.$(this.toggleTriggerEl);
        this.$locationWrapper = this.$(this.locationWrapperEl);
        this.$locationNameLabel = this.$(this.locationNameLabelEl);
        this.$replacementNumberLabel = this.$(this.replacementNumberLabelEl);
        this.$expandedLocationContainer = this.$(this.expandedLocationContainerEl);

        this.renderEditableLabels();

        return this;
    },

    /**
     * Sets isExpanded attribute on the state model to its opposite boolean
     * value.
     *
     * @method handleToggleClick
     */
    handleToggleClick: function () {
        // don't animate if it's already animating
        if (this.state.get('isAnimating')) {
            return;
        }

        this.state.set('isExpanded', !this.state.get('isExpanded'));
    },

    /**
     * Handler for expanded view state change, triggers animation
     * Will fetch() the full model if necessary
     *
     * @method handleStateChange
     */
    handleStateChange: function () {
        if (this.state.get('isExpanded') && !this.model.hasBeenFetched()) {
            this.model.fetch();
        }

        this.animateLocation();
    },

    /**
     * Handles the animation for expanding and collapsing the location DOM node.
     *
     * @method animateLocation
     */
    animateLocation: function () {
        var _this = this;

        this.$expandedLocationContainer.slideToggle({
            duration: 350,
            start: function () {
                _this.state.set({isAnimating: true});

                _this.$locationWrapper.addClass('animating');
                _this.$locationWrapper.toggleClass('expanded collapsed');
            },
            complete: function () {
                var state = {
                        isAnimating: false
                    };

                _this.$locationWrapper.removeClass('animating');
                _this.$expandedLocationContainer.removeClass('hidden');

                if (_this.state.get('focusName')) {
                    _this.views.nameLabel.focusInput();

                    state.focusName = false;
                }

                _this.state.set(state);
            }
        });
    },

    /**
     * Takes care of rendering the editable label views utilizing the Carbon module
     *
     * @method renderEditableLabels
     */
    renderEditableLabels: function () {
        var nameOptions = {
                $container: this.$locationNameLabel,
                labelName: null,
                labelAttribute: 'name',
                labelPlacholder: 'Enter Location Name',
                truncateAt: this.NAME_MAX_CHARS,
                model: this.model
            },
            numberOptions = {
                $container: this.$replacementNumberLabel,
                labelName: 'Number',
                labelAttribute: 'replacement_number',
                labelPlaceholder: 'Enter Phone Number',
                maskPhoneNumber: true,
                model: this.model
            };

        // remove old label views
        _.invoke(this.views, 'unrender');

        this.views.nameLabel = new Ibp.Carbon.Views.EditableLabel(nameOptions);

        this.views.numberLabel = new Ibp.Carbon.Views.EditableLabel(numberOptions);
    },

    /**
     * Renders modal to confirm or cancel deletion
     *
     * @method unrender
     */
    renderDeleteModal: function () {
        var data = {
            model: this.model,
            templateData: {
                name: this.model.get('name')
            }
        };

        this.publish('modal:render', {name: 'deleteLocation', data: data});
    },

    /**
     * Takes care of removing the editable label views once their parent el
     * is unrendered.
     *
     * @method onUnrender
     */
    onUnrender: function () {
        _.invoke(this.views, 'unrender');
    },

    /**
     * Renders the edit location pool wizard modal
     *
     * @method renderEditLocation
     */
    renderEditLocation: function () {
        var options = {
            name: 'pools',
            data: {
                viewType: 'wizard',
                wizardType: 'location',
                model: this.model
            }
        };

        this.publish('modal:render', options);
    },

    /**
     * Allows new locations to be expanded
     *
     * @method showNew
     */
    showNew: function () {
        this.state.set({
            isExpanded: true,
            focusName: true
        });
    }
});

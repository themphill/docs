Ibp.Sourcetrak.Views.Dashboard = Ibp.Carbon.Views.PageView.extend({
    name: 'dashboard',
    el: '#sourcetrak-dashboard',
    template: Ibp.Sourcetrak.Templates.dashboard,

    /**
     * A list of references to child views
     *
     * @property views
     * @type Object
     */
    views: null,

    /**
     * A list of references of the views for each rendered collection item
     *
     * @property itemViews
     * @type Object
     */
    itemViews: null,

    /**
     * Domain sets container selector
     *
     * @property filterAlertEl
     * @type String
     */
    domainSetsEl: '.js-domain-sets',

    /**
     * Domain sets container jQuery object
     *
     * @property $filterAlert
     * @type jQuery Object
     */
    $domainSets: null,

    /**
     * String selector for container holding number of items
     * displayed out of total items
     *
     * @property itemCountEl
     * @type String
     */
    itemCountEl: '.js-items-count',

    /**
     * Holds a jQuery reference to itemCountEl
     *
     * @property $itemCount
     * @type jQuery Object
     */
    $itemCount: null,

    /**
     * String selector for the paginator buttons
     *
     * @property paginatorControlsEl
     * @type String
     */
    paginatorControlsEl: '.js-paginator-controls',

    /**
     * Paginator controls jQuery object
     *
     * @property $paginatorControls
     * @type jQuery Object
     */
    $paginatorControls: null,

    /**
     * Domain sets container selector
     *
     * @property filterAlertEl
     * @type String
     */
    addItemEl: '.js-domain-set-add',

    /**
     * Domain sets container jQuery object
     *
     * @property $filterAlert
     * @type jQuery Object
     */
    $addItem: null,

    /**
     * Domain sets container selector
     *
     * @property filterAlertEl
     * @type String
     */
    emptyItemEl: '.js-domain-set-empty',

    /**
     * Domain sets container jQuery object
     *
     * @property $filterAlert
     * @type jQuery Object
     */
    $emptyItem: null,

    /**
     * searchboxEl
     *
     * @property searchboxEl
     * @type String
     */
    searchboxEl: '#search-box',

    events: {
        'click .js-domain-set-add': 'handleAddDomainSetClick'
    },

    /**
     * Init
     *
     * @method init
     */
    init: function () {
        _.bindAll(this);

        this.views = {};
        this.itemViews = [];

        // prime the paginator state
        this.collection.setPageState();

        this.state = new Ibp.Carbon.Models.State({
            query: '',
            showNew: false,
            showControls: this.shouldShowControls(),
            showEmptyItem: this.getEmptyItemState(),
            renderItems: true
        });

        this.listenTo(this.state, 'change', this.handleViewStateChange);
        this.listenTo(this.collection.ClientPagination.state, 'change', this.handlePageStateChange);

        // handle adding and removing domain sets
        this.listenTo(this.collection, 'add remove', this.handleCollectionChange);

        this.render();
    },

    /**
     * Renders dahsboard template and additional views like the Dashboard Toolbar, Searchboax, and Domain Set List.
     *
     * @method render
     * @return this view
     */
    render: function () {
        this.$el.append(this.template());

        this.$domainSets        = this.$(this.domainSetsEl);
        this.$itemCount         = this.$(this.itemCountEl);
        this.$paginatorControls = this.$(this.paginatorControlsEl);
        this.$addItem           = this.$(this.addItemEl);
        this.$emptyItem         = this.$(this.emptyItemEl);

        this.renderToolbar();

        this.renderSearchbox();

        this.renderItems();

        this.renderEmptyItem();

        return this;
    },

    /**
     * Renders the Dashbaord Toolbar
     *
     * @method renderToolbar
     */
    renderToolbar: function () {
        this.views.toolbar = new Ibp.Sourcetrak.Views.SourcetrakToolbar();
    },

    /**
     * Renders the Dashbaord Searchbox
     *
     * @method renderSearchbox
     */
    renderSearchbox: function () {
        this.views.searchBox = new Ibp.Carbon.Views.SearchBox({
            el: this.searchboxEl,
            placeholderText: 'Search By Name Or Domain'
        });

        this.listenTo(this.views.searchBox, 'search', this.handleSearch);
    },

    /**
     * Renders the Domain Sets List
     *
     * @method renderItems
     */
    renderItems: function () {
        var models   = this.collection.getPage(),
            modelIds = _.pluck(models, 'id'),
            viewIds  = _.chain(this.itemViews).pluck('model').pluck('id').value(),
            addedModelIds   = _.difference(modelIds, viewIds),
            removedModelIds = _.difference(viewIds, modelIds),
            viewState = {renderItems: false},
            // keep track of the container size to help prevent jank
            oldHeight = this.$domainSets.css('height'),
            newHeight;

        // just start from scratch for filters
        if (this.wasQueryChanged()) {
            _.invoke(this.itemViews, 'unrender');

            this.itemViews = _.map(models, function (model) {
                var view = new Ibp.Sourcetrak.Views.DomainSetItem({
                    model: model
                });

                this.$domainSets.append(view.el);

                return view;
            }.bind(this));
        }
        // attempt to preserve state for non-filter operations
        else {
            // unrender removed views
            if (removedModelIds.length) {
                var removed = [];

                _.each(this.itemViews, function (view) {
                    if (_.contains(removedModelIds, view.model.id)) {
                        view.unrender();

                        removed.push(view);
                    }
                }.bind(this));

                this.itemViews = _.difference(this.itemViews, removed);
            }

            // create and render new views
            if (addedModelIds.length) {
                _.each(addedModelIds, function (id) {
                    var view = new Ibp.Sourcetrak.Views.DomainSetItem({
                        model: _.findWhere(models, {id: id})
                    });

                    // handle the special case of adding a new model at the top of the list
                    if (this.state.get('showNew') && id === models[0].id) {
                        this.itemViews.unshift(view);

                        this.$domainSets.prepend(view.el);
                    }
                    else {
                        this.itemViews.push(view);

                        this.$domainSets.append(view.el);
                    }
                }.bind(this));
            }
        }

        // get new container size and set new base height if need be
        newHeight = this.$domainSets.css('height');

        if (newHeight > oldHeight) {
            this.$domainSets.css('min-height', newHeight);
        }

        this.renderPaginatorControls();

        this.renderItemCount();

        if (this.state.get('showNew')) {
            viewState.showNew = false;
        }

        // silently reset the view state
        this.state.set(viewState, {silent: true});
    },

    /**
     * Render "add domain set" row if there are no models
     * Render "no results found" if there are models but no search results
     *
     * @method renderEmptyItem
     */
    renderEmptyItem: function () {
        switch (this.state.get('showEmptyItem')) {
            case 'none':
                this.$addItem.hide();
                this.$emptyItem.hide();
                break;
            case 'add':
                this.$addItem.show();
                this.$emptyItem.hide();
                break;
            case 'empty':
                this.$addItem.hide();
                this.$emptyItem.show();
        }
    },

    /**
     * Renders the paginator buttons for switching pages
     *
     * @method renderPaginatorControls
     */
    renderPaginatorControls: function () {
        this.resetPaginatorControls();

        // don't show the controls
        if (!this.state.get('showControls')) {
            return;
        }

        this.views.paginatorControls = new Ibp.Sourcetrak.Views.listControls({
            $container: this.$paginatorControls,
            pageState:  this.collection.getPageState()
        });

        this.listenTo(this.views.paginatorControls, 'pageClick', this.handlePageClick);
    },

    /**
     * Renders the count of what the paginator is currently displaying
     * This is shown both above and below the paginated list of domainSets
     *
     * @method renderItemCount
     */
    renderItemCount: function () {
        // clear the old count
        this.$itemCount.html('');

        // don't show the count
        if (!this.state.get('showControls')) {
            return;
        }

        var info  = this.collection.getPageState(),
            total = info.totalItems,
            range = info.pageRange,
            displayed = range.end - range.start + 1,
            countMessage = 'Displaying ';

        // if we're displaying all of the items or just one item, don't show a range
        if (displayed === total || range.end - range.start === 0) {
            countMessage += range.end + ' of ' + total;
        }
        else {
            countMessage += range.start + ' - ' + range.end + ' of ' + total;
        }

        this.$itemCount.html(countMessage);
    },

    /**
     * Removes and cleans up the paginator controls so they can be re-rendered
     *
     * @method resetPaginatorControls
     */
    resetPaginatorControls: function () {
        var view = this.views.paginatorControls;

        if (view) {
            view.unrender();

            this.stopListening(view);

            delete this.views.paginatorControls;
        }
    },

    /**
     * Takes care of unrendering child views specific to this view.
     *
     * @method onUnrender
     */
    onUnrender: function () {
        _.invoke(this.views, 'unrender');

        _.invoke(this.itemViews, 'unrender');
    },

    /**
     * Navigates to the create domain set page
     *
     * @method handleAddDomainSetClick
     */
    handleAddDomainSetClick: function () {
        this.publish('navigate', 'create');
    },

    /**
     * Re-renders domainSets whenever the collection changes
     * Will automatically move back to the first page and expand a new domainSet when added
     *
     * @method handleCollectionChange
     * @param {Model} model Unused
     * @param {Collection} collection Unused
     * @param {Object} event Information about the event
     */
    handleCollectionChange: function (model, collection, event) {
        var viewState    = {
                showControls:  this.shouldShowControls(),
                showEmptyItem: this.getEmptyItemState()
            },
            resetOptions = {};

        // handle adding a new domainSet by checking if this was an add event
        if (event.add) {
            // move back to the first page and clear the query
            resetOptions.hard = true;

            // expand the new domainSet
            viewState.showNew = true;
        }

        // silently update view state
        this.state.set(viewState, {silent: true});

        // update the paginated collection and trigger re-render
        this.collection.resetPageState(resetOptions);
    },

    /**
     * Handles changes to the search box by setting the query state for the view and the domainSets
     *
     * @method handleSearch
     * @param query {String} The value of the search box
     */
    handleSearch: function (query) {
        this.collection.setQuery(query);
    },

    /**
     * Switches the page when a user clicks a paginator button
     *
     * @method collection
     * @param page {String} The page to change to
     */
    handlePageClick: function (page) {
        this.collection.setCurrentPage(page);
    },

    /**
     * Handles an updated query state
     *
     * @method handlePageStateChange
     * @param {Model} model The paginator state model
     */
    handlePageStateChange: function (model) {
        var state = {
            renderItems: true,
            query: model.get('query'),
            showEmptyItem: this.getEmptyItemState()
        };

        this.state.set(state);
    },

    /**
     * Handles an updated query state
     *
     * @method handleViewStateChange
     */
    handleViewStateChange: function () {
        // clear the search box if the paginator has been reset
        if (!this.state.get('query')) {
            this.views.searchBox.reset();
        }

        this.renderItems();

        this.renderEmptyItem();
    },

    /**
     * Determines if we should render the paginator controls
     * If the total domainSets only require a single page, the controls should be hidden
     * since they wouldn't actually do anything.
     * If the list has been filtered down to a single page, we'll still show the controls,
     * since they'll still be useful once the query has been cleared.
     *
     * @method shouldShowControls
     * @return {Boolean}
     */
    shouldShowControls: function () {
        return this.collection.length > this.collection.ClientPagination.state.get('pageSize');
    },

    /**
     * Determines which, if any, empty item should be shown
     * Show the add item when there are no domain sets for the account
     * Show the empty item when a filter shows no results
     *
     * @method getEmptyItemState
     * @return {String}
     */
    getEmptyItemState: function () {
        var pageState = this.collection.getPageState(),
            state;

        if (!this.collection.length) {
            state = 'add';
        }
        else if (pageState.query && !pageState.totalItems) {
            state = 'empty';
        }
        else {
            state = 'none';
        }

        return state;
    },

    /**
     * Determines if we're currently filtering or have just cleared a filter
     *
     * @method wasQueryChanged
     */
    wasQueryChanged: function () {
        var oldQuery = this.state.previous('query') || '',
            newQuery = this.state.get('query');

        return newQuery !== oldQuery;
    }
});

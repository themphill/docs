Ibp.KeywordSpotting.Views.KeywordTable = Ibp.Backbone.View.extend({
    name:        'keywordTable',
    template:    Ibp.KeywordSpotting.Templates.keyword_table,
    rows:        {},
    $tbody:      null,
    savePending: false,

    events: {
        'click .js-sort': 'sort'
    },

    subscriptions: {
        'newKeywordAdded': 'pushKeyword',
        'keywordDeleted':  'removeKeyword',
        'keywordsImported': 'refreshKeywords'
    },

    init: function(options) {
        this.render();
    },

    render: function() {
        this.$el.append(this.template());
        this.$tbody = $(this.$el.find(".table-body")[0]);

        this.renderRows(this.model.get('keywords'));
    },

    refreshKeywords: function() {
        this.renderRows(this.model.get('keywords'));
    },

    /**
     *
     * @param Array keywords
     * @TODO: use unrender() instead of remove()
     */
    renderRows: function(keywords) {
        var _this = this;

        _.each(this.rows, function(row) {
            row.remove();
        });

        this.$tbody.html('');
        this.rows = {};

        _.each(keywords, function(keyword) {
            _this.rows[keyword._id] = new Ibp.KeywordSpotting.Views.KeywordTableRow({
                container: _this.$tbody,
                model:     _this.model,
                keyword:   keyword
            });
        });

        if (keywords.length === 0) {
            this.rows.emptyRow = new Ibp.KeywordSpotting.Views.KeywordTableRowEmpty({
                container: _this.$tbody
            });
        }
    },

    /**
     * push a new keyword row into the table
     *
     * @param Object keyword
     */
    pushKeyword: function(keyword) {
        this.rows[keyword._id] = new Ibp.KeywordSpotting.Views.KeywordTableRow({
            container: this.$tbody,
            model:     this.model,
            keyword:   keyword,
            position:  'prepend'
        });

        if (this.rows.emptyRow) {
            this.rows.emptyRow.remove();
            delete this.rows.emptyRow;
        }

        this.publish('selectKeywordInput:' + keyword._id);
    },

    removeKeyword: function(keyword) {
        this.renderRows(this.model.get('keywords'));
    },

    sort: function(e) {
        var $target = $($(e.target).closest('.js-sort')[0]),
            sortBy = $target.attr('id'),
            order = $target.attr('data-order'),
            newOrder,
            newIcon;

        // reset sort headers and icons
        this.$el.find('.js-sort')
            .attr('data-order', 'default')
            .removeClass('active')
            .find('i').removeClass();

        // cycle sort order (asc > desc > default)
        switch (order) {
            case 'asc':
                newOrder = 'desc';
                newIcon = 'icon-arrow-down4';
                break;
            case 'desc':
                newOrder = 'default';
                newIcon = '';
                break;
            case 'default':
                newOrder = 'asc';
                newIcon = 'icon-arrow-up5';
                break;
            default:
            // do nothing
        }

        // set new order, active, and icon
        $target.attr('data-order', newOrder)
            .toggleClass('active', newOrder !== 'default')
            .find('i').addClass(newIcon);

        // sort collection
        this.sortBy(sortBy, newOrder);
    },

    /**
     * Methods for getting field names within the keyword object for sorting.
     * These get passed to _.sortBy() as the "iterator" param.
     */
    sortFields: {
        keyword: function(keyword) {
            return keyword._body.search_string.toLowerCase();
        },
        speaker: function(keyword) {
            var field = 'Agent';

            // set field to match label in select list
            switch (keyword._body.speaker) {
                case '1':
                    field = 'Agent';
                    break;
                case '2':
                    field = 'Caller';
                    break;
                case '3':
                    field = 'Either Caller or Agent';
                    break;
            }

            return field;
        },
        weight:  function(keyword) {
            // cast to integer to ensure proper sorting
            return Math.round(keyword._body.weight);
        }
    },

    /**
     * Sort keyword table using Underscore
     *
     * @param String sortBy
     * @param String order
     */
    sortBy: function(sortBy, order) {
        var _this = this;

        /**
         * DE1786
         * @link https://rally1.rallydev.com/#/16667679310d/detail/defect/19498459707
         *
         * Apparently while a model is being saved with only certain attributes (as opposed to the whole object) the
         * model only contains those properties that are being saved. In this case only one keyword is being updated so
         * the model only contains one keyword while saving is in progress. By waiting to sort until the save is
         * finished we will have all of our keywords back.
         */
        if (this.model.isSaving) {
            this.listenToOnce(this.model, 'saved', function() {
                _this.sortBy(sortBy, order);
            });
            return;
        }

        var sorted = _.sortBy(this.model.get('keywords'), this.sortFields[sortBy]);

        // _.sortBy() sorts ascending. if desc is requested simply reverse the array
        if ('desc' == order) {
            sorted.reverse();
        }

        this.renderRows(sorted);
    },

    /**
     * Save metric to the server
     *
     * @param Ibp.KeywordSpotting.Models.Metric metric
     */
    autoSave: function(metric) {
        var _this = this;

        // first determine if you are dealing with an unsaved new model, or an existing model
        if (metric.isNew()) {
            // listen for initial save, then save first change
            this.listenToOnce(metric, 'firstSave', function() {
                _this.savePending = true;
                _this.listenToOnce(metric, 'saved', function() {
                    _this.saveComplete();
                });

                // update the URL to reflect the new metric id
                Backbone.history.navigate("#" + metric.id, false);
            });

            // if the new model is not already being saved, save it
            // this save will trigger the listener above
            if (!this.savePending) {
                this.savePending = true;
                this.collection.autoSave(metric);
            }
        } else {
            if (this.savePending) {
                // The model is currently being saved, wait for successfull save before
                // attempting to save
                this.listenToOnce(metric, 'saved', function() {
                    _this.savePending = true;
                    _this.listenToOnce(metric, 'saved', function() {
                        _this.saveComplete();
                    });
                });
            } else {
                this.savePending = true;
                this.listenToOnce(metric, 'saved', function() {
                    _this.saveComplete();
                });
            }
        }
    },

    saveComplete: function() {
        this.savePending = false;
    },

    /**
     * Get array of selected keyword views (Ibp.KeywordSpotting.Views.KeywordTableRow)
     *
     * @return Array
     */
    getSelected: function() {
        return _.filter(this.rows, function(view) {
            return view.isSelected();
        });
    }
});

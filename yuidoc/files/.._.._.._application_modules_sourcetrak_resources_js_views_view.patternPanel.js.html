/**
 * Routing View
 *
 * @module Sourcetrak
 * @class Views.Wizard
 */
Ibp.Sourcetrak.Views.PatternPanel = Ibp.Backbone.View.extend({
    name: 'PatternPanel',
    template: Ibp.Sourcetrak.Templates.pattern_panel,

    /**
     * Pattern Choices DOM object
     *
     * @type jQuery Object
     */
    $patternChoices: null,

    /**
     * Pattern Choices CSS selector
     *
     * @type String
     */
    patternChoicesEl: '.js-pattern-choices',

    /**
     * Pattern Choices DOM object
     *
     * @type jQuery Object
     */
    $activePatterns: null,

    /**
     * Pattern Choices CSS selector
     *
     * @type String
     */
    activePatternsEl: '.js-active-patterns li',

    /**
     * Model to hold view state
     *
     * @property state
     * @type Model
     */
    state: null,

    /**
     * Define the channels to subscribe
     *
     * @type Object
     */
    subscriptions: {
        'routingState': 'handleRoutingStateChange'
    },

    /**
     * Define the Dom Events to bind to in delegateEvents
     *
     * @type Object
     */
    events: {
        'click .js-add-pattern': 'handleAddPattern',
        'click .js-delete-pattern': 'handleDeletePattern',
        'click .js-active-pattern': 'handleChangeActivePattern'
    },

    /**
     * Init state and render
     *
     * @method init
     */
    init: function (options) {
        // set initial state
        this.state = new Ibp.Carbon.Models.State(options.routingState);

        this.listenTo(this.state, 'change:activePatternId', this.renderActivePattern);

        if (this.model.arePatternsEnabled()) {
            this.render();
        }
    },

    /**
     * handlePatternChange
     *
     * @method handlePatternChange
     */
    handleRoutingStateChange: function (routingState) {
        this.state.set(routingState);
    },

    /**
     * Handle Add Pattern Event
     *
     * @method handleAddPattern
     */
    handleAddPattern: function () {
        var selectedPattern = parseInt(this.$patternChoices.val(), 10);

        if (selectedPattern !== 0) {
            this.model.getRouting().addPattern(selectedPattern);

            this.publish('routingState', this.buildBaseRoutingState(selectedPattern));

            this.render();
        }
    },

    /**
     * Handle Change Active Pattern Event
     *
     * @method handleChangeActivePattern
     */
    handleChangeActivePattern: function (e) {
        var $target = $(e.target),
            patternId = 0;

        if ($target.hasClass('js-active-pattern')) {
            patternId = parseInt($(e.target).data('pattern-id'), 10);

            this.state.set(this.buildBaseRoutingState(patternId));

            this.publish('routingState', this.buildBaseRoutingState(patternId));
        }
    },

    /**
     * Handle Delete Pattern Event
     *
     * @method handleDeletePattern
     */
    handleDeletePattern: function (e) {
        var patternId = parseInt($(e.target).data('pattern-id'), 10),
            routingState = this.buildBaseRoutingState(0);

        this.model.getRouting().deletePattern(patternId);

        this.state.set(routingState, {silent: true});

        this.publish('routingState', routingState);

        this.render();
    },

    buildBaseRoutingState: function (patternId) {
        var firstModeId  = this.model.getRouting().getRouteByPattern(patternId).getConfig().at(0).get('mode_id'),
            schedule     = Ibp.Sourcetrak.Data.Schedules.getScheduleByMode(firstModeId),
            routingState = {
                activePatternId: patternId,
                activeScheduleId: schedule ? schedule.id : 0,
                activeModeId: firstModeId
            };

        return routingState;
    },

    /**
     * Render
     *
     * @method render
     * @return this view
     */
    render: function () {
        var data = {
            availablePatterns: this.getAvailablePatterns(),
            noAvailablePatterns: false,
            activePatterns: []
        };

        data.noAvailablePatterns = !data.availablePatterns.length;
        data.activePatterns = this.getActivePatterns();

        this.$el.html(this.template(data));

        this.$patternChoices = this.$(this.patternChoicesEl);

        this.$activePatterns = this.$(this.activePatternsEl);

        if (Ibp.Utils.isIE()) {
            this.$el.addClass('no-touch');
            this.$('select').selecter({cover: true});
        }

        this.renderActivePattern();

        this.delegateEvents();

        return this;
    },

    /**
     * renderActivePattern
     *
     * @method renderActivePattern
     */
    renderActivePattern: function() {
        var patternEl = 'li[data-pattern-id="' + this.state.get('activePatternId') + '"]';

        this.$activePatterns.removeClass('active');

        this.$(patternEl).addClass('active');
    },

    /**
     * getAvailablePatterns
     *
     * @method getAvailablePatterns
     */
    getAvailablePatterns: function () {
        var patterns = Ibp.Sourcetrak.Data.Patterns.toJSON(),
            routes   = this.model.getRouting().toJSON();

        return _.filter(patterns, function (pattern) {
            var route = _.findWhere(routes, {pattern_id: pattern.id});

            if (route && route.config === null) {
                route = !route;
            }

            return !route;
        });
    },

    /**
     * getActivePatterns
     *
     * @method getActivePatterns
     */
    getActivePatterns: function () {
        var patterns = Ibp.Sourcetrak.Data.Patterns.toJSON(),
            routes   = this.model.getRouting().toJSON();

        return _.filter(patterns, function (pattern) {
            var route = _.findWhere(routes, {pattern_id: pattern.id});

            return (route && (route.config !== null));
        });
    },

    /**
     * onUnrender
     *
     * @method onUnrender
     */
    onUnrender: function() {
        this.undelegateSubscriptions();
    }

});

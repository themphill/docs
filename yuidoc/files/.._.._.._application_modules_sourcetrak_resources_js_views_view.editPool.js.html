Ibp.Sourcetrak.Views.EditPool = Ibp.Carbon.Views.ModalPageView.extend({
    mixins: ['Autosave', 'HighlightInvalid'],
    name: 'editPool',
    el: '#pools-edit',
    template: Ibp.Sourcetrak.Templates.edit_pool,

    /**
     * Backbone Model to hold view state
     *
     * @property state
     * @type Model
     */
    state: null,

    /**
     * The views object retains references to children views
     *
     * @property views
     * @type Object
     */
    views: null,

    /**
     * CSS selector for the area codes popover container
     *
     * @property areaCodesEl
     * @type String
     */
    areaCodesEl: '.js-area-codes',

    /**
     * CSS selector for the routing summary popopver container
     *
     * @property routingSummaryEl
     * @type String
     */
    routingSummaryEl: '.js-routing-summary',

    /**
     * Selector for the description container
     *
     * @property descriptionEl
     * @type String
     */
    descriptionEl: '.js-edit-pool-description',

    /**
     * Selector for the pool activities
     *
     * @property editModelEl
     * @type String
     */
    activitiesEl: '.js-pool-activities',

    /**
     * Max number of characters before we have to render a popover for
     * the routing summary
     *
     * @property ROUTING_MAX_CHARS
     * @type Number
     */
    ROUTING_MAX_CHARS: 35,

    /**
     * Max number of characters before we have to render a popover for the
     * area codes list
     *
     * @property AREA_CODES_MAX_CHARS
     * @type Number
     */
    AREA_CODES_MAX_CHARS: 20,

    /**
     * Init
     *
     * @method init
     */
    init: function () {
        _.bindAll(this);

        // determine whether we need popovers for area codes or the routing summary
        this.state = new Ibp.Carbon.Models.State({
            truncateAreaCodes: this.getAreaCodes().join('').length > this.AREA_CODES_MAX_CHARS,
            truncateRoutingSummary: this.getRoutingSummary().length > this.ROUTING_MAX_CHARS
        });

        this.views = {};

        // rerender every time the model is saved to the server
        this.listenTo(this.model, 'sync', this.render);
    },

    /**
     * Render
     *
     * @method render
     * @return this view
     */
    render: function () {
        var data = {
            routing: this.getRoutingSummaryString(),
            size: this.model.get('pool_size'),
            oneToOne: this.model.get('one_to_one')
        };

        if (this.model.get('one_to_one')) {
            data.number = this.getOtoNumber();
        } else {
            data.areaCodes = this.getAreaCodesString();
        }

        this.$el.removeClass('hidden');
        this.$el.html(this.template(data));

        if (this.state.get('truncateAreaCodes')) {
            this.renderAreaCodesPopover();
        }

        if (this.state.get('truncateRoutingSummary')) {
            this.renderRoutingSummaryPopover();
        }

        this.renderDescription();
        this.renderToolbar();
        this.renderActivities();

        this.$('i[title]').tooltip({placement: 'right'});

        this.delegateEvents();

        return this;
    },

    /**
     * Renders the area codes popover
     *
     * @method renderAreaCodesPopover
     */
    renderAreaCodesPopover: function () {
        this.views.areaCodesPopover = new Ibp.Sourcetrak.Views.Popover({
            $container: this.$(this.areaCodesEl),
            popoverTemplateData: {data: this.getAreaCodes()},
            popoverTemplate: Ibp.Sourcetrak.Templates.popover_area_codes,
            title: 'Area Codes'
        });
    },

    /**
     * Renders the routing summary popover
     *
     * @method renderRoutingSummaryPopover
     */
    renderRoutingSummaryPopover: function () {
        this.views.routingSummaryPopover = new Ibp.Sourcetrak.Views.Popover({
            $container: this.$(this.routingSummaryEl),
            popoverTemplateData: {data: this.getRoutingSummary()},
            popoverTemplate: Ibp.Sourcetrak.Templates.popover_routing_summary,
            title: 'Routing'
        });
    },

    /**
     * Renders the editable lable for description
     *
     * @method renderDescription
     */
    renderDescription: function () {
        if (this.views.name) {
            this.views.name.unrender();
        }

        this.views.name = new Ibp.Carbon.Views.EditableLabel({
            $container: this.$(this.descriptionEl),
            labelAttribute: 'description',
            model: this.model
        });
    },

    /**
     * Renders toolbar
     *
     * @method renderToolbar
     */
    renderToolbar: function () {
        if (this.views.toolbar) {
            this.stopListening(this.views.toolbar);
            this.views.toolbar.unrender();
        }

        this.views.toolbar = new Ibp.Sourcetrak.Views.EditPoolToolbar();

        this.listenTo(this.views.toolbar, 'edit', this.handleEditClick);
        this.listenTo(this.views.toolbar, 'delete', this.handleDeleteClick);
    },

    /**
     * Renders pool's activities
     *
     * @method renderActivities
     */
    renderActivities: function () {
        if (this.views.activities) {
            this.views.activities.unrender();
        }

        this.views.activities = new Ibp.Sourcetrak.Views.PoolActivities({
            $container: this.$(this.activitiesEl),
            model: this.model
        });
    },

    /**
     * Opens the wizard in 'edit mode'
     *
     * @method renderWizard
     */
    renderWizard: function () {
        this.publish('navigate', 'pools/' + this.model.id + '/edit');
    },

    /**
     * Click handler for the edit button
     *
     * @method handleEditClick
     */
    handleEditClick: function () {
        this.renderWizard();
    },

    /**
     * Click handler for the delete button
     *
     * @method handleDeleteClick
     */
    handleDeleteClick: function () {
        if (this.model.getActivities().length) {
            this.views.preventDelete = new Ibp.Sourcetrak.Views.PoolError({
                error: 'activities',
                model: this.model,
                $container: this.$el
            });
        }
        else {
            this.renderDeletePoolConfirmation();
        }
    },

    /**
     * Renders the view that asks the user to verify they want to delete
     * the pool
     *
     * @method renderDeletePoolConfirmation
     */
    renderDeletePoolConfirmation: function () {
        this.views.confirmDelete = new Ibp.Sourcetrak.Views.ConfirmDeletePool({
            model: this.model,
            $container: this.$el
        });
    },

    /**
     * Gets the formatted one to one number
     *
     * @method getOtoNumber
     */
    getOtoNumber: function () {
        var numbers = this.model.getPhoneNumbers(),
            phone = numbers.length > 0 ? numbers.pop() : '';

        return Ibp.Utils.PhoneNumber().formatNumber(phone,'-').number;
    },

    /**
     * Retrieves routing summary for pool
     *
     * @method getRoutingSummary
     * @return String
     */
    getRoutingSummary: function () {
        return this.model.get('routing_summary');
    },

    /**
     * Gets the array of this pool's area codes
     *
     * @method getAreaCodes
     * @return Array
     */
    getAreaCodes: function () {
        return this.model.getAreaCodes().pluck('area_code');
    },

     /**
     * Based on the ROUTING_MAX_CHARS, returns a shortened string for the popover.
     * The entire string will be shown in the popover on hover.
     *
     * @method getTruncatedRoutingSummary
     * @return String
     */
    getTruncatedRoutingSummary: function () {
        // Retrieve truncated string
        var truncatedRoutingSummary = Ibp.Utils.truncate(this.getRoutingSummary(), this.ROUTING_MAX_CHARS);

        // Remove ellipsis retured with string (we have to keep the existing ellipsis
        // because it triggers the popover on the hover event), and has it's own class
        return truncatedRoutingSummary.slice(0, -3);
    },

    /**
     * Returns a string based off the concatenated
     * array of this pool's area codes. The string is truncated based on AREA_CODES_MAX_CHARS.
     * The entire list will be shown in the popover on hover.
     *
     * @method getTruncatedAreaCodes
     * @return areaCodes String
     */
    getTruncatedAreaCodes: function () {
        var areaCodes = _.clone(this.getAreaCodes()),
            areaCodesLength,
            remainingChars,
            lastAreaCode,
            lastAreaCodeLength;

        // if we're still over the limit without the last domain, then we can just drop it
        while (this.getAreaCodesLength(areaCodes.slice(0, -1)) > this.AREA_CODES_MAX_CHARS) {
            areaCodes.pop();
        }

        // set values with the whittled down array
        areaCodesLength    = this.getAreaCodesLength(areaCodes);
        remainingChars     = areaCodesLength - this.AREA_CODES_MAX_CHARS;
        lastAreaCode       = _.last(areaCodes);
        lastAreaCodeLength = lastAreaCode.length;

        // truncate the last item
        areaCodes[areaCodes.length - 1] = lastAreaCode.slice(0, lastAreaCodeLength - remainingChars);

        return areaCodes;
    },

    /**
     * Creates and returns the appropriate routing summary text based on
     * the boolean value for whether a truncated routing summary is needed.
     *
     * @method getRoutingSummaryString
     * @return routingSummary String
     */
    getRoutingSummaryString: function () {
        var routingSummary = this.state.get('truncateRoutingSummary')
                    ? this.getTruncatedRoutingSummary()
                    : this.getRoutingSummary();

        return routingSummary;
    },

    /**
     * Gets the area codes as a concatenated string
     *
     * @method getAreaCodesString
     * @return areaCodeString String
     */
    getAreaCodesString: function () {
        var areaCodes = this.state.get('truncateAreaCodes')
                    ? this.getTruncatedAreaCodes()
                    : this.getAreaCodes(),
            areaCodeString = areaCodes.join(',&nbsp;');

        return areaCodeString;
    },

    /**
     * Gets the length of the area codes subCollection
     *
     * @method getAreaCodesLength
     * @return areaCodes String
     */
    getAreaCodesLength: function (areaCodes) {
        return areaCodes.join(',').length;
    }
});

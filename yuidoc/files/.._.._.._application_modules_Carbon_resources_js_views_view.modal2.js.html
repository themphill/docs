/**
 * A better modal than the first
 *
 * This class defines init, render, and unrender in a way that is dramatically better than the
 * implementation in Ibp.Carbon.Views.Modal. That class is deprecated. Don't use it.
 *
 * Views that inherit from this one only need to implement the
 * onInit(), onRender(), and onUnrender() hooks to add additional functionality.
 * The modal's parent view should then call show() to display the modal.
 * The modal's template must include the root modal element.
 *
 * @module Carbon
 * @class Views.Modal2
 */
Ibp.Carbon.Views.Modal2 = Ibp.Backbone.View.extend({
    name: 'modal2',
    template: null,

    /**
     * Model to hold view state
     *
     * @property state
     * @type Model
     */
    state: null,

    /**
     * Data to be passed into the modal template
     *
     * @property templateData
     * @type Object
     */
    templateData: null,

    /**
     * Flag for allowing the modal to be closed using the escape key
     *
     * @property keyboard
     * @type Boolean
     */
    keyboard: true,

    /**
     * Flag for allowing the modal to be closed by clicking the backdrop
     * true/false toggles the backdrop. 'static' disables click to close
     *
     * @property backdrop
     * @type Boolean
     */
    backdrop: true,

    /**
     * The Modal2 event handlers
     *
     * @property baseModalEvents
     * @type Object
     */
    baseModalEvents: {
        'click #modal-confirm': 'confirm',
        'click #modal-cancel, .js-modal-close': 'cancel',
        'hidden': 'hidden'
    },

    /**
     * The child modal event handlers
     *
     * @property modalEvents
     * @type Object
     */
    modalEvents: {},

    /**
     * Container for notifications
     *
     * @property notificationsEl
     * @type String
     */
    notificationsEl: '#notifications',

    /**
     * The function that merges all the events to be bound at instantiation
     *
     * @property events
     * @type Function
     */
    events: function () {
        return _.extend(this.baseModalEvents, this.modalEvents);
    },

    /**
     * Eventhub subscriptions handlers
     *
     * @property subscriptions
     */
    subscriptions: {
        'notification:modal:error': 'notify'
    },

    /**
     * Refactored init function from Ibp.Carbon.Views.Modal
     * Stores
     *
     * @method init
     * @params options {Object} Config object
     * @params options.template {Function} Template function. Required if not defined in the view.
     * @params options.templateData {Object} Data to pass to template function. Defaults to empty object.
     * @params options.confirmHandler {Function} 'confirm' button click handler. Defaults to a no-op.
     * @params options.keyboard {Boolean} Allow the user to exit the modal by hitting 'Escape'
     * @params options.backdrop {Boolean|String} Set to 'static' to prevent closing the modal by clicking the backdrop
     */
    init : function (options) {
        this.template = options.template || this.template;
        this.templateData = options.templateData || {};
        this.confirm  = options.handler || this.confirm;
        this.keyboard = options.hasOwnProperty('keyboard') ? options.keyboard : this.keyboard;
        this.backdrop = options.hasOwnProperty('backdrop') ? options.backdrop : this.backdrop;

        this.state = new Ibp.Carbon.Models.State({
            backdropActive: true
        });

        this.onInit(options);

        this.render(options);
    },

    /**
     * Handles removing existing modal, rendering this modal, and calling its hook function
     *
     * @method render
     * @return this view
     */
    render: function (options) {
        // create modal fragment
        this.$el = $(this.template(this.templateData));

        // add this modal to the DOM
        $('body').append(this.$el);

        // create the modal, but don't display until show() is called
        this.$el.modal({
            keyboard: this.keyboard,
            backdrop: this.backdrop
        });

        // allow the backdrop options to be changed
        Ibp.Bootstrap.patchBackdrop(this.$el);

        this.captureNotifications();

        this.onRender(options);

        return this;
    },

    /**
     * Destroys view
     *
     * @method unrender
     */
    unrender: function () {
        this.onUnrender();

        this.returnNotifications();

        this.undelegateEvents();
        this.stopListening();
        this.hide();
        this.$el.remove();
    },

    /**
     * Cancel function is used by both cancel click and hide events. Passes the DOM event and passes true for the
     * navigate param for unrender.
     *
     * @method cancel
     * @params e {Object} DOM Event
     */
    cancel: function () {
        this.publish('modal:unrender');
    },

    /**
     * The confirm button click handler
     *
     * @method confirm
     */
    confirm: function () {},

    /**
     * Handler for the 'hidden' event
     *
     * @method hidden
     * @param e {Event} The event data
     */
    hidden: function (e) {
        // make sure we're only listening to hide events from this modal
        if (e && e.target.id !== this.$el.attr('id')) {
            return;
        }

        this.cancel();
    },

    /**
     * Shows notification
     *
     * @method notify
     * @param msg {String} The message to display
     */
    notify: function (msg) {
        $('#notification').html(msg);
    },

    /**
     * Captures the notifications element that would otherwise be hidden
     * behind the modal, and places it within the modal.
     *
     * @method captureNotifications
     */
    captureNotifications: function () {
        $('.modal').prepend($(this.notificationsEl));
    },

    /**
     * Places the notifications element back into its original container
     *
     * @method returnNotifications
     */
    returnNotifications: function () {
        $('#app-main').append($(this.notificationsEl));
    },

    /**
     * Shows modal
     *
     * @method show
     */
    show: function () {
        this.$el.modal('show');
    },

    /**
     * Hides modal
     *
     * @method hide
     */
    hide: function () {
        this.$el.modal('hide');
    },

    /**
     * Init hook
     *
     * @method onInit
     */
    onInit: function () {},

    /**
     * Render hook
     *
     * @method onRender
     */
    onRender: function () {},

    /**
     * Unrender hook
     *
     * @method onUnrender
     */
    onUnrender: function () {},

    /**
     * Changes the backdrop settings to given option
     * This method only works after onRender() has been called and the view's $el has the modal data
     * Only supports disabling/enabling click-to-close. Animation will remain the same as it was at creation.
     *
     * @method updateBackdrop
     * @param option {String|Boolean}
     */
    updateBackdrop: function () {
        var option = this.state.get('backdropActive') ? true : 'static';

        this.$el.data('modal').updateBackdrop(option);
    }
});

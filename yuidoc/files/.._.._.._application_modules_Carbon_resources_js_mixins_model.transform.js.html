/**
 * Mixin for Models that allows them to cast values being set on them to the appropriate type
 * Its primary purpose is to handle converting string values from HTML form inputs
 *
 * Transforms are done by matching the property being set on the model to
 * to a transform method specified in the model's 'transforms' object
 *
 * Transform relies on a modified version of DeepModel.set() and only works with DeepModels
 *
 * @module Carbon
 * @class Mixins.Transform
 */
;(function (_) {
    /**
     * List of tranform methods
     *
     * @property transformers
     * @type Object
     */
    var transformers = {
        /**
         * Casts value to Boolean
         * Casts the strings 'true' and 'false' to Boolean true and false
         *
         * @method bool
         * @param val {Mixed} The value to be transformed
         */
        bool: function (val) {
            // don't need to do anything
            if (_.isBoolean(val)) {
                return val;
            }

            // handle 'true' and 'false'
            if (_.isString(val)) {
                if (val.toLowerCase() === 'true') {
                    return true;
                }

                if (val.toLowerCase() === 'false') {
                    return false;
                }
            }

            // let JS take care of it
            return Boolean(val);
        },

        /**
         * Casts value to Number
         *
         * @method number
         * @param val {Mixed} The value to be transformed
         */
        number: function (val) {
            // don't need to do anything
            if (_.isNumber(val)) {
                return val;
            }

            // let JS take care of it
            return parseInt(val, 10);
        },

        /**
         * Casts value to String
         *
         * @method string
         * @param val {Mixed} The value to be transformed
         */
        string: function (val) {
            // don't need to do anything
            if (_.isString(val)) {
                return val;
            }

            // if the value has its own string representation
            if (val.hasOwnProperty('toString')) {
                return val.toString();
            }

            // let JS handle it
            return String(val);
        }
    };

    Ibp.Carbon.Mixins.Transform = {
        /**
         * Takes the value(s) being set on the model and casts them to the specified data type(s)
         * If set() is called as set({key: val}), modifies the object in place, returning 'val' unaltered
         * If set() is called as set(key, val), returns the tranformed 'val'
         *
         * @method transform
         * @param key {Object|String} set()'s first argument. The attributes hash or the attribute name
         * @param val {Mixed|Object} set()'s second argument. The attribute value or the options hash
         */
        transform: function transform(key, val) {
            var transformedVal = val,
                setTransformedVal = false,
                attrs;

            // handle both key,value and {key: value} -style arguments
            if (typeof key === 'object') {
                attrs = key;
            } else {
                (attrs = {})[key] = val;

                // we can't modify primitive data types by reference
                // so we'll pass the transformed value back to be set
                setTransformedVal = true;
            }

            // loop over each attribute and transform its value
            _.each(attrs, function (attrVal, attrKey) {
                if (attrKey in this.transforms) {
                        // get the method that will transform the value from model.transforms
                    var transformer = transformers[this.transforms[attrKey]],
                        // don't attempt to transform 'undefined', 'null', and 'NaN'
                        doTransform = (!_.isUndefined(attrVal) && !_.isNull(attrVal) && !_.isNaN(attrVal));

                    // do the actual value transform
                    attrs[attrKey] = doTransform ? transformer(attrVal) : attrVal;

                    // set the transformed value to be returned
                    if (setTransformedVal) {
                        transformedVal = attrs[attrKey];
                    }
                }
            }.bind(this));

            return transformedVal;
        }
    };
})(_);

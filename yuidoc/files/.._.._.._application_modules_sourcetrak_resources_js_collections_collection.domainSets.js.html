/**
 * Domain Sets Collection
 *
 * @module SourceTrak
 * @class Collections.DomainSets
 * @extends Ibp.Backbone.Collection
 */
Ibp.Sourcetrak.Collections.DomainSets = Ibp.Backbone.Collection.extend({
    mixins: ['ClientPagination'],
    name: 'DomainSets',
    model: Ibp.Sourcetrak.Models.DomainSet,
    url: BootstrappedData.apiUrlBase + 'sourcetrak/domainset',

    init: function () {
        this.listenTo(this, 'save', this.save);
        this.listenTo(this, 'delete', this.destroy);
    },

    /**
     * Saves the given attributes on the model and notifies of the result
     *
     * @method save
     * @param model {Model} The model to save
     * @param changed {Object} The new attributes to save
     */
    save: function (model, changed) {
        var _this = this,
            isNew = model.isNew(),
            attrs = changed || model.changed;

        this.publish('notify', {type: 'saving'});

        model.isSaving = true;

        model.save(attrs, {
            wait: true,
            patch: true,
            validate: true,
            success: function () {
                var notification = {
                    type: 'success',
                    msg: 'Successfully saved ' + model.names[_.first(_.keys(attrs))] + '.'
                };

                model.isSaving = false;

                if (isNew) {
                    _this.add(model);

                    _this.publish('newDomainSet', model);

                    notification.msg = 'Successfully created Domain Set: ' + model.get('name') + '.';
                }

                _this.publish('notify', notification);
            },
            error: function () {
                model.isSaving = false;

                _this.publish('notify', {type: 'error', msg: 'Error saving ' + model.get('name') + '.'});
            }
        });
    },

    /**
     * Delete event handler
     * Destroys the model that triggers the event
     *
     * @method destroy
     * @param model {Model} The model to be destroyed
     */
    destroy: function (model) {
        model.destroy({wait: true});
    },

    /**
     * Gets a new model that isn't bound to the collection
     * The model will be added on create and unbound on remove
     *
     * @method getNew
     * @param attrs {Object} Attributes to set on the new model
     * @return Model
     */
    getNew: function (attrs) {
        var model = new this.model(attrs);

        this.listenTo(model, 'create', this.save);
        this.listenTo(model, 'remove', this.trim);

        return model;
    },

    /**
     * Discards a model from getNew() by removing the listener, allowing it to be garbage collected
     *
     * @method trim
     * @param model {Model} The new model to stop listening to
     */
    trim: function (model) {
        this.stopListening(model);
    },

    /**
     * Returns the models that match the query
     * Filters on name and domains
     *
     * @method search
     * @param query {Object} The attributes to filter on
     * @return Array
     */
    search: function (query) {
        var _this = this;

        if (query) {
            query = this.generateMatchableString(query);

            return _.filter(this.models, function (model) {
                var domains = model.getDomains().pluck('domain').join(','),
                    name = model.get('name'),
                    searchString = _this.generateMatchableString(domains + ',' + name);

                if (searchString.indexOf(query) !== -1) {
                    return true;
                }

                return false;
            });
        }

        // return the whole collection
        return this.models;
    },

    /**
     * Returns a lowercased string without spaces, hyphens, or parentheses
     * to facilitate matching despite how the user inputs their query.
     *
     * @method generateMatchableString
     * @param string {String} The string to replace characters for
     * @return String
     */
    generateMatchableString: function (string) {
        return string.replace(/-|\(|\)|\s/g, "").toLowerCase();
    },

    /**
     * Backbone method for sorting models
     * Sorts by updated date descending
     *
     * @method comparator
     */
    comparator: function (modelA, modelB) {
        var a = new Date(modelA.get('updated_date').replace(/\-/g,'\/').replace(/[T|Z]/g,' ')),
            b = new Date(modelB.get('updated_date').replace(/\-/g,'\/').replace(/[T|Z]/g,' '));

        return (a > b) ? -1 : ((a < b) ? 1 : 0);
    }
});

/**
 * A library of common validation functions
 *
 * email, url, and len are courtesy of https://github.com/chriso/node-validator
 * phone number regex courtesy of Steven Levithan, Regular Expressions Cookbook
 *
 * @module Carbon
 * @class Validator
 */

;(function() {
    'use strict';

    Backbone.Validator = {
        /**
         * Checks if the value is Boolean
         *
         * @method bool
         */
        bool: function (str) {
            return _.isBoolean(str);
        },

        /**
         * Checks if the value is Number
         * Optionally accepts min and max values to test if the value in within a range
         *
         * @method num
         */
        num: function (str, min, max) {
            var hasMin,
                hasMax;

            str = parseInt(str, 10);
            min = parseInt(min, 10);
            max = parseInt(max, 10);

            hasMin = !_.isNaN(min);
            hasMax = !_.isNaN(max);

            if (_.isNaN(str)) {
                return false;
            }
            if (hasMin && str < min) {
                return false;
            }
            if (hasMax && str > max) {
                return false;
            }

            return true;
        },

        /**
         * Checks if the value is String
         *
         * @method string
         */
        string: function (str) {
            return _.isString(str);
        },

        /**
         * Checks if the value contains only alpha-numeric characters
         *
         * @method alphaNumeric
         */
        alphaNumeric: function (str) {
            str = String(str);

            return (/^[a-zA-Z0-9]+$/).test(str);
        },

        /**
         * Checks if the value contains only alpha-numeric characters, but allows spaces
         *
         * @method alNumSpaces
         */
        alNumSpaces: function (str) {
            str = String(str);

            return (/^[a-zA-Z0-9\s]+$/).test(str);
        },

        /**
         * Checks if the value is a phone number
         *
         * @method phone
         */
        phone: function (str) {
            str = String(str);

            if ((str.slice(0, 3) === '555') || (str.slice(0, 4) === '1555')) {
                return true;
            }
            if (str.slice(0, 3) === '211') {
                return true;
            }
            if (Ibp.Utils.PhoneNumber) {
                return Ibp.Utils.PhoneNumber().isValidNumber(str);
            }

            return (/^\(?([0-9]{3})\)?[-. ]?([0-9]{3})[-. ]?([0-9]{4})$/).test(str);
        },

        /**
         * Checks if the value is email address
         *
         * @method email
         */
        email: function (str) {
            str = String(str);

            return (/^(?:[\w\!\#\$\%\&\'\*\+\-\/\=\?\^\`\{\|\}\~]+\.)*[\w\!\#\$\%\&\'\*\+\-\/\=\?\^\`\{\|\}\~]+@(?:(?:(?:[a-zA-Z0-9](?:[a-zA-Z0-9\-](?!\.)){0,61}[a-zA-Z0-9]?\.)+[a-zA-Z0-9](?:[a-zA-Z0-9\-](?!$)){0,61}[a-zA-Z0-9]?)|(?:\[(?:(?:[01]?\d{1,2}|2[0-4]\d|25[0-5])\.){3}(?:[01]?\d{1,2}|2[0-4]\d|25[0-5])\]))$/).test(str);
        },

        /**
         * Checks if the value is
         *
         * @method url
         */
        url: function (str) {
            str = String(str);

            return (/^(?!mailto:)(?:(?:https?|ftp):\/\/)?(?:\S+(?::\S*)?@)?(?:(?:(?:[1-9]\d?|1\d\d|2[01]\d|22[0-3])(?:\.(?:1?\d{1,2}|2[0-4]\d|25[0-5])){2}(?:\.(?:[1-9]\d?|1\d\d|2[0-4]\d|25[0-4]))|(?:(?:[a-z\u00a1-\uffff0-9]+-?)*[a-z\u00a1-\uffff0-9]+)(?:\.(?:[a-z\u00a1-\uffff0-9]+-?)*[a-z\u00a1-\uffff0-9]+)*(?:\.(?:[a-z\u00a1-\uffff]{2,})))|localhost)(?::\d{2,5})?(?:\/[^\s]*)?$/i.test(str) || str.length > 2083);
        },

        /**
         * Checks if the value is of a certain length
         *
         * @method len
         */
        len: function (str, min, max) {
            str = String(str);

            if (str.length < min) {
                return false;
            }
            if (max && str.length > max) {
                return false;
            }

            return true;
        },

        /**
         * Checks if the value matches any of the given terms
         *
         * @method match
         */
        match: function () {
            var args = [].slice.call(arguments),
                // get first arg as string to be matched and use the rest as matchers
                val = args.shift(),
                matches = 0,
                pattern;

            // discard the context object at the end
            args.pop();

            _.each(args, function (arg) {
                pattern = new RegExp(arg);

                if (pattern.test(val)) {
                    matches++;
                }
            });

            return !!matches;
        },

        /**
         * Checks if the value is present
         *
         * @method required
         */
        required: function (str) {
            // correctly handle strings that are just spaces
            str = _.isString(str) ? str.trim() : str;

            // NaN is never a valid value
            if (_.isNaN(str)) {
                return false;
            }

            // all numbers are valid values
            if (_.isNumber(str)) {
                return true;
            }

            return !_.isEmpty(str);
        },

        /**
         * Checks if the value is not present
         *
         * @method isEmpty
         */
        isEmpty: function (str) {
            // correctly handle strings that are just spaces
            str = _.isString(str) ? str.trim() : str;

            // NaN is never a valid value
            if (_.isNaN(str)) {
                return false;
            }

            // numbers are never considered "empty"
            if (_.isNumber(str)) {
                return false;
            }

            return _.isEmpty(str);
        },

        /**
         * Checks if the value is just the host part of a URI
         *
         * @method host
         */
        host: function (str) {
            // make sure the url is valid
            if (!Backbone.Validator.url(str)) {
                return false;
            }

            var url = purl(str);

            // make sure the url is just the host
            return url.attr('source') === url.attr('host');
        }
    };
})();

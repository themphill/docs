/**
 * Main entry point for Sourcetrak
 * Loads Notifications, Dashboard, and Pool Management Toolbar ("Poolbar")
 *
 * @module Sourcetrak
 * @class Views.Main
 */
Ibp.Sourcetrak.Views.Main = Ibp.Backbone.View.extend({
    name: 'main',
    el: '#app-main',
    template: Ibp.Sourcetrak.Templates.main,

    /**
     * Model to hold view state
     *
     * @property state
     * @type Model
     */
    state: null,

    /**
     * The views object retains references to children views
     *
     * @property views
     * @property.views.notifications {Object} Notifications View
     * @property.views.poolbar {Object} Poolbar View
     * @property.views.dashboard {Object} Dashboard View
     * @property.views.editDomainSet {Object} Edit Domain Set View
     * @type Object
     */
    views: {},

    /**
     * List of version hash error codes to display the error modal for
     *
     * @property apiErrors
     * @type Array
     */
    apiErrors: [
        1100,
        1200,
        1300
    ],

    subscriptions: {
        'apiError': 'handleError'
    },

    /**
     * Init notifications and render
     *
     * @method init
     */
    init: function () {
        // set initial state
        this.state = new Ibp.Carbon.Models.State({
            // whether the Dashboard view is rendered
            isDashboardRendered: false,
            // whether the Edit Domain Set view is rendered
            isEditDomainSetRendered: false,
            // whether this is first time we're rendering the app
            isInitialRender: true
        });

        // render app header and container to page
        this.render();
    },

    /**
     * Renders poolbar and dashboard
     *
     * @method render
     * @return this view
     */
    render: function () {
        this.$el.append(this.template());

        this.renderNotifications();

        this.renderPoolbar();

        this.renderModalManager();

        return this;
    },

    /**
     * Initializes the Notifications Manager
     *
     * @method renderNotifications
     */
    renderNotifications: function () {
        this.views.notifications = new Ibp.Carbon.Views.NotificationManager();
    },

    /**
     * Renders the Modal Manager
     *
     * @method renderModalManager
     */
    renderModalManager: function () {
        this.views.modalManager = new Ibp.Sourcetrak.Views.ModalManager();
    },

    /**
     * Renders poolbar
     *
     * @method renderPoolbar
     */
    renderPoolbar: function () {
        this.views.poolbar = new Ibp.Sourcetrak.Views.Poolbar({
            collection: Ibp.Sourcetrak.Data.Pools
        });
    },

    /**
     * Renders dashboard
     *
     * @method renderDashboard
     */
    renderDashboard: function () {
        if (!this.views.dashboard) {
            this.views.dashboard = new Ibp.Sourcetrak.Views.Dashboard({
                collection: Ibp.Sourcetrak.Data.DomainSets
            });
        }

        if (this.views.editDomainSet && this.state.get('isEditDomainSetRendered')) {
            this.views.editDomainSet.hide();
            this.state.toggle('isEditDomainSetRendered');
        }

        if (!this.state.get('isDashboardRendered')) {
            this.state.toggle('isDashboardRendered');

            // only animate the view for transitions, i.e. not the initial app load
            if (this.state.get('isInitialRender')) {
                this.state.toggle('isInitialRender');
            }
            else {
                this.views.dashboard.show();
            }
        }
    },

    /**
     * Renders edit domain set view
     *
     * @method renderEditDomainSet
     */
    renderEditDomainSet: function (domainSetId) {
        var domainSet = (domainSetId === null)
                      ? Ibp.Sourcetrak.Data.DomainSets.getNew()
                      : Ibp.Sourcetrak.Data.DomainSets.get(domainSetId);

        domainSetId = (domainSetId === null) ? null : parseInt(domainSetId, 10);

        if (!this.views.editDomainSet) {
            this.views.editDomainSet = new Ibp.Sourcetrak.Views.EditDomainSet({
                model: domainSet
            });
        }
        else if (this.views.editDomainSet.model.id !== domainSetId) {
            this.views.editDomainSet.remove();
            this.views.editDomainSet = new Ibp.Sourcetrak.Views.EditDomainSet({
                model: domainSet
            });
        }

        if (this.views.dashboard && this.state.get('isDashboardRendered')) {
            this.views.dashboard.hide();
            this.state.toggle('isDashboardRendered');
        }

        if (!this.state.get('isEditDomainSetRendered')) {
            this.state.toggle('isEditDomainSetRendered');

            // only animate the view for transitions, i.e. not the initial app load
            if (this.state.get('isInitialRender')) {
                this.state.toggle('isInitialRender');
            }
            else {
                this.views.editDomainSet.show();
            }

        }
    },

    /**
     * Display modal for version hash errors
     *
     * @method handleError
     */
    handleError: function (error) {
        if (_.contains(this.apiErrors, error.code)) {
            this.publish('modal:render', {name: 'versionHashError'});
        }
    }
});

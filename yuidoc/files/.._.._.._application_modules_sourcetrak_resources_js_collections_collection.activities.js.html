/**
 * Activity Collection
 *
 * @module SourceTrak
 * @class Collections.Activities
 * @extends Ibp.Backbone.Collection
 */
Ibp.Sourcetrak.Collections.Activities = Ibp.Backbone.Collection.extend({
    name: 'Activities',
    model: Ibp.Sourcetrak.Models.Activity,
    url: BootstrappedData.apiUrlBase + 'sourcetrak/activity',

    /**
     * List of API errors to display messages for
     *
     * @property apiErrors
     * @type Array
     */
    apiErrors: [
        1201,
        1202
    ],

    subscriptions: {
        'newDomainSet': 'handleUpdateDomainSets',
        'apiError': 'handleError'
    },

    init: function () {
        this.listenTo(this, 'save', this.save);
        this.listenTo(this, 'delete', this.destroy);

        // Update the activities whenever a domain set is deleted
        this.listenTo(Ibp.Sourcetrak.Data.DomainSets, 'destroy', this.handleUpdateDomainSets);
    },

    /**
     * Saves the given attributes on the model and notifies of the result
     * New models will be added to the collection once saved
     *
     * @method save
     * @param model {Model} The model to save
     * @param changed {Object} The new attributes to save
     */
    save: function (model, changed) {
        var _this  = this,
            isNew = model.isNew(),
            attrs = changed || model.changed,
            isDefaultActivity = model.isMainNumberActivity();

        this.publish('notify', {type: 'saving'});

        model.isSaving = true;

        model.save(attrs, {
            wait: true,
            patch: true,
            validate: !isDefaultActivity,
            success: function() {
                var attrName = attrs.keyword ? model.getActivityType().label : model.names[_.first(_.keys(attrs))],
                    notification = {
                        type: 'success'
                    };

                notification.msg = isDefaultActivity
                                 ? 'Successfully saved Advanced Options.'
                                 : 'Successfully saved the Activity\'s ' + attrName + '.';

                model.isSaving = false;

                if (isNew) {
                    _this.add(model);

                    notification.msg = 'Successfully created Activity: ' + model.getActivityType().description + '.';
                }

                _this.publish('notify', notification);
            },
            error: function () {
                var message = isDefaultActivity
                            ? 'Error saving Advanced Options.'
                            : 'Error saving Activity: ' + model.getActivityType().description + '.';

                model.isSaving = false;

                _this.publish('notify', {
                    type: 'error',
                    msg: message
                });
            },
            always: function () {
                _this.sort();
            }
        });
    },

    /**
     * Delete event handler
     * Destroys the model that triggers the event
     *
     * @method destroy
     * @param model {Model} The model to be destroyed
     */
    destroy: function (model) {
        model.destroy({wait: true});
    },

    /**
     * Returns a list of models that match a list of attributes as key/value pairs
     * Basically, a slightly different implementation of Collection.where()
     *
     * @method find
     * @return Array
     */
    find: function (queries) {
        // return the whole collection
        if (!queries || _.isEmpty(queries)) {
            return this.models;
        }

        // return the filtered collection
        return _.filter(this.models, function (model) {
            // array to hold query results
            var match = [];

            // run each query
            _.each(queries, function (value, key) {
                // for strings, we see if the search term occurs in the string
                if (typeof model.get(key) === 'string') {
                    value = value.toLowerCase();

                    match.push(model.get(key).toLowerCase().indexOf(value) !== -1);

                    return;
                }

                // for everything else, we test by equality
                match.push(model.get(key) == value);
            });

            // if any of our queries returned false,
            // this model is not a match
            return !_.contains(match, false);
        });
    },

    /**
     * Gets a new model that isn't bound to the collection
     * The model will be added on create and unbound on remove
     *
     * @method getNew
     * @param attrs {Object} Attributes to set on the new model
     * @return Model
     */
    getNew: function (attrs) {
        var model = new this.model(attrs);

        // set the collection now so it takes precedence if the model is added another collection before it is saved
        model.collection = this;

        this.listenTo(model, 'create', this.save);
        this.listenTo(model, 'remove', this.trim);

        return model;
    },

    /**
     * Discards a model from getNew() by removing the listener, allowing it to be garbage collected
     *
     * @method trim
     * @param model {Model} The new model to stop listening to
     */
    trim: function (model) {
        this.stopListening(model);
    },

    /**
     * Refreshes activities collection whenever a domain set is added or removed
     *
     * @method handleUpdateDomainSets
     */
    handleUpdateDomainSets: function () {
        // use 'reset' to make sure activities that have been deleted are removed from the collection
        this.fetch({reset: true});
    },

    /**
     * Handler for API errors
     * Essentially, out-of-band validation for problems only the server catches
     *
     * @method handleError
     */
    handleError: function (error, model) {
        if (_.contains(this.apiErrors, error.code)) {
            this.publish('notify', {type: 'error', msg: error.message});
        }

        // reset the model saving flag so future updates work
        model.isSaving = false;
    }
});

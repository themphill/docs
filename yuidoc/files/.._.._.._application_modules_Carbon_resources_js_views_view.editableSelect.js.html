Ibp.Carbon.Views.EditableSelect = Ibp.Backbone.View.extend({
    name: 'editableSelect',
    template: Ibp.Carbon.Templates.editable_select,

    /**
     * jQuery reference for the select container
     *
     * @property $container
     * @type jQuery Object
     */
    $container: null,

    /**
     * The name of the model attribute to use as the select
     *
     * @property selectAttribute
     * @type String
     */
    selectAttribute: null,

    /**
     * The length at which to truncate the select for truncatedSelect
     *
     * @property truncatedAt
     * @type Number
     */
    truncatedAt: null,

    /**
     * Selector for the select wrapper
     *
     * @property selectTextEl
     * @type String
     */
    selectTextEl: '.js-select-wrapper',

    /**
     * jQuery reference for the select wrapper
     *
     * @property $select
     * @type jQuery Object
     */
    $selectText: null,

    /**
     * Selector for the select input
     *
     * @property selectInputEl
     * @type String
     */
    selectInputEl: '.js-select-input',

    /**
     * jQuery reference for the select input
     *
     * @property $selectInput
     * @type jQuery Object
     */
    $selectInput: null,

    /**
     * DOM Event handlers
     *
     * @property events
     */
    events:  {
        'click': 'showSelectInput',
        'blur select': 'render',
        'keyup': 'handleKeyboard'
    },

    /**
     * Determines on which side of the tooltip object the tooltip should be placed
     *
     * @property tooltipDirection
     * @type String
     */
    tooltipDirection: null,

    /**
     * Init
     *
     * @method init
     */
    init: function (options) {
        _.bindAll(this);

        this.$container        = options.$container;
        this.selectName        = options.selectName;
        this.selectAttribute   = options.selectAttribute;
        this.selectPlaceholder = options.selectPlaceholder || 'Select value...';
        this.choices           = options.choices;
        this.tooltipDirection  = options.tooltipDirection || 'right';

        this.listenTo(this.model, 'change:' + this.selectAttribute, this.handleChange);

        // listen for escape key on document, since they won't type in the select
        $(document).on('keyup', this.handleKeyboard);
        $(document).on('click', this.handleDocumentClick);

        this.onInit();

        this.render();
    },

    onInit: function () {},

    /**
     * Render
     *
     * @method render
     * @return this view
     */
    render: function () {
        var data = {
            attribute:         this.selectAttribute,
            selectName:        this.selectName,
            selectPlaceholder: this.escapeSpaces(this.selectPlaceholder),
            selectedValue:     this.model.get(this.selectAttribute),
            choices:           this.choices,
            namespace:         this.model.name.toLowerCase()
        };

        this.$el = $(this.template(data));
        this.$container.html(this.$el);

        this.$selectText  = this.$(this.selectTextEl);
        this.$selectInput = this.$(this.selectInputEl);

        this.delegateEvents();

        this.$('i[title]').tooltip({placement: this.tooltipDirection});

        return this;
    },

    onUnrender: function () {
        $(document).off('keyup', this.handleKeyboard);
        $(document).off('click', this.handleDocumentClick);
    },

    /**
     * Handles the change event for the model's select attribute
     * Rerenders when the model select attribute changes
     *
     * @method handleChange
     */
    handleChange: function () {
        this.render();
    },

    /**
     * Handles key press events in the label input
     * Rerenders the view if 'escape' is pressed while editing
     *
     * @method handleKeyboardInput
     */
    handleKeyboard: function (e) {
        var keyCode = e.keyCode || e.which;

        // for escape, rerender to hide and reset the select
        if (keyCode === 27) {
            e.stopPropagation();

            this.render();
        }
    },

    /**
     * Handles click events outsdie the select
     * Rerenders when the user clicks outside the select
     *
     * @method handleDocumentClick
     */
    handleDocumentClick: function (e) {
        if (!this.$el.has(e.target).length) {
            this.render();
        }
    },

    /**
     * Shows the edit select input
     *
     * @method showSelectInput
     */
    showSelectInput: function () {
        this.$selectText.addClass('hidden');
        this.$selectInput.removeClass('hidden');
    },

    /**
     * Hides the edit select input
     *
     * @method hideSelectInput
     */
    hideSelectInput: function () {
        this.$selectText.removeClass('hidden');
        this.$selectInput.addClass('hidden');
    },

    /**
     * Replaces breaking spaces with non-breaking spaces
     *
     * @method escapeSpaces
     * @return String
     */
    escapeSpaces: function (str) {
        return str ? str.replace(/\s/g, '&nbsp;') : str;
    }
});

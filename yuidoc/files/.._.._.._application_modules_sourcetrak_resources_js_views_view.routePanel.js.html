/**
 * Routing View
 *
 * @module Sourcetrak
 * @class Views.Wizard
 */
Ibp.Sourcetrak.Views.RoutePanel = Ibp.Backbone.View.extend({
    mixins: ['Autoset', 'AudioFileUploadBtn', 'HighlightInvalid'],
    name: 'RoutePanel',
    template: Ibp.Sourcetrak.Templates.route_panel,

    /**
     * Routing form DOM object
     *
     * @type jQuery Object
     */
    $routingForm: null,

    /**
     * Routing form CSS selector
     *
     * @type String
     */
    routingFormEl: '.js-routing-form',

    /**
     * The subviews for a given application
     * These dependencies defined in the GruntFile
     *
     * @property applicationViews
     * @type Object
     */
    applicationViews: {
        broadcast_from_phone:   Ibp.Sourcetrak.Views.RoutingBroadcastFromPhone,
        queue:                  Ibp.Sourcetrak.Views.RoutingCallDistributor,
        findme:                 Ibp.Sourcetrak.Views.RoutingFindme,
        ivr:                    Ibp.Sourcetrak.Views.RoutingIvr,
        phone_directory:        Ibp.Sourcetrak.Views.RoutingPhoneDirectory,
        read_text:              Ibp.Sourcetrak.Views.RoutingReadText,
        schedule_router:        Ibp.Sourcetrak.Views.RoutingScheduleRouter,
        locator:                Ibp.Sourcetrak.Views.RoutingStoreLocator,
        survey:                 Ibp.Sourcetrak.Views.RoutingSurvey,
        transfer_call:          Ibp.Sourcetrak.Views.RoutingTransfer,
        transfer_w_whisper:     Ibp.Sourcetrak.Views.RoutingTransferWithWhisper,
        virtual_receptionist:   Ibp.Sourcetrak.Views.RoutingVirtualReceptionist,
        opt_out:                Ibp.Sourcetrak.Views.RoutingOptOut,
        vmail_greet_and_record: Ibp.Sourcetrak.Views.RoutingVoicemail,
        play_audio:             Ibp.Sourcetrak.Views.RoutingPlayAudio
    },

    /**
     * The options in pool.routingTypes that belong to 'action' instead of 'add_on_name'
     *
     * @property nonApplicationActions
     * @type Array
     */
    nonApplicationActions: [
        'play_audio',
        'read_text',
        'transfer_call',
        'virtual_receptionist'
    ],

    /**
     * The views object retains references to children views
     *
     * @property views
     * @type Object
     */
    views: {},

    /**
     * Model to hold view state
     *
     * @property state
     * @type Model
     */
    state: null,

    /**
     * Define the channels to subscribe
     *
     * @type Object
     */
    subscriptions: {
        'routingState': 'handleRoutingStateChange'
    },

    /**
     * Define the Dom Events to bind to in delegateEvents
     *
     * @type Object
     */
    events: {
        'change .js-routing-type': 'updateRoutingType'
    },

    /**
     * Init state and render
     *
     * @method init
     */
    init: function (options) {
        this.state = new Ibp.Carbon.Models.State(options.routingState);

        this.render();
    },

    /**
     * Render
     *
     * @method render
     * @return this view
     */
    render: function () {
        var data = {
            routingTypes: this.getAvailableRoutingTypes(),
            currentAppName: this.getRoutingAppName()
        };

        this.$el.html(this.template(data));

        // cache routing form panel
        this.$routingForm = this.$(this.routingFormEl);

        this.renderRoutingForm();

        if (Ibp.Utils.isIE()) {
            this.$el.addClass('no-touch');
            this.$('select').selecter({cover: true});
        }

        this.delegateEvents();
    },

    /**
     * handleRoutingStateChange
     *
     * @method handleRoutingStateChange
     */
    handleRoutingStateChange: function (routingState) {
        this.state.set(routingState);

        this.render();
    },

    /**
     * Get Current Mode Object
     *
     * @method getActiveMode
     * @return SubModel
     */
    getActiveMode: function() {
        return this.model
                   .getRouting()
                   .getRouteByPattern(this.state.get('activePatternId'))
                   .getConfigMode(this.state.get('activeModeId'));
    },

    /**
     * Update Routing Type
     *
     * @method updateRoutingType
     */
    updateRoutingType: function (e) {
        var configModel = this.getActiveMode(),
            newType     = this.$(e.target).val(),
            isValidAppAction  = !(_.contains(this.nonApplicationActions, newType)),
            addOnName   = isValidAppAction ? newType : '',
            action      = isValidAppAction ? 'application_transfer' : newType,
            routing     = {
                add_on_name: addOnName,
                action: action
            };

        // reset routing by removing all of the old routing settings
        _.each(configModel.attributes, function (val, key) {
            // don't remove the id or mode id
            if (key === 'id' || key === 'mode_id') {
                return;
            }

            configModel.unset(key, {silent: true});
        });

        // set new routing settings
        configModel.set(routing);

        this.renderRoutingForm();
    },

    /**
     * Render the correct routing form view
     *
     * @method renderRoutingForm
     */
    renderRoutingForm: function () {
        var ViewName = this.applicationViews[this.getRoutingAppName()];

        if (this.views.appView) {
            this.views.appView.unrender();
        }

        // clear the validation error notification
        this.publish('notify:dismiss');

        if (ViewName !== undefined) {
            this.views.appVew = new ViewName({
                $container: this.$routingForm,
                model: this.getActiveMode()
            });
        }
    },

    /**
     * Get Routing App Name
     *
     * @method getRoutingAppName
     * @return String
     */
    getRoutingAppName: function () {
        var routing = this.model.getRouting(),
            config,
            appName;

        if (routing.models.length === 0) {
            appName = '';
        }
        else {
            config = this.getActiveMode().attributes;

            if (config.action !== 'application_transfer') {
                appName = config.action;
            }
            else {
                appName = config.add_on_name;
            }

            if (appName === 'survey' && config.survey_version === '2') {
                appName = 'ivr';
            }
        }

        return appName;
    },

    /**
     * onUnrender
     *
     * @method onUnrender
     */
    onUnrender: function() {
        this.undelegateSubscriptions();
    },

    /**
     * Compare the pool model's list of routable applications to the permissions based list bootstrapped into the app. Return only allowed routes.
     * @method getAvailableRoutingTypes
     * @return [Array]
     */
    getAvailableRoutingTypes: function () {
        var allTypes = this.model.routingTypes,
            permissionFlags = BootstrappedData.allowedRoutingApplications;

        return _.filter(allTypes, function (type) {
            return permissionFlags[type.value];
        });
    }
});

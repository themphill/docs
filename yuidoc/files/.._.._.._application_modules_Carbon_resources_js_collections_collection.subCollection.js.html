/**
 * SubCollection
 * A general-purpose Collection for dealing with nested Models
 *
 * SubCollections can be assigned to properties of Models, i.e. Model._subCol (they will not be synced)
 * SubCollections contain SubModels and proxy all changes to original Model, while still
 * maintaining the built-in events
 * i.e. If we make Model.domains a SubCollection, you can still listen for changes on 'change:domains'
 *
 * Nested model structure must be: { _id: 1, _body: { key: value } }
 *
 * You should be able to nest SubCollections and SubModels, but this hasn't been tested
 *
 * @module Carbon
 * @class Collections.SubCollection
 * @example See sourcetrak/resources/js/models/model.domainSet.js
 */
Ibp.Backbone.Collection.SubCollection = Ibp.Backbone.Collection.extend({
    name: 'subCollection',
    model: Ibp.Backbone.Model.SubModel,

    /**
     * Creates SubModels and listens for changes
     *
     * @method init
     * @param models {Array} The models to populate the Collection. This is unused and should be ignored.
     * @param options {Object} The references for the parent Model
     * @param options.parent {Model} The reference to the parent Model
     * @param options.colName {Model} The "collection name". The name of the Model attribute this collection is for
     */
    init: function (models, options) {
        _.bindAll(this);

        this.parse = this.deserialize;

        this.parent = options.parent;
        this.colName = options.colName;

        this.resetSubModels();

        this.listenTo(this, 'add change reset', this.setParent);
        this.listenTo(this, 'destroy', this.destroy);

        this.onInit();
    },

    onInit: function () {},

    /**
     * Converts the parent's nested models into SubModels
     *
     * @method deserialize
     * @param parentCollection {Array} The value of parent Model attribute containing the nested models
     * @return {Array} The array of SubModels
     */
    deserialize: function (parentCollection) {
        // copy and flatten each nested model and create a SubModel
        return _.map(parentCollection, function (model) {
            var attrs = $.extend(true, {}, model._body);
            attrs.id = model._id;

            return new this.model(attrs);
        }.bind(this));
    },

    /**
     * Converts the SubModels back into the parent Model's nested structure
     *
     * @method serialize
     * @return {Array} The array of nested models
     */
    serialize: function () {
        if (!this.length) {
            return null;
        }

        return _.map(this.models, function (model) {
            // create nested model structure
            var origModel = {
                _body: null
            };

            // set id. new models will not have an id
            if (!model.isNew()) {
                origModel._id = model.id;
            }

            // if this is new or has at least 1 non-id attribute, set body
            // for deletes, there will only be 1 attribute (id), and body will stay null
            if (!origModel._id || _.size(model.attributes) > 1) {
                origModel._body = {};

                _.each(model.attributes, function (value, key) {
                    if (key !== 'id') {
                        origModel._body[key] = value;
                    }
                });
            }

            return origModel;
        });
    },

    /**
     * Generates new set of SubModels from the parent attribute
     *
     * @method resetSubModels
     */
    resetSubModels: function () {
        var subModels;

        // make a copy of the parent nested model collection
        this.origCol = $.extend(true, [], this.parent.attributes[this.colName]);

        subModels = this.parse(this.origCol);

        this.reset(subModels);
    },

    /**
     * Manually destroy a SubModel by unsetting all its attributes and removing it from the collection
     *
     * @method destroy
     */
    destroy: function (model) {
        // unset the model attributes
        _.each(model.attributes, function (value, key) {
            if (key !== 'id') {
                model.unset(key, ({silent: true}));
            }
        });

        this.setParent();

        // remove it from this collection
        this.remove(model);
    },

    /**
     * Sets the changes on the parent Model
     * Includes the entire collection of nested models
     *
     * @method setParent
     */
    setParent: function () {
        // set changed attritbute, letting the parent model handle validating itself
        this.parent.set(this.colName, this.serialize(), {validate: false});
    },

    /**
     * Override Backbone default to not automatically remove models on destroy
     *
     * @method _onModelEvent
     */
    _onModelEvent: function(event, model, collection, options) {
        if ((event === 'add' || event === 'remove') && collection !== this) return;
        // added '&& collection === this', so that not passing the collection will let us remove the model manually
        if (event === 'destroy' && collection === this) this.remove(model, options);
        if (model && event === 'change:' + model.idAttribute) {
            delete this._byId[model.previous(model.idAttribute)];
            if (model.id != null) this._byId[model.id] = model;
        }
        this.trigger.apply(this, arguments);
    },

    /**
     * No-op
     * Overrides the native function to avoid sync-ing
     *
     * @method sync
     */
    sync: function () {}
});

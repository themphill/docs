/**
 * Pool Activity Edit Card View
 *
 * @module SourceTrak
 * @class Views.ActivityPoolEdit
 * @extends Ibp.Backbone.View
 */
Ibp.Sourcetrak.Views.ActivityPoolEdit = Ibp.Backbone.View.extend({
    mixins: ['Autosave'],
    name: 'ActivityPoolEdit',
    template: Ibp.Sourcetrak.Templates.activity_pool_edit,

    /**
     * @property cardType
     * @type String
     */
    cardType: 'poolEdit',

    /**
     * DOM selector for the div that contains the minimum numbers view template
     * @property numbersInputEl
     * @type string
     */
    numbersInputEl: '.js-reserved-numbers',

    /**
     * Object containing views controlled by this view
     * @property views
     * @type array of Objects
     */
    views: null,

    /**
     * Listens for:
     * clicks on the done button
     *
     * @property events
     * @type String
     */
    events: {
        'click .js-activity-done': 'cancelEdit'
    },

    /**
     * Calls render
     *
     * @method init
     */
    init: function (options) {
        this.views = {};
        this.$container = options.$container;

        // update max available numbers in the UI whenever any Activity changes its reserverd numbers
        this.listenTo(this.model.collection, 'change:min_numbers', this.updateMaxNumbers);

        this.render();
    },

    /**
     * Renders view and configures the edit button fade in/out
     *
     * @method render
     * @return this view
     */
    render: function () {
        // derived properties
        var data = {
            activity_type_name: this.model.getActivityType().description,
            platform_type     : this.model.getPlatformType() ? this.model.getPlatformType().description : '',
            visits            : this.model.getStats('total_visits_30_days'),
            calls             : this.model.getStats('total_calls_30_days')
        };

        // add in model attributes
        _.extend(data, this.model.toJSON());

        // create activity body
        this.$el = $(this.template(data));

        // render
        this.$container.html(this.$el);

        // initialize the minimum number select view
        this.views.minimumNumberSelect = new Ibp.Sourcetrak.Views.ReservedNumbers({
            model: this.model,
            $container: this.$(this.numbersInputEl)
        });

        return this;
    },

    /**
     * Cancels edit and flips back to the display card
     *
     * @method canceledit
     */
    cancelEdit: function () {
        this.trigger('flip', {card: this.cardType, event: 'pool'});
    }
});

Ibp.Sourcetrak.Views.NumbersAreaCodeItem = Ibp.Carbon.Views.ModalPageView.extend({
    name: 'numbersAreaCodeItem',
    template: Ibp.Sourcetrak.Templates.numbers_area_code_item,
    optionsTemplate: Ibp.Sourcetrak.Templates.select_options,

    /**
     * Model to hold view state
     *
     * @property state
     * @type Model
     */
    state: null,

    /**
     * Container for this search view
     *
     * @property $container
     * @type jQuery Object
     */
    $container: null,

    /**
     * String selector for quantity select input
     *
     * @property quantityEl
     * @type string
     */
    quantityEl: ".js-quantity",

    /**
     * jQuery Object containing reference to quantity select input
     *
     * @property $quantity
     * @type jQuery Object
     */
    $quantity: null,

    /**
     * The views object retains references to children views
     *
     * @property views
     * @type Object
     */
    views: {},

    /**
     * Holds the area code data passed into the view
     * Will contain the following properties:
     *
     * areaCode             - the area code being represented by this view
     * quantity             - the current quantity of numbers for this area code
     * inventoryLimit       - the maximum amount of numbers available from all of our sources
     * unconfiguredCount    - the amount of unconfigured numbers for this area code (freebies)
     * maxAddons            - the largest number of 'new numbers' that can be added to this account
     * totalUnconfigured    - the total amount of unconfigured numbers for all area codes currently rendered
     *
     * @property views
     * @type Object
     */
    areaCodeData: null,

    /**
     * Stores the current number total for this pool. Used in calculating maximum numbers for this area code item
     *
     * @property poolTotal
     * @type integer
     */
    poolTotal: null,

    /**
     * Stores the current number total for this pool. Used in calculating maximum numbers for this area code item
     *
     * @property newInventoryTotal
     * @type integer
     */
    newInventoryTotal: null,

    /**
     * Text option to be inserted into the select input to indicate the bandwidth inventory should be checked
     *
     * @property bandwidthTriggerText
     * @type String
     */
    bandwidthTriggerText: null,

    /**
     */
    events: {
        "change select.js-quantity": "handleQuantityChange",
        "click .js-remove": "handleRemove"
    },

    /**
     * Init state and render
     *
     * @method init
     */
    init: function (options) {
        this.areaCodeData = options.areaCodeData;
        this.$container = options.$container;
        this.poolTotal = 0;
        this.newInventoryTotal = 0;

        // set initial state
        this.state = new Ibp.Carbon.Models.State({
            showBandwidth: options.showBandwidth || false,
            inventoryLimit: options.inventoryLimit ? options.inventoryLimit + this.areaCodeData.originalQuantity : this.areaCodeData.originalQuantity
        });

        // we setup this function as a once-only firing object to prevent repeatedly adding this view to the parent list
        this.prependToContainer = _.once(this.prependToContainer);

        // any changes to the state properties should trigger a re-render
        this.listenTo(this.state, 'change', this.render);

        this.render();
    },

    /**
     * Prepend this view's html into the parent container
     *
     * @method prependToContainer
     */
    prependToContainer: function () {
        this.$container.prepend(this.$el);
    },

    /**
     * Render
     *
     * @method render
     * @return this view
     */
    render: function () {
        var areaCode = this.areaCodeData.areaCode,
            // format the label depending on its length
            label = (areaCode.length === 6) ? areaCode.slice(0, 3) + '-' + areaCode.slice(3) : areaCode;

        this.$el.html(this.template({
            areaCode: label,
            quantity: this.areaCodeData.quantity
        }));

        // this function will only fire the first time render is called
        this.prependToContainer();

        this.$quantity = this.$(this.quantityEl);

        // populate the select input
        this.renderSelectOptions();

        return this;
    },

    /**
     * Figure out the maximum option to display in this view's select input. Depends on tons of variables. I'm so sorry.
     *
     * @method calculateMaxNumbers
     * @return integer
     */
    calculateMaxNumbers: function () {
            // figure out how many of this view's current quantity are not covered by unconfigured numbers
        var newInventoryQuantity = this.getNewInventoryQuantity(),
            // the limit imposed by ifbyphone/bandwidth's reserves. does not depend on other area code item views
            inventoryLimit = this.state.get('inventoryLimit') + this.areaCodeData.unconfiguredCount,
            //the limit imposed by this and all other area code item views with respect to your addon slots
            accountLimit = this.areaCodeData.maxAddons + this.areaCodeData.unconfiguredCount - this.newInventoryTotal + newInventoryQuantity + this.areaCodeData.quantity,
            // return whichever of these two limits is lower
            max = (accountLimit > inventoryLimit) ? inventoryLimit : accountLimit;

        // if it's maxed out, but this area code already exists, use the current quantity as the max
        if (!max && this.areaCodeData.quantity) {
            max = this.areaCodeData.quantity;
        }

        return max;
    },

    /**
     * Take the highest option integer and use it to render the select options
     *
     * @method renderSelectOptions
     */
    renderSelectOptions: function () {
        var maxNumbers = this.calculateMaxNumbers(),
            choices = _.range(1, maxNumbers + 1),
            plural;

        // detect when the bandwidth inventory check trigger has been clicked, and if so, alert the parent view
        if (this.state.get('showBandwidth')) {
            plural = maxNumbers === 1 ? '' : 's';
            this.bandwidthTriggerText = 'Add more than ' + maxNumbers + ' number' + plural + '.';
            choices.push(this.bandwidthTriggerText);
        }

        this.$quantity.html(this.optionsTemplate({
            // range return an array from 1 to N, useful for sequential select option values
            choices: choices,
            current: this.areaCodeData.quantity
        }));
    },

    /**
     * Remove this item view after alerting the summary of the impending change
     *
     * @method handleRemove
     */
    handleRemove: function () {
        // set the quantity to zero and trigger a quantity update before removing this view
        this.areaCodeData.quantity = 0;

        // tell the summary view to take this view out of its list property
        this.trigger('remove', this.areaCodeData.areaCode);
        this.remove();
    },

    /**
     * Updates the view's state with some number of updated options
     *
     * @method updateState
     * @param {Object} attributes
     */
    updateState: function (attributes) {
        if (_.has(attributes, 'inventoryLimit')) {
            // if we get an update to the inventory limit state
            attributes.inventoryLimit += this.state.get('inventoryLimit');
        }

        this.state.set(attributes);
    },

    /**
     * React to the select input being updated
     *
     * @method handleQuantityChange
     */
    handleQuantityChange: function () {
        var newVal = this.$quantity.val();

        // if this is a bandwidth trigger, take some special steps
        if (newVal === this.bandwidthTriggerText) {
            this.trigger('searchBandwidthInventory', this.areaCodeData.areaCode);
        }
        else {
            this.areaCodeData.quantity = parseInt(newVal, 10);

            // tell the summary view that this view has a new quantity
            this.trigger('update:quantity');
        }
    },

    /**
     * Returns the area code of this item view
     *
     * @method getAreaCode
     * @return integer
     */
    getAreaCode: function () {
        return this.areaCodeData.areaCode;
    },

    /**
     * Return the current quantity of this item view
     *
     * @method getQuantity
     * @return integer
     */
    getQuantity: function () {
        return this.areaCodeData.quantity;
    },

    /**
     * Return the amount of this view's quantity that is not covered by its unconfigured inventory
     *
     * @method getNewInventoryQuantity
     * @return integer
     */
    getNewInventoryQuantity: function () {
        var newInventoryCount = this.areaCodeData.quantity - this.areaCodeData.unconfiguredCount - this.areaCodeData.originalQuantity;
        return (newInventoryCount > 0) ? newInventoryCount : 0;
    },

    /**
     * Update this item's total pool numbers property
     *
     * @method setPoolTotal
     * @param integer newTotal
     */
    setPoolTotal: function (newTotal) {
        this.poolTotal = newTotal;
    },

    /**
     * Update this item's total unconfigured numbers count property
     *
     * @method setNewInventoryTotal
     * @param integer newTotal
     */
    setNewInventoryTotal: function (newTotal) {
        this.newInventoryTotal = newTotal;
    }
});

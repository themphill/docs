/**
 * Domain Set View For The Domain Set's Locations List
 *
 * @module SourceTrak
 * @class Views.LocationsList
 * @extends Ibp.Backbone.View
 */
Ibp.Sourcetrak.Views.LocationsList = Ibp.Backbone.View.extend({
    name: 'LocationsList',
    template: Ibp.Sourcetrak.Templates.locations_list,

    /**
     * The views object retains references to children views (single locations)
     *
     * @property views
     * @type Object
     */
    views: null,

    /**
     * A list of references the views for each location in the list
     *
     * @property locationViews
     * @type Object
     */
    locationViews: null,

    /**
     * A state model object we use for tracking filter activity
     *
     * @property state
     * @type Model
     */
    state: null,

    /**
     * The activities being displayed
     *
     * @property paginatedLocations
     * @type Collection
     */
    paginatedLocations: null,

    /**
     * String storing the DOM element that this view will populate
     *
     * @property $container
     * @type jQuery Object
     */
    $container: null,

    /**
     * String selector for this view's search box DOM element
     *
     * @property searchBoxEl
     * @type String
     */
    searchBoxEl: '.js-locations-search-box',

    /**
     * String for the placeholder attribute in the search box child view.
     *
     * @property searchPlaceholder
     * @type String
     */
    searchPlaceholder: 'Filter Locations',

    /**
     * String selector for this view's filter alert DOM element
     *
     * @property filterAlertEl
     * @type String
     */
    filterAlertEl: '.js-locations-filter-alert',

    /**
     * Holds a jQuery reference to the filter DOM element
     *
     * @property $filterAlert
     * @type jQuery Object
     */
    $filterAlert: null,

    /**
     * Locations container selector
     *
     * @property filterAlertEl
     * @type String
     */
    locationsEl: '.js-locations-container',

    /**
     * Locations container jQuery object
     *
     * @property $filterAlert
     * @type jQuery Object
     */
    $locations: null,

    /**
     * String selector for add location button selector
     * within the list.
     *
     * @property addLocationBtnEl
     * @type String
     */
    addLocationBtnEl: '.js-add-location-btn',

    /**
     * Holds a jQuery reference to the add location button
     * selector within the list.
     *
     * @property $addLocationBtn
     * @type jQuery Object
     */
    $addLocationBtn: null,

    /**
     * String selector for add location row selector
     * within the list.
     *
     * @property addLocationLinkEl
     * @type String
     */
    addLocationRowEl: '.js-add-location-row',

    /**
     * Holds a jQuery reference to the add location row
     * selector within the list.
     *
     * @property $addLocationRow
     * @type jQuery Object
     */
    $addLocationRow: null,

    /**
     * String selector for container holding number of locations
     * displayed out of total locations
     *
     * @property locationsCountEl
     * @type String
     */
    locationsCountEl: '.js-locations-count',

    /**
     * Holds a jQuery reference to locationsCountEl
     *
     * @property $locationCount
     * @type jQuery Object
     */
    $locationsCount: null,

    /**
     * String selector for the paginator buttons and count
     *
     * @property paginatorControlsEl
     * @type String
     */
    paginatorControlsEl: '.js-paginator-controls',

    /**
     * Paginator controls jQuery object
     *
     * @property $paginatorControls
     * @type jQuery Object
     */
    $paginatorControls: null,

    /**
     * DOM Event handlers
     *
     * @property events
     * @type Object
     */
    events: {
        'click .js-locations-filter-alert-dismiss': 'handleDismissFilterAlert',
        'click .js-add-location': 'handleAddLocationClick'
    },

    /**
     * Init
     *
     * @method init
     * @param options {Object}
     */
    init: function (options) {
        _.bindAll(this);

        this.$container = options.$container;
        this.domainSet  = options.domainSet;

        this.views = {};
        this.locationViews = [];

        // create a collection of locations that can be paginated
        this.paginatedLocations = new Ibp.Sourcetrak.Collections.PaginatedLocations(
            this.domainSet.getLocations()
        );

        // prime the paginator state
        this.paginatedLocations.setPageState();

        this.state = new Ibp.Carbon.Models.State({
            query: '',
            showNew: false,
            showAddButtons: this.getButtonsState(),
            showControls: this.shouldShowControls(),
            renderLocations: true
        });

        this.listenTo(this.state, 'change', this.handleViewStateChange);
        this.listenTo(this.paginatedLocations.ClientPagination.state, 'change', this.handlePageStateChange);

        // handle adding and removing locations
        this.listenTo(Ibp.Sourcetrak.Data.Locations, 'add remove', this.handleLocationsChange);

        // allow users to create locations once a new domain set has been saved
        if (this.domainSet.isNew()) {
            this.listenToOnce(this.domainSet, 'sync', this.handleCreateDomainSet);
        }

        this.render();
    },

    /**
     * Render
     *
     * @method render
     * @return this view
     */
    render: function () {
        this.$el = $(this.template());

        this.$container.append(this.$el);

        this.$filterAlert       = this.$(this.filterAlertEl);
        this.$locations         = this.$(this.locationsEl);
        this.$addLocationBtn    = this.$(this.addLocationBtnEl);
        this.$addLocationRow    = this.$(this.addLocationRowEl);
        this.$locationsCount    = this.$(this.locationsCountEl);
        this.$paginatorControls = this.$(this.paginatorControlsEl);

        this.renderSearchbox();

        this.renderFilterAlert();

        this.renderLocations();

        this.renderAddLocationButtons();

        return this;
    },

    /**
     * Renders the search box for this locations list
     *
     * @method renderSearchbox
     */
    renderSearchbox: function () {
        this.views.searchBox = new Ibp.Carbon.Views.SearchBox({
            el: this.searchBoxEl,
            placeholderText: this.searchPlaceholder
        });

        this.listenTo(this.views.searchBox, 'search', this.handleSearch);
    },

    /**
     * Shows or hides the query alert box, with updated search text.
     *
     * @method renderFilterAlert
     * @param query {String} The value of the search box
     */
    renderFilterAlert: function () {
        var query = this.state.get('query'),
            hideFilter = !Boolean(query);

        // if we have a query, show the filter alert
        this.$filterAlert.toggleClass('hidden', hideFilter);

        this.$filterAlert.find('.value').html(query);
    },

    /**
     * Renders the locations in the locations list container based on the page and filtering
     * Also handles rendering the paginator controls and current display count
     *
     * @method renderLocations
     */
    renderLocations: function () {
        var locations   = this.paginatedLocations.getPage(),
            locationIds = _.pluck(locations, 'id'),
            viewIds     = _.chain(this.locationViews).pluck('model').pluck('id').value(),
            addedLocationIds   = _.difference(locationIds, viewIds),
            removedLocationIds = _.difference(viewIds, locationIds),
            viewState = {renderLocations: false},
            // keep track of the container size to help prevent jank
            oldHeight = this.$locations.css('height'),
            newHeight;

        // just start from scratch for filters
        if (this.wasQueryChanged()) {
            _.invoke(this.locationViews, 'unrender');

            this.locationViews = _.map(locations, function (location) {
                var view = new Ibp.Sourcetrak.Views.LocationItem({
                    model: location
                });

                this.$locations.append(view.el);

                return view;
            }.bind(this));
        }
        // attempt to preserve state for non-filter operations
        else {
            // unrender removed views
            if (removedLocationIds.length) {
                var removed = [];

                _.each(this.locationViews, function (view) {
                    if (_.contains(removedLocationIds, view.model.id)) {
                        view.unrender();

                        removed.push(view);
                    }
                }.bind(this));

                this.locationViews = _.difference(this.locationViews, removed);
            }

            // create and render new views
            if (addedLocationIds.length) {
                _.each(addedLocationIds, function (id) {
                    var view = new Ibp.Sourcetrak.Views.LocationItem({
                        model: _.findWhere(locations, {id: id})
                    });

                    // handle the special case of adding a new location at the top of the list
                    if (this.state.get('showNew') && id === locations[0].id) {
                        this.locationViews.unshift(view);

                        this.$locations.prepend(view.el);
                    }
                    else {
                        this.locationViews.push(view);

                        this.$locations.append(view.el);
                    }
                }.bind(this));
            }
        }

        // get new container size and set new base height if need be
        newHeight = this.$locations.css('height');

        if (newHeight > oldHeight) {
            this.$locations.css('min-height', newHeight);
        }

        this.renderPaginatorControls();

        this.renderLocationsCount();

        // expand the new location if we have one
        if (this.state.get('showNew')) {
            this.showNewLocation();

            viewState.showNew = false;
        }

        // silently reset the view state
        this.state.set(viewState, {silent: true});
    },

    /**
     * Re-attaches the add location button to the DOM once the domain set has synced.
     *
     * @method renderAddLocationButtons
     */
    renderAddLocationButtons: function () {
        switch (this.state.get('showAddButtons')) {
            case 'none':
                this.$addLocationBtn.hide();
                this.$addLocationRow.hide();
                break;
            case 'btn':
                this.$addLocationBtn.show();
                this.$addLocationRow.hide();
                break;
            case 'both':
                this.$addLocationBtn.show();
                this.$addLocationRow.show();
        }
    },

    /**
     * Renders the paginator buttons for switching pages
     *
     * @method renderPaginatorControls
     */
    renderPaginatorControls: function () {
        this.resetPaginatorControls();

        // don't show the controls
        if (!this.state.get('showControls')) {
            return;
        }

        this.views.paginatorControls = new Ibp.Sourcetrak.Views.listControls({
            $container: this.$paginatorControls,
            pageState:  this.paginatedLocations.getPageState()
        });

        this.listenTo(this.views.paginatorControls, 'pageClick', this.handlePageClick);
    },

    /**
     * Renders the count of what the paginator is currently displaying
     * This is shown both above and below the paginated list of locations
     *
     * @method renderLocationsCount
     */
    renderLocationsCount: function () {
        // clear the old count
        this.$locationsCount.html('');

        // don't show the count
        if (!this.state.get('showControls')) {
            return;
        }

        var info  = this.paginatedLocations.getPageState(),
            total = info.totalItems,
            range = info.pageRange,
            displayed = range.end - range.start + 1,
            countMessage = 'Displaying ';

        // if we're displaying all of the items or just one item, don't show a range
        if (displayed === total || range.end - range.start === 0) {
            countMessage += range.end + ' of ' + total;
        }
        else {
            countMessage += range.start + ' - ' + range.end + ' of ' + total;
        }

        this.$locationsCount.html(countMessage);
    },

    /**
     * Expands a new location after it is saved
     *
     * @method showNewLocation
     */
    showNewLocation: function () {
        this.locationViews[0].showNew();
    },

    /**
     * Removes and cleans up the paginator controls so they can be re-rendered
     *
     * @method resetPaginatorControls
     */
    resetPaginatorControls: function () {
        var view = this.views.paginatorControls;

        if (view) {
            view.unrender();

            this.stopListening(view);

            delete this.views.paginatorControls;
        }
    },

    /**
     * Takes care of unrendering child views specific to this view.
     *
     * @method onUnrender
     */
    onUnrender: function () {
        _.invoke(this.views, 'unrender');

        _.invoke(this.locationViews, 'unrender');
    },

    /**
     * Creates a new location when the Add Location button or row is clicked
     *
     * @method handleAddLocationClick
     */
    handleAddLocationClick: function () {
        // create the new location
        var model = Ibp.Sourcetrak.Data.Locations.getNew({
            domain_set_id: this.domainSet.id
        });

        model.trigger('create', model);
    },

    /**
     * Re-renders locations whenever the collection changes
     * Will automatically move back to the first page and expand a new location when added
     *
     * @method handleLocationsChange
     * @param {Model} model Unused
     * @param {Collection} collection Unused
     * @param {Object} event Information about the event
     */
    handleLocationsChange: function (model, collection, event) {
        var newLocations = this.domainSet.getLocations(),
            viewState    = {
                showControls:   this.shouldShowControls(),
                showAddButtons: this.getButtonsState()
            },
            resetOptions = {
                models: newLocations
            };

        // handle adding a new location if this was an add event
        if (event.add) {
            // expand the new location
            viewState.showNew = true;

            // move back to the first page and clear the query
            resetOptions.hard = true;
        }

        // silently update view state
        this.state.set(viewState, {silent: true});

        // update the paginated collection and trigger re-render
        this.paginatedLocations.resetPageState(resetOptions);
    },

    /**
     * Handles changes to the search box by setting the query state for the view and the locations
     *
     * @method handleSearch
     * @param query {String} The value of the search box
     */
    handleSearch: function (query) {
        this.paginatedLocations.setQuery(query);
    },

    /**
     * Switches the page when a user clicks a paginator button
     *
     * @method paginatedLocations
     * @param page {String} The page to change to
     */
    handlePageClick: function (page) {
        this.paginatedLocations.setCurrentPage(page);
    },

    /**
     * Handles an updated query state
     *
     * @method handlePageStateChange
     * @param {Model} model The paginator state model
     */
    handlePageStateChange: function (model) {
        var state = {
            renderLocations: true,
            query: model.get('query')
        };

        this.state.set(state);
    },

    /**
     * Handles an updated query state
     *
     * @method handleViewStateChange
     */
    handleViewStateChange: function () {
        this.renderAddLocationButtons();

        // clear the search box if the paginator has been reset
        if (!this.state.get('query')) {
            this.views.searchBox.reset();
        }

        this.renderFilterAlert();

        this.renderLocations();
    },

    /**
     * Clears the location list filter and resets the query state
     *
     * @method handleDismissFilterAlert
     */
    handleDismissFilterAlert: function () {
        this.paginatedLocations.setQuery('');

        this.views.searchBox.reset();
    },

    /**
     * Show the add location row once a domain set has been created
     *
     * @method handleCreateDomainSet
     */
    handleCreateDomainSet: function () {
        this.state.set('showAddButtons', this.getButtonsState());
    },

    /**
     * Determines if we should render the paginator controls
     * If the total locations only require a single page, the controls should be hidden
     * since they wouldn't actually do anything.
     * If the list has been filtered down to a single page, we'll still show the controls,
     * since they'll still be useful once the query has been cleared.
     *
     * @method shouldShowControls
     */
    shouldShowControls: function () {
        return this.domainSet.getLocations().length > this.paginatedLocations.ClientPagination.state.get('pageSize');
    },

    /**
     * Determines which add location buttons should be shown
     * No buttons should show until the domain set has been created on the server.
     * The add location row should only be shown for domain sets with no locations.
     * The add location button should always be shown for created domain sets.
     *
     * @method getButtonsState
     */
    getButtonsState: function () {
        var state;

        if (this.domainSet.isNew()) {
            state = 'none';
        }
        else if (!this.domainSet.getLocations().length) {
            state = 'both';
        }
        else {
            state = 'btn';
        }

        return state;
    },

    /**
     * Determines if we're currently filtering or have just cleared a filter
     *
     * @method wasQueryChanged
     */
    wasQueryChanged: function () {
        var oldQuery = this.state.previous('query') || '',
            newQuery = this.state.get('query');

        return newQuery !== oldQuery;
    }
});

Ibp.Sourcetrak.Views.EditDomains = Ibp.Carbon.Views.PageView.extend({
    name: 'editDomains',
    el: '#sourcetrak-editDomains',
    template: Ibp.Sourcetrak.Templates.edit_domains,

    /**
     * Backbone Model to hold view state
     *
     * @property state
     * @type Model
     */
    state: null,

    /**
     * Object containing this view's associated views
     *
     * @property views
     */
    views: null,

    /**
    * Add new domain container
    *
    * @property addFormEl
    * @type String
    */
    addFormEl: '.js-domain-form',

    /**
     * Selector for domains list container
     *
     * @property domainsListEl
     * @type String
     */
    domainsListEl: '.js-domains-list',

    /**
     * jQuery Object for domains container element
     *
     * @property $domainsList
     * @type jQuery Object
     */
    $domainsList: null,

    /**
     * Selector for add domain input
     *
     * @property addDomainInputEl
     * @type String
     */
    addDomainInputEl: '.js-domain-input',

    /**
     * jQuery Object for add domain input
     *
     * @property $addDomainInput
     * @type jQuery Object
     */
    $addDomainInput: null,

    /**
     * Class for the add domain input which indicates a validation error
     *
     * @property errorClass
     * @type String
     */
    errorClass: 'validation-error',

    /**
     * DOM Event handlers
     *
     * @property events
     */
    events:  {
        'click .js-cancel':   'handleCancel',
        'click .js-add':      'handleAdd',
        'click .js-save-new': 'handleNewDomain'
    },

    /**
     * Init
     *
     * @method init
     */
    init: function (options) {
        this.$container = options.$container;

        this.views = {};

        this.state = new Ibp.Carbon.Models.State({
            showAdd: false
        });

        this.listenTo(this.collection, 'reset', this.handleCollectionChange);

        this.listenTo(this.state, 'change:showAdd', this.handleShow);

        this.render();
    },

    /**
     * Render
     *
     * @method render
     * @return this view
     */
    render: function () {
        this.$el = $(this.template());
        this.$container.html(this.$el);

        this.$addForm = this.$(this.addFormEl);
        this.$addDomainInput = this.$(this.addDomainInputEl);
        this.$domainsList = this.$(this.domainsListEl);

        this.renderDomains();

        return this;
    },

    /**
     * Renders the list of domains to the template.
     *
     * @method renderDomains
     */
    renderDomains: function () {
        var domains = Backbone.sortCollection(this.collection.models, 'modified_date', 'desc');

        _.invoke(this.views, 'unrender');

        _.each(domains, function(model) {
            this.views[model.get('id')] = new Ibp.Sourcetrak.Views.Domain({
                $container: this.$(this.domainsListEl),
                model: model
            });
        }.bind(this));
    },

    /**
     * Handler for cancel button
     *
     * @method handleCancel
     */
    handleCancel: function () {
        this.state.set('showAdd', false);
    },

    /**
     * Handler for add new domain button
     *
     * @method handleAdd
     */
    handleAdd: function () {
        this.state.set('showAdd', true);
    },

    /**
     * Handler for save new domain button
     *
     * @method handleNewDomain
     */
    handleNewDomain: function () {
        this.saveDomain();
    },

    /**
     * Handler for 'showAdd' state
     *
     * @method handleShow
     */
    handleShow: function () {
        this.toggleForm();
    },

    /**
     * Handler for the domains collection
     *
     * @method handleCollectionChange
     */
    handleCollectionChange: function () {
        this.state.set('showAdd', false);

        this.renderDomains();
    },

    /**
     * Show/hide the add new domain form
     *
     * @method toggleForm
     */
    toggleForm: function () {
        if (this.state.get('showAdd')) {
            this.showForm();
        }
        else {
            this.hideForm();
        }
    },

    /**
     * hideForm
     * Hides the add new domain form from view
     *
     * @method hideForm
     */
    hideForm: function () {
        this.$addDomainInput.val('');

        this.$addDomainInput.removeClass(this.errorClass);

        if (this.collection.parent.validationError) {
            this.publish('notify:dismiss');
        }

        this.$addForm.hide();
    },

    /**
     * showForm
     * renders the form for adding a new domain to the domain set
     *
     * @method showForm
     */
    showForm: function () {
        this.$addForm.show();

        this.$addDomainInput.focus();
    },

    /**
     * Save a new domain
     *
     * @method saveDomain
     */
    saveDomain: function () {
        var val = this.$addDomainInput.val();

        if (val) {
            this.collection.create({domain: val});
        }
    }
});

/**
 * Create new Pool wizard manager View
 *
 * @module Sourcetrak
 * @class Views.Wizard
 */
Ibp.Sourcetrak.Views.Wizard = Ibp.Carbon.Views.ModalPageView.extend({
    name: 'wizard',
    el: '#pools-wizard',
    template: Ibp.Sourcetrak.Templates.wizard,

    /**
     * Model to hold view state
     *
     * @property state
     * @type Model
     */
    state: null,

    /**
     * The views object retains references to children views
     *
     * @property views
     * @type Object
     */
    views: {},

    /**
     * Selector for the current wizard view container
     *
     * @property wizardViewEl
     * @type String
     */
    wizardViewEl: '#wizard-current',

    /**
     * The views that correspond to each wizard state
     *
     * @property wizardViews
     * @type Object
     */
    wizardTypeTitles: {
        main: 'Create Main Number',
        pool: 'Create Pool',
        oneToOne: 'Create One-To-One',
        edit: 'Edit ',
        location: 'Edit Location Pool'
    },

    /**
     * The views that correspond to each wizard state
     *
     * @property wizardViews
     * @type Object
     */
    wizardViews: {
        numbers:    Ibp.Sourcetrak.Views.Numbers,
        routing:    Ibp.Sourcetrak.Views.NumberRouting,
        analytics:  Ibp.Sourcetrak.Views.Analytics,
        review:     Ibp.Sourcetrak.Views.Review,
        error:      Ibp.Sourcetrak.Views.PoolError
    },

    /**
     * Maps child view events to the child view the wizard should transition to
     *
     * @property wizardFlow
     * @type Object
     */
    wizardFlow: {
        numbers: {
            prev: null,
            next: 'routing'
        },
        routing: {
            prev: 'numbers',
            next: 'analytics'
        },
        analytics: {
            prev: 'routing',
            next: 'review'
        },
        review: {
            prev: 'analytics',
            next: null
        },
        error: {
            prev: null,
            next: null
        }
    },

    wizardOrder: [
        'numbers', 'routing', 'analytics', 'review'
    ],

    /**
     * The child View currently being shown
     *
     * @property currentView
     * @type View
     */
    currentView: null,

    /**
     * The the default card state to display
     *
     * @property DEFAULT_CARD
     * @type String
     */
    DEFAULT_VIEW: 'numbers',

    /**
     * Object to hold data that needs to persist between steps
     * Passed into each view
     *
     * @property wizardData
     * @type Object
     */
    wizardData: null,

    /**
     * Ref table constants for addon types
     *
     * @property addonIds
     * @type Object
     */
    addonIds: {
        tollfree: 3,
        local: 6,
        canadian: 17,
        true800: 18
    },

    /**
     * Url for accessing the Addon endpoint of Organic
     *
     * @property addonUrl
     * @type String
     */
    addonUrl: '/v1/addon',

    /**
     * Url for accessing the Addon endpoint of Organic
     *
     * @property numberUrl
     * @type String
     */
    numberUrl: '/v1/phonenumber',

    /**
    * DOM container for current wizard view
    *
    * @property currentViewEl
    * @type String
    */
    currentViewEl: '#wizard-current',

    /**
    * Selector for the 'previous' button
    *
    * @property prevEl
    * @type String
    */
    prevEl: '.js-prev',

    /**
     * Reference to prev button
     *
     * @property $prev
     * @type jQuery Object
     */
    $prev: null,

    /**
    * Selector for the 'next' button
    *
    * @property nextEl
    * @type String
    */
    nextEl: '.js-next',

    /**
     * Reference to next button
     *
     * @property $next
     * @type jQuery Object
     */
    $next: null,

    /**
    * Selector for the 'save' button
    *
    * @property saveEl
    * @type String
    */
    saveEl: '.js-save',

    /**
     * Reference to save button
     *
     * @property $save
     * @type jQuery Object
     */
    $save: null,

    /**
    * Selector for the 'cancel' button
    *
    * @property cancelEl
    * @type String
    */
    cancelEl: '.js-cancel',

    /**
     * Reference to cancel button
     *
     * @property $cancel
     * @type jQuery Object
     */
    $cancel: null,

    /**
     * Class for disabling a button
     *
     * @property btnDisabledClass
     * @type String
     */
    btnDisabledClass: 'disabled',

    /**
    * Selector for the wizard progress steps
    *
    * @property progressEl
    * @type String
    */
    progressEl: '.wizard-progress > span',

    subscriptions: {
        'beginLoadStep': 'handleBeginLoadStep',
        'endLoadStep': 'handleEndLoadStep',
        'beginSearch': 'handleBeginSearch',
        'endSearch': 'handleEndSearch',
        'noSlots': 'handleNoSlots',
        'apiError': 'handleError',
        'goToStep': 'handleGoToStep'
    },

    events:  {
        'click .js-prev':   'handlePrev',
        'click .js-next':   'handleNext',
        'click .js-cancel': 'handleCancel',
        'click .js-save':   'handleSave'
    },

    /**
     * Init state and render
     *
     * @method init
     */
    init: function (options) {
        _.bindAll(this);

        // set initial state
        this.state = new Ibp.Carbon.Models.State({
            currentView: options.currentView || this.DEFAULT_VIEW,
            wizardType: options.type
        });

        this.wizardData = {
            addons: {},
            addonInfo: {},
            unconfigured: {},
            numbers: {},
            currentNumbers: $.extend(true, [], this.model.getAreaCodes().toJSON()),
            available: {},
            error: null
        };

        // snapshot the model before we make changes so we can revert if the user cancels this edit
        if (!this.model.isNew()) {
            this.model.snapshot();
        }

        // listen for card flips and render new card
        this.listenTo(this.state, 'change:currentView', this.handleViewChange);
    },

    /**
     * Render
     *
     * @method render
     * @return this view
     */
    render: function () {
        var data = {
            title: this.wizardTypeTitles[this.state.get('wizardType')]
        };

        if (!this.model.isNew() && this.state.get('wizardType') !== 'location') {
            data.title += this.model.get('description');
        }

        this.$el.removeClass('hidden');

        this.$el.html(this.template(data));

        this.$prev = this.$(this.prevEl);
        this.$next = this.$(this.nextEl);
        this.$save = this.$(this.saveEl);
        this.$cancel = this.$(this.cancelEl);

        // render the current card state
        this.renderWizardView();

        return this;
    },

    /**
     * Destroy child views on remove
     *
     * @method onUnrender
     */
    onUnrender: function () {
        _.invoke(this.views, 'unrender');
    },

    /**
     * Render Wizard View
     *
     * @method render
     * @return this view
     */
    renderWizardView: function () {
        var View = this.wizardViews[this.state.get('currentView')];

        if (this.views.currentView) {
            this.views.currentView.unrender();
        }

        this.views.currentView = new View({
            $container: this.$(this.wizardViewEl),
            type: this.state.get('wizardType'),
            model: this.model,
            wizardData: this.wizardData,
            edit: (this.state.get('wizardType') === 'edit')
		});

        this.updateFooterNavigation();
        this.updateWizardProgress();
    },

    /**
     * Update Wizard Footer Buttons
     *
     * @method updateFooterNavigation
     */
    updateFooterNavigation: function () {
        var flow  = this.wizardFlow[this.state.get('currentView')];

        // handle previous button
        if (flow.prev !== null) {
            this.$prev.removeClass('hide');
        }
        else {
            this.$prev.addClass('hide');
        }

        // handle next button
        if (flow.next !== null) {
            this.$next.removeClass('hide');
            this.$save.addClass('hide');
        }
        else {
            this.$next.addClass('hide');
            this.$save.removeClass('hide');
        }
    },

    /**
     * Update Wizard Progress
     *
     * @method updateWizardProgress
     */
    updateWizardProgress: function () {
        var index = _.indexOf(this.wizardOrder, this.state.get('currentView')),
            progressItems = this.$(this.progressEl);

        progressItems.removeClass('completed current');

        for(var i = 0; i <= index; i++) {
            this.$(progressItems[i-1]).addClass('completed').removeClass('current');
            this.$(progressItems[i]).addClass('current');
        }
    },

    /**
     * Handles the "currentView" state change event
     *
     * @method handleViewChange
     */
    handleViewChange: function () {
        this.renderWizardView();
    },

    /**
     * Handles click on prev button
     *
     * @method handlePrev
     */
    handlePrev: function (e) {
        if ($(e.target).hasClass(this.btnDisabledClass)) {
            return;
        }

        this.prev();
    },

    /**
     * Handles click on next button
     *
     * @method handleNext
     */
    handleNext: function (e) {
        if ($(e.target).hasClass(this.btnDisabledClass)) {
            return;
        }

        this.next();
    },

    /**
     * Handles click on cancel button
     *
     * @method handleCancel
     */
    handleCancel: function (e) {
        if ($(e.target).hasClass(this.btnDisabledClass)) {
            return;
        }

        this.cancel();
    },

    /**
     * Handles click on cancel button
     *
     * @method handleCancel
     */
    handleSave: function (e) {
        if ($(e.target).hasClass(this.btnDisabledClass)) {
            return;
        }

        this.save();
    },

    /**
     * Handles any 'apiError' events during save
     *
     * @method handleError
     */
    handleError: function (error) {
        this.wizardData.error = error;

        this.unlockNavigation();

        this.publish('notify:dismiss');

        this.error();
    },

    /**
     * Handler for the 'goToStep' event
     *
     * @method handleError
     */
    handleGoToStep: function (view) {
        this.state.set('currentView', view);
    },

    /**
     * Handler for when a new step is navigated to
     *
     * @method handleBeginLoadStep
     */
    handleBeginLoadStep: function () {
        this.lockNavigation({lockCancel: false});
    },

    /**
     * Handler for when a new step has finished bootstrapping
     *
     * @method handleEndLoadStep
     */
    handleEndLoadStep: function () {
        this.unlockNavigation();
    },

    /**
     * Handler for when an area code search begins
     *
     * @method handleBeginSearch
     */
    handleBeginSearch: function () {
        this.disableNext();
    },

    /**
     * Handler for when an area code search ends
     *
     * @method handleEndSearch
     */
    handleEndSearch: function () {
        this.enableNext();
    },

    /**
     * Handler for when the addon and number order requests complete
     *
     * @method handleOrderComplete
     */
    handleOrderComplete: function () {
        this.saveModel();
    },

    /**
     * Navigate to previous wizard view
     *
     * @method prev
     */
    prev: function () {
        if (this.model.isValid()) {
            var flow = this.wizardFlow[this.state.get('currentView')];

            this.publish('notify:dismiss');

            if (flow.prev !== null) {
                this.state.set('currentView', flow.prev);
            }
        }
    },

    /**
     * Navigate to next wizard view
     * Validates model and only navigates if it is valid
     *
     * @method next
     */
    next: function () {
        if (this.model.isValid()) {
            var flow = this.wizardFlow[this.state.get('currentView')];

            this.publish('notify:dismiss');

            if (flow.next !== null) {
                this.state.set('currentView', flow.next);
            }
        }
    },

    /**
     * Cancel the wizard
     *
     * @method cancel
     */
    cancel: function () {
        var url = this.model.isNew() ? 'pools' : 'pools/' + this.model.id;

        if (this.model.isNew()) {
            this.model.trigger('remove', this.model);
        }
        else {
            this.model.revert();
        }

        if (this.state.get('wizardType') === 'location') {
            this.publish('modal:unrender');

            return;
        }

        //TODO: Be More Smartlike, DUH
        this.publish('navigate', url);
    },

    error: function () {
        this.state.set('currentView', 'error');
    },

    // TODO: Error handling
    /**
     * Save the wizard configuration
     *
     * @method cance
     */
    save: function () {
        this.lockNavigation();

        this.orderAddons()
            .then(this.orderNumbers)
            .done(this.handleOrderComplete);
    },

    /**
     * Sends the addon purchase request
     *
     * @method orderAddons
     * @return {Deferred} Promise with the results of the addon order
     */
    orderAddons: function () {
        var type = _.keys(this.wizardData.addons)[0],
            quantity = this.wizardData.addons[type],
            order = {
                item_id: this.addonIds[type],
                quantity: quantity
            },
            deferred;

        if (quantity) {
            this.publish('notify', {type: 'warn', msg: 'Purchasing Add-Ons...'});

            deferred = $.post(this.addonUrl, order);
        } else {
            deferred = new $.Deferred();

            // resolve it now so we can proceed immediately to the next request
            deferred.resolve();
        }

        return deferred;
    },

    /**
     * Sends the number purchase request
     *
     * @method orderNumbers
     * @return {Deferred} Promise with the results of the number order
     */
    orderNumbers: function () {
        var order = _.map(this.wizardData.numbers, function (quantity, code) {
                return {
                    search: code,
                    quantity: quantity,
                    allOrNothing: true
                };
            }),
            deferred;

        if (!_.isEmpty(order)) {
            this.publish('notify', {type: 'warn', msg: 'Ordering Phone Numbers...'});

            deferred = $.ajax({
                url: this.numberUrl,
                type: 'post',
                data: JSON.stringify(order),
                dataType: 'json',
                contentType: 'application/json'
            });
        } else {
            deferred = new $.Deferred();

            // resolve it now so we can proceed immediately to the next request
            deferred.resolve();
        }

        return deferred;
    },

    /**
     * Saves the model after the addon and number orders are complete
     *
     * @method saveModel
     */
    saveModel: function () {
        var event,
            handleSync;

        // handle locations
        if (this.state.get('wizardType') === 'location') {
            event = 'save';

            // go back to domain set edit page
            handleSync = function () {
                this.publish('modal:unrender');
            };
        }
        // handle new pools
        else if (this.model.isNew()) {
            event = 'create';

            // go back to pool modal dashboard
            handleSync = function () {
                this.publish('navigate', 'pools');
            };
        }
        // handle existing pools
        else {
            event = 'save';

            // go back to pool edit modal
            handleSync = function () {
                this.publish('navigate', 'pools/' + this.model.id);
            };
        }

        this.model.trigger(event, this.model, this.model.attributes);

        this.listenToOnce(this.model, 'sync', handleSync.bind(this));

        this.listenToOnce(this.model, 'error', function () {
            this.unlockNavigation();
        }.bind(this));
    },

    /**
     * Enables the next button
     *
     * @method enableNext
     */
    enableNext: function () {
        this.$next.removeClass(this.btnDisabledClass);
    },

    /**
     * Disables the next button
     *
     * @method disableNext
     */
    disableNext: function () {
        this.$next.addClass(this.btnDisabledClass);
    },

    /**
     * Disables all nav buttons
     *
     * @method lockNavigation
     * @param options
     * @param options.lockCancel {Boolean} Specifies if the Cancel button should be locked
     */
    lockNavigation: function (options) {
        this.$prev.addClass(this.btnDisabledClass);
        this.$next.addClass(this.btnDisabledClass);
        this.$save.addClass(this.btnDisabledClass);

        // leave cancel unlocked if explicitly told to do so
        if (options && options.lockCancel !== false) {
            this.$cancel.addClass(this.btnDisabledClass);
        }
    },

    /**
     * Enables all nav buttons
     *
     * @method unlockNavigation
     */
    unlockNavigation: function () {
        this.$prev.removeClass(this.btnDisabledClass);
        this.$next.removeClass(this.btnDisabledClass);
        this.$save.removeClass(this.btnDisabledClass);
        this.$cancel.removeClass(this.btnDisabledClass);
    }
});

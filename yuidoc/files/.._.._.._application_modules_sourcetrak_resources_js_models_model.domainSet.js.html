;(function () {
    function validateReplacmentNumberUnique(val, context) {
        var locations = context.model.getLocations(),
            locationNumbers = Backbone.pluckCollection(locations, 'replacement_number');

        return !(_.contains(locationNumbers, val));
    }

    function validateDomainUnique(val, context) {
        var model = context.model,
            domainsByModel = Ibp.Sourcetrak.Data.DomainSets.pluck('domains'),
            domainsList,
            allDomains;

        // the new model isn't in the collection, so add its domains manually
        if (model.isNew()) {
            domainsByModel.push(model.get('domains'));
        }

        // get the actual domains from each model's domain object list
        domainsList = _.map(domainsByModel, function (domains) {
            return _.map(domains, function (domain) {
                return domain._body ? domain._body.domain : null;
            });
        });

        // get a flat array of domains with nulls removed
        allDomains = _.compact(_.flatten(domainsList));

        return allDomains.length === _.unique(allDomains).length;
    }

    Ibp.Sourcetrak.Models.DomainSet = Ibp.Backbone.Model.extend({
        name: 'DomainSet',
        defaults: {
            name: 'New Domain Set',
            replacement_number: ''
        },

        names: {
            'name'                         : 'Name',
            'domains'                      : 'Domain',
            'phone_number'                 : 'Main Number',
            'css_id'                       : 'Number Replacement ID',
            'css_class'                    : 'Number Replacement Class',
            'replacement_number_whitelist' : 'Number to Replace',
            'anchor_tag_flag'              : 'Advanced Options',
            'vanity_number'                : 'Advanced Options'
        },

        validators: {
            'name'                   : [ { validate: 'required',                        message: 'Name cannot be blank.' } ],
            'phone_number'           : [ { validate: 'allowEmpty',                      message: null },
                                         { validate: 'phone',                           message: 'Main Number must be a valid phone number.' } ],
            'replacement_number'     : [ { validate: 'allowEmpty',                      message: null },
                                         { validate: validateReplacmentNumberUnique,    message: 'A location is already using that replacement number.'},
                                         { validate: 'phone',                           message: 'Enter a valid phone number.' } ],
            'domains.X._body.domain' : [ { validate: 'host',                            message: 'Enter the domain only.' },
                                         { validate: 'url',                             message: 'Enter a valid domain.' },
                                         { validate: validateDomainUnique,              message: 'This domain already exists.' } ]
        },

        _domains: null,

        url: function () {
            var url = BootstrappedData.apiUrlBase + 'sourcetrak/domainset';

            return this.isNew() ? url : url + '/' + this.id;
        },

        init: function () {
            _.bindAll(this);

            // if there is an error during save, call this._domains.resetSubModels()
            this._domains = new Ibp.Backbone.Collection.SubCollection(null, {
                parent: this,
                colName: 'domains'
            });

            this._whitelist = new Ibp.Sourcetrak.Collections.ReplacementNumbers(null, {
                parent: this,
                colName: 'replacement_number_whitelist'
            });

            this._blacklist = new Ibp.Sourcetrak.Collections.ReplacementNumbers(null, {
                parent: this,
                colName: 'replacement_number_blacklist'
            });

            this.listenTo(this, 'sync', this._domains.resetSubModels);

            this.listenTo(this._domains, 'add remove change', this.subCollectionHandler);
        },

        /**
         * Takes a key representing the piece of data to get, and returns the
         * corresponding result for this model. 0 is returned if there are no
         * results.
         *
         * @method getStats
         * @param key {String}
         * @return -varies-
         */
        getStats: function (key) {
            // find the stats object that matches this one
            var stats = _.find(BootstrappedData.domainSetStats, function (set) {
                    return set.id === this.get('id');
                }.bind(this));

            // default to zero if we don't have stats
            return stats ? stats[key] : 0;
        },

        getDomains: function () {
            return this._domains;
        },

        getWhitelist: function () {
            return this._whitelist;
        },

        getBlacklist: function () {
            return this._blacklist;
        },

        /**
         * Returns activities belonging to this domain set, filtered by any additional criteria provided
         *
         * @method getActivities
         * @param {Object} [options]
         * @return {Array} The list matching of Activity Models
         */
        getActivities: function (options) {
            var query = {
                    domain_set_id: this.get('id')
                },
                activities;

            if (options) {
                _.extend(query, options);
            }

            activities = Ibp.Sourcetrak.Data.Activities.find(query);

            // return the list of activities without the domain set activity
            return _.filter(activities, function (activity) {
                return activity.get('activity_type') !== 1;
            });
        },

        /**
         * Returns the default activity belonging to this domain set.
         *
         * @method getDefaultActivity
         * @return {Model} Activity model with type 1
         */
        getDefaultActivity: function () {
            var query = {
                domain_set_id: this.get('id'),
                activity_type: 1
            };

            // where method is used instead of findWhere because findWhere does not behave
            // as expected with this statement, returning an array of all activities rather than
            // the query's first match
            return Ibp.Sourcetrak.Data.Activities.where(query)[0];
        },

        /**
         * Returns locations belonging to this domain set
         *
         * @method getLocations
         * @return {Array} The list of Location Models for this domain set
         */
        getLocations: function () {
            var locations;

            if (this.isNew()) {
                locations = [];
            } else {
                locations = Ibp.Sourcetrak.Data.Locations.where({domain_set_id: this.id});
            }

            return locations;
        },

        /**
         * Validates the model and triggers a save() when the subcollection changes
         * We don't need to pass the changed attributes since the subcollection has already set() them on the model
         *
         * @method subCollectionHandler
         */
        subCollectionHandler: function (subModel) {
            // run validation
            if (!this.isValid()) {
                // roll back the change
                this.set('domains', this.previousAttributes().domains);

                subModel.collection.remove(subModel, {silent: true});

                return;
            }

            if (this.isNew()) {
                this.trigger('create', this);
            }
            else {
                this.collection.trigger('save', this);
            }
        },

        /**
         * Determines if the domain set is using multinumber replacement
         *
         * @method isMultiNumber
         * @return {Boolean}
         */
        isMultiNumber: function () {
            var whitelist = this.getWhitelist().pluck('phone_number'),
                blacklist = this.getBlacklist().pluck('phone_number');

            // any numbers in the blacklist indicate multinumber
            if (blacklist.length) {
                return true;
            }

            // more than a single number or use of the wildcard indicates multinumber
            if (_.contains(whitelist, '*') || whitelist.length > 1) {
                return true;
            }

            return false;
        }
    });
})();

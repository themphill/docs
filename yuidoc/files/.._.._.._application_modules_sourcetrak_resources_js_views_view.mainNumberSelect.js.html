Ibp.Sourcetrak.Views.MainNumberSelect = Ibp.Carbon.Views.EditableSelect.extend({
    name: 'MainNumberSelect',
    template: Ibp.Sourcetrak.Templates.main_number_select,
    select_template: Ibp.Sourcetrak.Templates.main_number_select_element,

    rpcUrl: '/v1/rpc',

    /**
     * The main number select element as rendered from the template
     *
     * @property renderedSelectElement
     * @type DOM Object
     */
    renderedSelectElement: null,

    /**
     * The Selected Main Number
     *
     * @property selectedNumber
     * @type String
     */
    selectedNumber: null,

    /**
     * Selector for the select wrapper
     *
     * @property selectWrapperEl
     * @type String
     */
    selectWrapperEl: '.js-select-element-wrapper',

    /**
     * jQuery reference for the select wrapper
     *
     * @property $selectWrapperElement
     * @type jQuery Object
     */
    $selectWrapperElement: null,

    /**
     * Array of number objects returned from the rpc endpoint
     *
     * @property unformattedChoices
     * @type Array of number objects
     */
    unformattedChoices: null,

    onInit: function () {
        this.getRegisteredNumbers()
            .always(function (res) {
                this.unformattedChoices = res;
                this.render();
            }.bind(this));
    },

    getRegisteredNumbers: function () {
        return $.getJSON(this.rpcUrl, {
            action: 'getRegisteredNumbers',
            includeIbp: 1
        });
    },

    formatChoices: function (numbers) {
        // convert the array of objects passed into this function into an array of number strings
        var allNumbers = _.pluck(numbers, 'number');

        // grab a list of all main numbers from our domain set collection
        var currentNumbers = Ibp.Sourcetrak.Data.DomainSets.pluck('phone_number');

        // see if the current domain set has a main number (we'll need to add this back into the list)
        var myNumber = this.model.get('phone_number');

        // remove all of the main numbers from other domain sets from our potential list
        var availableNumbers = _.difference(allNumbers, currentNumbers);

        // if this domain set has a main number already, add it to the front of the list
        if (myNumber) {
            availableNumbers.unshift(myNumber);
        }

        // return the formatted list so its friendly for the editable select view
        return _.map(availableNumbers, function (number) {
            return {
                description: Ibp.Utils.PhoneNumber().formatNumber(number).number,
                value: number
            };
        });
    },

    /**
     * Shows the edit select input
     * This is overriding the default showSelectInput function in Ibp.Carbon.Views.EditableSelect
     * specifically so that we can only build the select element only when it is needed. Such as the
     * first time the user attempts to interact with the editable control. This allows us to avoid
     * building it on page load, in order to avoid page hang when crunching thousands of main numbers.
     *
     * @method showSelectInput
     */
    showSelectInput: function () {
        // If the select input is currently rendered, ignore render request
        if (this.$selectText.hasClass('hidden')) {
            return;
        }

        // If this is the first time the select element is being rendered, or if the rendered
        // element no longer matches the selected number, regenerate the select element
        if (this.renderedSelectElement === null || (this.selectedNumber !== this.model.get('phone_number'))) {
            this.generateMainNumberSelectElement();
        }

        // Since the editable select re-renders itself on click, we need to make sure
        // we get new references to dom elements and also re-render the select element
        this.$selectWrapperElement = this.$(this.selectWrapperEl);
        this.$selectWrapperElement.html(this.renderedSelectElement);

        this.$selectInput = this.$(this.selectInputEl);

        this.$selectText.addClass('hidden');
        this.$selectInput.removeClass('hidden');
    },

    /**
     * Generate the Main Number Select Element
     * This allows us to delay the formatting of the choices and the building of the
     * DOM element until the user decides to interact with the control
     *
     * @method generateMainNumberSelectElement
     */
    generateMainNumberSelectElement: function() {
        var data = null;

        // format and store the choices
        this.choices = this.formatChoices(this.unformattedChoices);

        // build the data object for the template
        data = {
            namespace:         this.model.name.toLowerCase(),
            selectedValue:     this.model.get(this.selectAttribute),
            selectPlaceholder: this.escapeSpaces(this.selectPlaceholder),
            attribute:         this.selectAttribute,
            choices:           this.choices
        };

        // generate a main number select element
        this.renderedSelectElement = this.select_template(data);

        // keep track of the selected number that we have rendered
        this.selectedNumber = this.model.get('phone_number');
    }
});

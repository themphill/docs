Ibp.KeywordSpotting.Views.MetricTable = Ibp.Backbone.View.extend({
    name:        'metricTable',
    template:    Ibp.KeywordSpotting.Templates.metric_table,
    rows:        {},
    $tbody:      null,
    savePending: false,

    events: {
        'click .js-sort': 'sort'
    },

    subscriptions: {
        'metric:create': 'autoSave'
    },

    init: function(options) {
        // re-render on sort
        this.listenTo(this.collection, 'sort change reset add remove', this.rerender);

        this.render();
    },

    render: function() {
        this.$el.append(this.template());
        this.$tbody = $(this.$el.find(".table-body")[0]);

        this.renderRows(this.collection.models);
    },

    renderRows: function(rowCollection) {
        var _this = this;

        _.each(this.rows, function(row) {
            row.remove();
        });
        this.$tbody.html('');
        this.rows = {};

        _.each(rowCollection, function(metric) {
            _this.rows[metric.cid] = new Ibp.KeywordSpotting.Views.MetricTableRow({
                container: _this.$tbody,
                model:     metric
            });
        });

        if (rowCollection.length === 0) {
            this.rows.emptyRow = new Ibp.KeywordSpotting.Views.MetricTableRowEmpty({
                container: _this.$tbody
            });
        }
    },

    rerender: function() {
        this.renderRows(this.collection.models);
    },

    sort: function(e) {
        var $target = $($(e.target).closest('.js-sort')[0]),
            sortBy = $target.attr('id'),
            order = $target.attr('data-order'),
            newSort = this.getNewSort(sortBy, order);

        // reset sort headers and icons
        this.$el.find('.js-sort')
            .attr('data-order', 'default')
            .removeClass('active')
            .find('i').removeClass();

        // set new order, active, and icon
        $target.attr('data-order', newSort["newOrder"])
            .toggleClass('active', newSort["newOrder"] !== 'default')
            .find('i').addClass(newSort["newIcon"]);

        // sort collection
        this.collection.sortBy(sortBy, newSort["newOrder"]);
    },

    /**
     * Get sorts for the given column
     *
     * This is needed because, for some reason that is unexplained to me, keywords are
     * supposed to sort differently than EVERY OTHER COLUMN. https://rally1.rallydev.com/#/16667679310d/detail/defect/20369499070
     *
     * @param column
     * @param currentSort
     */
    getNewSort: function(column, currentSort) {
        var sorts = {
            // cycle sort order (desc > asc > default)
            'keywords': {
                'asc': {
                    'newOrder': 'default',
                    'newIcon': ''
                },
                'desc': {
                    'newOrder': 'asc',
                    'newIcon': 'icon-arrow-up5'
                },
                // first click on keywords should sort desc
                'default': {
                    'newOrder': 'desc',
                    'newIcon': 'icon-arrow-down4'
                }
            },
            // cycle sort order (asc > desc > default)
            'default': {
                'asc': {
                    'newOrder': 'desc',
                    'newIcon': 'icon-arrow-down4'
                },
                'desc': {
                    'newOrder': 'default',
                    'newIcon': ''
                },
                'default': {
                    'newOrder': 'asc',
                    'newIcon': 'icon-arrow-up5'
                }
            }
        },
        columnSort = sorts[column];

        return (columnSort == undefined) ? sorts["default"][currentSort] : columnSort[currentSort];
    },

    /**
     * Save metric to server
     *
     * @param Ibp.KeywordSpotting.Models.Metric metric
     */
    autoSave: function(metric) {
        var _this = this;

        // first determine if you are dealing with an unsaved new model, or an existing model
        if (metric.isNew()) {
            // listen for initial save, then save first change
            this.listenToOnce(metric, 'firstSave', function() {
                _this.savePending = true;
                _this.listenToOnce(metric, 'saved', function() {
                    _this.saveComplete();
                });

                // update the URL to reflect the new metric id
                Backbone.history.navigate("#" + metric.id, false);
            });

            // if the new model is not already being saved, save it
            // this save will trigger the listener above
            if (!this.savePending) {
                this.savePending = true;
                this.collection.autoSave(metric);
            }
        } else {
            if (this.savePending) {
                // The model is currently being saved, wait for successful save before
                // attempting to save
                this.listenToOnce(metric, 'saved', function() {
                    _this.savePending = true;
                    _this.listenToOnce(metric, 'saved', function() {
                        _this.saveComplete();
                    });
                });
            } else {
                this.savePending = true;
                this.listenToOnce(metric, 'saved', function() {
                    _this.saveComplete();
                });
            }
        }
    },

    saveComplete: function() {
        this.savePending = false;
    }
});

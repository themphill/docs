Ibp.Sourcetrak.Views.RoutingApp = Ibp.Carbon.Views.ModalPageView.extend({
    mixins: ['Autoset'],

    /**
     * The RPC endpoint for getting the list of phone numbers that can be used for caller ID
     *
     * @property callerIdUrl
     * @type String
     */
    callerIdUrl: '/v1/rpc',

    /**
     * The API endpoint for getting the list apps an account has
     *
     * @property appUrl
     * @type String
     */
    appUrl: null,

    /**
     * The comma separated list of fields to include the app API response
     *
     * @property fieldList
     * @type String
     */
    fieldList: null,

    /**
     * The response from fetching the apps via the API
     *
     * @property callerIds
     * @type Array
     */
    appList: null,

    /**
     * The list of phone numbers that can be used for caller ID
     *
     * @property callerIds
     * @type Array
     */
    callerIds: null,

    /**
     * The attributes specific to this action/application to set on the model
     *
     * @property appDefaults
     * @type Object
     */
    appDefaults: null,

    /**
     * The model attribute key specific to this action/application
     *
     * @property modelKey
     * @type String
     */
    modelKey: '',

    /**
     * The attribute key specific to this action/application as it comes back from the server
     *
     * @property serverKey
     * @type String
     */
    serverKey: '',

    init: function (options) {
        _.bindAll(this);

        this.$container = options.$container;

        this.appList = [];

        this.callerIds = [];

        if (this.appDefaults) {
            this.setAppDefaults();
        }

        this.onInit(options);
    },

    /**
     * Hook to allow custom code to run after init
     * render() should be called in here
     *
     * @method onInit
     * @param options {Object} The options passed in when view is created
     */
    onInit: function (options) {},

    /**
     * The attribute key specific to this action/application
     *
     * @method fetchApps
     * @param callback {Function} comma-separated list of fields to include in the response
     */
    fetchApps: function (callback) {
        return $.get(this.appUrl, {
            fl: this.fieldList,
            rows: 500
        }, function (res) {
            if (res.length) {
                this.appList = res;

                // use the first one as a default, if a current value doesn't exist
                if (this.model.get(this.modelKey) === null) {
                    this.model.set(this.modelKey, res[0][this.serverKey]);
                }
            }
        }.bind(this));
    },

    fetchCallerIds: function () {
        return $.getJSON(this.callerIdUrl, {
            action: 'getRegisteredNumbers',
            includeIbp: 1
        }, function (res) {
            this.callerIds = res;
        }.bind(this));
    },

    render: function () {
        var data = {
            appList:    this.appList,
            currentApp: parseInt(this.model.get(this.modelKey), 10),
            callerIds:  this.callerIds
        };

        _.defaults(data, this.model.toJSON());

        this.$el.html(this.template(data));

        this.$container.html(this.$el);

        if (Ibp.Utils.isIE()) {
            this.$el.addClass('no-touch');
            this.$('select').selecter({cover: true});
        }
    },

    /**
     * Set defaults for this action/application
     *
     * @method setAppDefaults
     */
    setAppDefaults: function () {
        var attrs = _.deepClone(this.model.attributes);

        _.defaults(attrs, this.appDefaults);

        this.model.set(attrs);
    }
});

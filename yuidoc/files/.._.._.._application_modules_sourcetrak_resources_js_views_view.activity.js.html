/**
 * Activity Card View
 * View container for a single Activity and swaps between create, edit, and display views
 *
 * @module SourceTrak
 * @class Views.Activity
 * @extends Ibp.Backbone.View
 */
Ibp.Sourcetrak.Views.Activity = Ibp.Backbone.View.extend({
    name: 'Activity',
    template: Ibp.Sourcetrak.Templates.activity,

    /**
     * Backbone Model to hold view state
     *
     * @property state
     * @type Model
     */
    state: null,

    /**
     * The views that correspond to each card state
     *
     * @property cardViews
     * @type Object
     */
    cardViews: {
        readOnly: Ibp.Sourcetrak.Views.ActivityReadOnly,
        display: Ibp.Sourcetrak.Views.ActivityDisplay,
        create: Ibp.Sourcetrak.Views.ActivityCreate,
        advanced: Ibp.Sourcetrak.Views.ActivityEditAdvanced,
        edit: Ibp.Sourcetrak.Views.ActivityEdit,
        destroy: Ibp.Sourcetrak.Views.ActivityDestroy,
        pool: Ibp.Sourcetrak.Views.ActivityPool,
        poolEdit: Ibp.Sourcetrak.Views.ActivityPoolEdit
    },

    /**
     * Maps card events to the next card they should flip to
     *
     * @property cardFlow
     * @type Object
     */
    cardFlow: {
        create: {
            advanced: 'advanced',
            create: 'display'
        },
        advanced: {
            create: 'create',
            edit: 'edit'
        },
        display: {
            edit: 'edit'
        },
        edit: {
            cancel: 'display',
            advanced: 'advanced',
            destroy: 'destroy'
        },
        destroy: {
            cancel: 'edit'
        },
        pool: {
            poolEdit: 'poolEdit'
        },
        poolEdit: {
            pool: 'pool'
        }
    },

    /**
     * The card View currently being shown
     *
     * @property currentView
     * @type View
     */
    currentView: null,

    /**
     * The the default card state to display
     *
     * @property DEFAULT_CARD
     * @type String
     */
    DEFAULT_CARD: 'display',

    /**
     * Calls render
     *
     * @method init
     */
    init: function (options) {
        this.$container = options.$container;

        this.state = new Ibp.Carbon.Models.State({
            currentCard: options.currentCard || this.DEFAULT_CARD
        });

        // listen for card flips and render new card
        this.listenTo(this.state, 'change:currentCard', this.handleCardChange);

        this.render();
    },

    /**
     * Create card container and renders the card
     *
     * @method render
     * @return this view
     */
    render: function () {
        // create activity container
        this.$el = $(this.template());

        // render the current card state
        this.renderCard();

        // render card
        if (this.$container) {
            this.$container.prepend(this.$el);
        }

        return this;
    },

    /**
     * Renders the current card state.
     *
     * @method renderCard
     */
    renderCard: function () {
        var View = this.cardViews[this.state.get('currentCard')];

        // in preparation for creating a new view, remove the old 'current view' if it exists
        if (this.currentView) {
            this.currentView.unrender();
            this.stopListening(this.currentView);
            this.currentView = null;
        }

        // set the appropriate saving mixin for advanced options
        if (this.state.get('currentCard') === 'advanced') {
            if (this.model.isNew()) {
                Cocktail.mixin(View, 'Autoset');
            }
            else {
                Cocktail.mixin(View, 'Autosave');
            }
        }

        this.currentView = new View({
            $container: this.$el,
            model: this.model
        });

        this.listenTo(this.currentView, 'flip', this.handleCardFlip);
    },

    /**
     * Handles the 'currentCard' state change event by
     * rendering the appropriate view, or removing this
     * view if the currentCard is 'cancel' on a new model (e.g. cancel create)
     *
     * @method handleCardChange
     */
    handleCardChange: function () {
        if (this.model.isNew() && this.state.get('currentCard') === 'cancel') {
            this.unrender();
        } else {
            this.renderCard();
        }
    },

    /**
     * Handles the "flip" child view event
     *
     * @method handleCardFlip
     */
    handleCardFlip: function (data) {
        this.flip(data);
    },

    /**
     * Determines the next card to show and sets it as the new state
     *
     * @method flip
     */
    flip: function (data) {
        var card = data.card,
            flipTo = data.event,
            nextCard = this.cardFlow[card][flipTo];

        this.state.set({currentCard: nextCard});
    },

    show: function () {
        this.$el.show();
    },

    hide: function () {
        this.$el.hide();
    },

    onUnrender: function () {
        this.trigger('remove');
        this.currentView.unrender();
    }
});

 Ibp.Sourcetrak.Views.TestDomainSetModal = Ibp.Carbon.Views.Modal2.extend({
    name: 'testDomainSetModal',
    template: Ibp.Sourcetrak.Templates.modal_test_domain_set,
    noMatchTemplate: Ibp.Sourcetrak.Templates.test_no_match,

    /**
     * Stores the DOM selector for the referer input
     * @property refererEl
     * @type String
     */
    refererEl: '.js-referer',

    /**
     * Stores the jQuery Object pointing to the referer input
     * @property $referer
     * @type jQuery Object
     */
    $referer: null,

    /**
     * Stores the DOM selector for the url tag input
     * @property tagEl
     * @type String
     */
    tagEl: '.js-tag',

    /**
     * Stores the jQuery Object pointing to the url tag input
     * @property $tag
     * @type jQuery Object
     */
    $tag: null,

    /**
     * Stores the DOM selector for the search query input
     * @property searchEl
     * @type String
     */
    searchEl: '.js-search',

    /**
     * Stores the jQuery Object pointing to the search query input
     * @property $search
     * @type jQuery Object
     */
    $search: null,

    /**
     * Stores the DOM selector for the result output DIV
     * @property resultEl
     * @type String
     */
    resultEl: '.js-result',

    /**
     * Stores the jQuery Object pointing to the result output DIV
     * @property $result
     * @type jQuery Object
     */
    $result: null,

    /**
     * Api URL for testing configuration parameters
     * @property testUrl
     * @type String
     */
    testUrl: '/st/test',

    /**
     * Holds a pointer to the activity view of the current match
     * @property resultView
     * @type Model
     */
    resultView: null,

    /**
     * Stores the constant for the main number activity, as we don't display a card for these matches
     * @property MAIN_NUMBER_TYPE
     * @type Integer
     */
    MAIN_NUMBER_TYPE: 1,

    /**
     * DOM Event handlers
     *
     * @property events
     */
    modalEvents: {
        'keyup .js-input-list input': 'handleKeyup',
        'click .js-submit': 'handleSubmit'
    },

    /**
     * Calls render
     *
     * @method init
     */
    init : function (options) {
        return this.render();
    },

    /**
     * Refactored render function from Ibp.Carbon.Views.Modal
     * Handles removing existing modal, rendering this modal, and calling its hook function
     *
     * @method render
     * @return this view
     */
    render: function () {
        // create modal fragment
        this.$el = $(this.template(this.templateData));

        // add this modal to the DOM
        $('body').append(this.$el);

        this.$referer = this.$(this.refererEl);
        this.$tag = this.$(this.tagEl);
        this.$search = this.$(this.searchEl);
        this.$result = this.$(this.resultEl);

        this.$('i[title]').tooltip({placement: 'left'});

        return this;
    },

    /**
     * Checks a keyup on an input to see if it was an "enter" to perform a search
     *
     * @method handleKeyup
     */
    handleKeyup: function (e) {
        var keyCode = e.keyCode || e.which;

        // only trigger save on 'enter' or 'tab'
        if (keyCode && keyCode !== 13 ) {
            return;
        }

        this.search();
    },

    /**
     * Responds to a click on the submit input
     *
     * @method handleSubmit
     */
    handleSubmit: function () {
        this.search();
    },

    /**
     * Pings the server to see if an activity matches the given parameters
     *
     * @method search
     */
    search: function () {
        // clean up the current view
        if (this.resultView) {
            this.resultView.remove();
            delete this.resultView;
        }
        this.$result.html('');

        // check the server for a matching activity, then display the results
        this.lookupActivity()
        .then(this.displayResults);
    },

    /**
     * Passes the entered parameters to the server to see which activity best fits the scenario
     *
     * @method lookupActivity
     */
    lookupActivity: function () {
        // return a deferred object so we can chain the response with the display function
        return $.getJSON(this.testUrl, {
            _stk: BootstrappedData.publicKey,
            dsi: this.model.get('id'),
            r: this.$referer.val(),
            t: this.$tag.val(),
            q: this.$search.val()
        });
    },

    /**
     * Updates the view with the results from an activity lookup AJAX request
     *
     * @method displayResults
     * @param res - {Object}
     */
    displayResults: function (res) {
        // holds the matching activity if we find one
        var activity = Ibp.Sourcetrak.Data.Activities.get(res.activityId);

        if (res.activityId === null
            || activity === undefined
            || activity.get('activity_type') === this.MAIN_NUMBER_TYPE) {
            // we didn't match any activities, so populate the results div with the no match text
            this.renderNoMatch();
        } else {
            // matching activity found
            this.renderActivity(activity);
        }
    },

    /**
     * Populates the results container with the contents of the no-match template
     *
     * @method renderNoMatch
     */
    renderNoMatch: function () {
        this.$result.html(this.noMatchTemplate());
    },

    /**
     * Populates the results container with an activity card view
     *
     * @method renderActivity
     * @param activity - Model
     */
    renderActivity: function (activity) {
        // we have an activity to display, so initialize a new card view
        this.resultView = new Ibp.Sourcetrak.Views.Activity({
            model: activity,
            $container: this.$result,
            currentCard: 'readOnly'
        });
    }
});

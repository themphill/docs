Ibp.Sourcetrak.Views.EditDomainSet = Ibp.Carbon.Views.PageView.extend({
    mixins: ['Autosave', 'HighlightInvalid'],
    name: 'editDomainSet',
    el: '#sourcetrak-editDomainSet',
    template: Ibp.Sourcetrak.Templates.edit_domain_set,

    /**
     * Object containing this view's associated views
     *
     * @property views
     */
    views: {},

    /**
    * A subCollection object that is an array of objects,
    * each representing a domain model
    *
    * @property domainsSubCollection
    * @type SubCollection
    */
    domainsSubCollection: null,

    /**
    * DOM container for domain set's domains list
    *
    * @property domainsEl
    * @type String
    */
    nameEl: '.js-edit-name',

    /**
    * DOM container for main number select
    *
    * @property mainNumberEl
    * @type String
    */
    mainNumberEl: '.js-edit-main-number',

    /**
    * DOM container for domain set's domains list
    *
    * @property domainsEl
    * @type String
    */
    domainsEl: '.js-edit-domains',

    /**
    * DOM container for number replacement number
    *
    * @property replaceNumberEl
    * @type String
    */
    replaceNumberEl: '#replace_number',

    /**
    * DOM container for number replacement class
    *
    * @property replaceClassEl
    * @type String
    */
    replaceClassEl: '#replace_class',

    /**
    * DOM container for number replacement id
    *
    * @property replaceIdEl
    * @type String
    */
    replaceIdEl: '#replace_id',

    /**
    * DOM container for set of elements that comprise
    * the presentation & functionality for adding a new
    * domain to the domain set model.
    *
    * @property domainForm
    * @type String
    */
    domainForm: '.js-domain-form',

    /**
     * DOM container for one-to-one activity list
     *
     * @property otoContainerEl
     * @type String
     */
    otoContainerEl: '.js-oto-activity-list',

    /**
     * DOM container for pool activity list
     *
     * @property poolContainerEl
     * @type String
     */
    poolContainerEl: '.js-pool-activity-list',

    /**
     * DOM container for locations list
     *
     * @property locationsListContainerEl
     * @type String
     */
    locationsListContainerEl: '.js-locations-section',

    /**
     * References locationsListContainerEl
     *
     * @property $locationsListContainer
     * @type jQuery Object
     */
    $locationsListContainer: null,
    /**
     * Eventhub subscriptions handlers
     *
     * @property subscriptions
     */

    subscriptions: {
        'deleteDomainSetModal'   : 'renderDeleteDomainSetModal',
        'getAdvancedOptionsModal': 'renderAdvancedOptionsModal',
        'getTestModal'           : 'renderTestModal',
        'refreshActivitiesList'  : 'handleRefreshActivitiesList'
    },

    /**
     * DOM Event handlers
     *
     * @property events
     */
    events:  {
        'click .js-cancel': 'hideNewDomainForm',
        'click .js-add'   : 'showNewDomainForm'
    },

    /**
     * Init
     *
     * @method init
     */
    init: function () {
        this.domainsSubCollection = this.model.getDomains();

        // if this is a new model, we want to know the first time it is synced so we can show the add activity cards
        if (this.model.isNew()) {
            this.listenToOnce(this.model, 'sync', this.renderLists);
        }

        this.render();
    },

    /**
     * Render
     *
     * @method render
     * @return this view
     */
    render: function () {
        var data = {
            domainsList: [],
            model: this.model.toJSON(),
            mainNumber: this.model.get('phone_number')
        };

        this.$el.html(this.template(data));

        this.$locationsListContainer = this.$(this.locationsListContainerEl);

        this.renderToolbar();
        this.renderName();
        this.renderDomains();
        this.renderMainNumber();
        this.renderFindAndReplace();
        this.renderLists();

        this.$('i[title]').tooltip({placement: 'right'});

        return this;
    },

    /**
     * renderToolbar
     *
     * @method renderToolbar
     */
    renderToolbar: function () {
        this.views.toolbar = new Ibp.Sourcetrak.Views.DomainSetToolbar({
            model: this.model
        });
    },

    /**
     * RenderAdvancedOptionsModal displays the modal overlay that will contain configuration
     * options such as negative keywords, IP exclusions, and Toggle A tag on mobile.
     *
     * @method renderAdvancedOptionsModal
     */
    renderAdvancedOptionsModal: function ()  {
        this.publish('modal:render', {name: 'advancedOptions', data: {model: this.model}});
    },

    /**
     * RenderDeleteDomainSetModal displays teh modal overlay that will ask a
     * user to confirm that they actually want to delete this entire Domain Set.
     * Once the user confirms, this function publishes a deleteDomainSet event and
     * acts upon the model specified in the data object.
     *
     * @method renderDeleteDomainSetModal
     */
    renderDeleteDomainSetModal: function() {
        this.publish('modal:render', {name: 'deleteDomainSet', data: {model: this.model}});
    },

    /**
     * Renders the configuration test modal that allows users to input
     * request parameters and match activities.
     *
     * @method renderTestModal
     */
    renderTestModal: function() {
        this.publish('modal:render', {name: 'testDomainSet', data: {model: this.model}});
    },

    /**
     * Renders the domain set name editable label
     *
     * @method renderName
     */
    renderName: function () {
        this.views.name = new Ibp.Carbon.Views.EditableLabel({
            $container: this.$(this.nameEl),
            model: this.model,
            labelAttribute: 'name'
        });
    },

    /**
     * Renders the main number editable select list
     *
     * @method renderMainNumber
     */
    renderMainNumber: function () {
        this.views.mainNumber = new Ibp.Sourcetrak.Views.MainNumberSelect({
            $container: this.$(this.mainNumberEl),
            selectName: 'Main Number',
            selectAttribute: 'phone_number',
            selectPlaceholder: 'Select Main Number...',
            model: this.model,
            tooltipDirection: 'top'
        });
    },

    /**
     * Renders the number replacement options editable labels
     *
     * @method renderFindAndReplace
     */
    renderFindAndReplace: function() {
        var _this = this,
            findAndReplaceProperties = [
                {
                    el: this.replaceClassEl,
                    label: 'CSS Class',
                    attribute: 'css_class',
                    placeholder: 'Enter Class',
                    maskPhoneNumber: false
                },
                {
                    el: this.replaceIdEl,
                    label: 'CSS ID',
                    attribute: 'css_id',
                    placeholder: 'Enter ID',
                    maskPhoneNumber: false
                }
            ];

        // render replacement number
        if (this.model.isMultiNumber()) {
            this.views.replacement_number = new Ibp.Sourcetrak.Views.MultiNumberLabel({
                $container: $(this.replaceNumberEl)
            });
        }
        else {
            this.views.replacement_number = new Ibp.Sourcetrak.Views.MultiNumberEditableLabel({
                $container: this.$(this.replaceNumberEl),
                labelName: 'Number',
                labelAttribute: 'phone_number',
                labelPlaceholder: 'Enter Phone Number',
                model: this.model.getWhitelist().at(0),
                maskPhoneNumber: true
            });
        }

        // render class and id
        _.each(findAndReplaceProperties, function (property) {
            _this.views[property.attribute] = new Ibp.Carbon.Views.EditableLabel({
                $container: _this.$(property.el),
                labelName: property.label,
                labelAttribute: property.attribute,
                labelPlaceholder: property.placeholder,
                model: _this.model,
                maskPhoneNumber: property.maskPhoneNumber
            });
        });
    },

    /**
     * Renders the list of domains to the template
     *
     * @method renderDomains
     */
    renderDomains: function () {
        this.views.domains = new Ibp.Sourcetrak.Views.EditDomains({
            $container: this.$(this.domainsEl),
            collection: this.model.getDomains()
        });
    },

    /**
     * Handles the rendering of the locations and lists for both oto activites
     * and pool activities.
     *
     * @method renderLists
     */
    renderLists: function () {
        // Render both activities lists
        this.renderOtoActivitiesList();
        this.renderPoolActivitiesList();

        // Render the locations list
        this.renderLocationsList();
    },

    /**
     * Handles the rendering of the Pool Activities List
     *
     * @method renderPoolActivitiesList
     */
    renderPoolActivitiesList: function () {
        // if view already exists, blow it away before rendering fresh version
        if (this.views.poolActivityList) {
            this.views.poolActivityList.unrender();
        }

        // render the activity list that will contain only pooled activities
        this.views.poolActivityList = new Ibp.Sourcetrak.Views.ActivityList({
            $container: this.$(this.poolContainerEl),
            namespace: 'pool',
            label: 'Pool Activities',
            domainSet: this.model,
            isOto: false
        });
    },

    /**
     * Handles the rendering of the OtO Activities List
     *
     * @method renderOtoActivitiesList
     */
    renderOtoActivitiesList: function () {
        // if view already exists, blow it away before rendering fresh version
        if (this.views.otoActivityList) {
            this.views.otoActivityList.unrender();
        }

        // render the activity list that will contain only one-to-one activities
        this.views.otoActivityList = new Ibp.Sourcetrak.Views.ActivityList({
            $container: this.$(this.otoContainerEl),
            namespace: 'oto',
            label: 'One-To-One Activities',
            domainSet: this.model,
            isOto: true
        });
    },

    /**
     * Renders locations to the template.
     *
     * @method renderLocationsList
     */
    renderLocationsList: function () {
        if (this.views.locationsList) {
            this.views.locationsList.unrender();
        }

        this.views.locationsList = new Ibp.Sourcetrak.Views.LocationsList({
            $container: this.$locationsListContainer,
            domainSet: this.model
        });
    },

    /**
     * Renders only the list with the activity that was just synced.
     *
     * @method handleActivitiesListRefresh
     * @params model {Backbone Model}
     */
    handleRefreshActivitiesList: function (model) {
        if (model.get('one_to_one')) {
            this.renderOtoActivitiesList();
        } else {
            this.renderPoolActivitiesList();
        }
    },

    /**
     * hideNewDomainForm
     * Hides the add new domain form from view
     *
     * @method hideNewDomainForm
     */
    hideNewDomainForm: function () {
        $(this.domainForm).addClass('hidden');
    },

    /**
     * showNewDomainForm
     * renders the form for adding a new domain to the domain set
     *
     * @method showNewDomainForm
     */
    showNewDomainForm: function () {
        $(this.domainForm).removeClass("hidden").find('input').focus();
    },

    /**
     * Remove clears the edit element and removes listeners
     *
     * @method remove
     * @return this view
     */
    remove: function() {
        if (this.model.isNew()) {
            this.model.trigger('remove', this.model);
        }

        this.$el.html('');
        this.stopListening();
        this.undelegateEvents();

        return this;
    }
});

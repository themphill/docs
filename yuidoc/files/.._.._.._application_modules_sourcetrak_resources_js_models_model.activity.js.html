/**
 * Activity Model
 *
 * @module SourceTrak
 * @class Models.Activity
 * @extends Ibp.Backbone.Model
 */
;(function () {
    function validateDomain(val) {
        if (!val || !_.isString(val)) {
            return false;
        }

        // the first and last characters can be *, indicating wildcards. remove these before validating if present
        var deStarred = val.replace(/^\*|\*$/g, '');

        return Backbone.Validator.host(deStarred);
    }

    function validatePath(val) {
        if (!val || !_.isString(val)) {
            return false;
        }

        // make sure the path forms a valid url
        var path = val.charAt(0) === '/' ? val : '/' + val,
            url  = 'example.com' + path;

        return Backbone.Validator.url(url);
    }

    function validateRequired(val) {
        return Backbone.Validator.required(val);
    }

    function validateKeyword(val, context) {
        // the only reason this works is because we set() the activity type on the model before a create
        // and you can't change the type again during edits
        var typeId    = context.attributes.activity_type || context.model.get('activity_type'),
            validator = _.findWhere(context.model.activityTypes, {id: typeId}).validator;

        // if there is no validator, this type doesn't require validation
        return validator ? validator(val) : true;
    }

    function keywordMessage(context) {
        var typeId  = context.attributes.activity_type || context.model.get('activity_type'),
            label   = _.findWhere(context.model.activityTypes, {id: typeId}).label.toLowerCase();

        return 'Enter a valid ' + label + '.';
    }

    Ibp.Sourcetrak.Models.Activity = Ibp.Backbone.Model.extend({
        mixins: ['Transform'],
        name: 'Activity',

        searchString: '',

        defaults: {
            keyword: null,
            activity_type: 2,
            pool_id: null,
            sourceguard_flag: 1
        },

        validators: {
            'keyword': [ { validate: validateKeyword, message: keywordMessage } ],
            'pool_id': [ { validate: 'required',      message: 'You must select a Pool.' } ]
        },

        transforms: {
            'activity_type': 'number',
            'pool_id': 'number'
        },

        names: {
            // keyword is handled in save()
            'pool_id': 'Pool',
            'platform': 'Platform',
            'campaign': 'Campaign',
            'min_numbers': 'Reserved Numbers'
        },

        /**
         * The activity types this activity can be
         *
         * @property poolTypes
         */
        activityTypes: [
            {
                id:          1,
                description: 'Direct Default'
            },
            {
                id:          2,
                description: 'Referrer Domain',
                label:       'Domain',
                validator:   validateDomain
            },
            {
                id:          3,
                description: 'Tagged URL',
                label:       'Tag',
                validator:   validateRequired
            },
            {
                id:          4,
                description: 'Organic Search Keyword',
                label:       'Keyword',
                validator:   validateRequired
            },
            {
                id:          5,
                description: 'All Referral Traffic',
                label:       'Keyword'
            },
            {
                id:          6,
                description: 'URL Path',
                label:       'Path',
                validator:   validatePath
            }
        ],

        /**
         * The pool types this activity can have
         *
         * @property poolTypes
         */
        poolTypes: [
            'pool',
            'one-to-one'
        ],

        /**
         * The platform types this activity can have
         *
         * @property platformTypes
         */
        platformTypes: [
            {
                id:          'google',
                description: 'Google AdWords'
            },
            {
                id:          'bing',
                description: 'Bing Ads'
            },
            {
                id:          'facebook',
                description: 'Facebook'
            },
            {
                id:          'doubleclick',
                description: 'DoubleClick'
            },
            {
                id:          'twitter',
                description: 'Twitter'
            },
            {
                id:          'linkedin',
                description: 'LinkedIn'
            },
            {
                id:          'aol',
                description: 'AOL'
            },
            {
                id:          'adroll',
                description: 'AdRoll'
            }
        ],

        /**
         * The type of this card instance
         *
         * @property poolType
         */
        _poolType: null,

        init: function () {
            _.bindAll(this);

            // store a compressed string that we use when searching and filtering
            this.compileSearchString();

            this.listenTo(this, 'sync', this.compileSearchString);
        },

        url: function () {
            var url = BootstrappedData.apiUrlBase + 'sourcetrak/activity';

            return this.isNew() ? url : url + '/' + this.id;
        },

        /**
         * Gets the Activity type info
         *
         * @method getActivityType
         * @return {String}
         */
        getActivityType: function () {
            return _.findWhere(this.activityTypes, {id: this.get('activity_type')});
        },

        /**
         * Gets the type of Pool this Activity belongs to
         *
         * @method getPoolDescription
         * @return {String}
         */
        getPoolType: function () {
            return this.get('one_to_one') ? this.poolTypes[1] : this.poolTypes[0];
        },

        /**
         * Gets the platform info
         *
         * @method getActivityType
         * @return {String}
         */
        getPlatformType: function () {
            return _.findWhere(this.platformTypes, {id: this.get('platform')});
        },

        /**
         * Gets this Activity's Domain Set Model
         *
         * @method getDomainSet
         * @return Model
         */
        getDomainSet: function () {
            return Ibp.Sourcetrak.Data.DomainSets.findWhere({id: this.get('domain_set_id')});
        },

        /**
         * Gets this Activity's Pool Model
         *
         * @method getPool
         * @return Model
         */
        getPool: function () {
            return Ibp.Sourcetrak.Data.Pools.findWhere({id: this.get('pool_id')});
        },

        /**
         * Gets the stats for this Activity
         *
         * @method getStats
         * @param key {String} The name of the stat to get
         * @return Mixed
         */
        getStats: function (key) {
            var activities = this.getDomainSet().getStats('activities'),
                stats = _.findWhere(activities, {id: this.get('id')});

            return stats ? stats[key] : 0;
        },

        /**
         * Checks for default activity type
         *
         * @method isMainNumberActivity
         * @return Boolean
         */
        isMainNumberActivity: function () {
            return this.get('activity_type') === 1;
        },

        /**
         * Compile and store the string that will be used for search filters
         * @method compileSearchString
         */
        compileSearchString: function () {
            if (this.isNew() || this.get('activity_type') === 1) {
                return;
            }

            var rawString = this.get('platform')
                + this.get('campaign')
                + this.get('keyword')
                + this.getPool().get('description')
                + this.getActivityType().description;

            this.searchString = rawString.replace(/-|\(|\)|\s/g, "").toLowerCase();
        }
    });
})();

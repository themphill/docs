Ibp.Utils.numToWord = function (num) {
    var ones = ['', 'One', 'Two', 'Three', 'Four', 'Five', 'Six', 'Seven', 'Eight', 'Nine', 'Ten',
                'Eleven', 'Twelve', 'Thirteen', 'Fourteen', 'Fifteen', 'Sixteen', 'Seventeen', 'Eighteen', 'Nineteen'],
        tens = ['', '', 'Twenty', 'Thirty', 'Forty', 'Fifty', 'Sixty', 'Seventy', 'Eighty', 'Ninety'],
        words = '';

    if (num === 0) {
        words = 'zero';
    } else if (num < 20) {
        words = ones[num];
    } else if (num >= 20 && num < 100) {
        words = tens[Math.floor(num / 10)] + ' ' + ones[num % 10];
    } else if (num >= 100 && num < 1000) {
        words = ones[Math.floor(num / 100)] + ' hundred'
                + (tens[Math.floor((num / 10) % 10)] !== '' ? ' ' + tens[Math.floor((num / 10) % 10)] : '')
                + ' ' + ones[num % 10];
    } else {
        words += num;
    }

    return words;
};

Ibp.Utils.PhoneNumber = function (options) {
    var that               = {},
        $numberInput       = null,
        maxLength          = 16,
        regionCode         = 'US',
        formattedNumber    = '',
        formattedCaret     = null,
        phoneUtil          = i18n.phonenumbers.PhoneNumberUtil.getInstance(),
        phoneNumberFormat  = i18n.phonenumbers.PhoneNumberFormat,
        asYouTypeFormatter = new i18n.phonenumbers.AsYouTypeFormatter(regionCode);

    that.init = function (options) {
        if (options.hasOwnProperty('maxLength')) {
            maxLength = options.maxLength;
        }
        if (options.hasOwnProperty('el')) {
            $numberInput = options.el;
            $numberInput.attr('maxlength', maxLength);
            $numberInput.val( (that.formatNumber($numberInput.val(), 0)).number );
        }
        if (options.hasOwnProperty('regionCode')) {
            regionCode = options.regionCode;
        }
        if (options.hasOwnProperty('addListener') && options.addListener === true) {
            $numberInput.bind({
                'keyup': that.inputListener,
                'paste': that.inputListener,
                'cut': that.inputListener
            });
        }
    };

    that.inputListener = function () {
        var number = $numberInput.val();
        if (number == formattedNumber) { return; }

        var caret     = $numberInput.caret(),
            formatted = that.formatNumber(number, caret);

        formattedNumber = formatted.number.trim();
        formattedCaret  = formatted.caret;

        that.updateNumber($numberInput);
    };

    that.updateNumber = function (el) {
        var $el = el !== undefined ? el : $numberInput;

        $el.val(formattedNumber);
        $el.caret(formattedCaret);
    };

    that.formatNumber = function (number, caret) {
        var formatted = '',
            c = '',
            d = 0,
            numberLength;

        asYouTypeFormatter.clear();

        number = i18n.phonenumbers.PhoneNumberUtil.convertAlphaCharactersInNumber(number);
        numberLength = number.length;

        for (var i=0; i < numberLength; i++) {
            c = number.charAt(i);
            d = number.charCodeAt(i);
            if (!(d >= 48 && d <= 57) && c != '+') { continue; }

            formatted = (i < caret) ? asYouTypeFormatter.inputDigitAndRememberPosition(c) : asYouTypeFormatter.inputDigit(c);
        }

        return {
            number: formatted,
            caret: asYouTypeFormatter.getRememberedPosition() || caret
        };
    };

    that.isValidNumber = function (number) {
        if (!number) {
            return false;
        }
        try {
            var parsedNumber = phoneUtil.parseAndKeepRawInput(number, regionCode);
            return phoneUtil.isValidNumber(parsedNumber);
        } catch (err) {
            // if the validator throws an exception, this isn't a phone number
            return false;
        }
    };

    that.normalize = function (number) {
        // returns phone number without mask
        return number ? i18n.phonenumbers.PhoneNumberUtil.normalize(number) : '';
    };

    if (options !== undefined) {
        that.init(options);
    }

    return that;
};

Ibp.Utils.truncate = function (str, num, addEllipsis) {
    var len         = num || 27,
        // default to using ellipsis for backwards compatibility
        ellipsis    = addEllipsis === false ? '' : '...',
        ellipsisLen = ellipsis.length;

    if (!str || str.length <= len) {
        return str;
    }

    return str.slice(0, len - ellipsisLen) + ellipsis;
};

/**
 * Determines if the browser is IE via user-agent sniffing
 * Used in cases where feature detection is not an option
 *
 * @method isIE
 * @return Boolean
 */
Ibp.Utils.isIE = function () {
    return ((navigator.appName === 'Microsoft Internet Explorer') || ((navigator.appName === 'Netscape')
            && (new RegExp("Trident/.*rv:([0-9]{1,}[\.0-9]{0,})").exec(navigator.userAgent) != null)));
};

Ibp.Utils.objectPathToObject = function (objectPath) {
    var elements = objectPath.split("."),
        object = window;

    for (var i = 0; i < elements.length; i++) {
        if ((i < (elements.length-1))
            && (typeof(object[elements[i]]) !== "object")) {
            object = object[elements[i]] = new Object();
        } else {
            object = object[elements[i]];
        }
    }

    return object;
};

/**
 * Looks at the contents of the given iFrame and parse the contents into JSON.
 *
 * @method parseIframeBody
 * @return {Object}
 */
Ibp.Utils.parseIframeBody = function (iframeId) {
    var iframe      = document.getElementById(iframeId),
        theDoc      = iframe.contentWindow.document,
        preElements = theDoc.getElementsByTagName('PRE'),
        justTheDoc;

    if (preElements.length > 0) {
        justTheDoc = preElements[0].firstChild.data;
    }
    else {
        justTheDoc = iframe.contentDocument.documentElement.textContent;
    }

    return JSON.parse(justTheDoc);
};

/**
 * Allows pluck() to be called on a normal array of Models
 *
 * @method pluckCollection
 * @return Array
 */
Backbone.pluckCollection = function (models, key) {
    return _.invoke(models, 'get', key);
};

/**
 * Allows sortBy() to be called on a normal array of Models
 * Additionally allows the array to be returned in descending order
 *
 * @method sortCollection
 * @return Array
 */
Backbone.sortCollection = function (models, key, desc) {
    var activities = _.sortBy(models, function (model) {
        return model.get(key);
    });

    if (desc === 'desc') {
        return activities.reverse();
    }

    return activities;
};

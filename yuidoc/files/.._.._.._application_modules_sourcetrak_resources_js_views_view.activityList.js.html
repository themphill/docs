/**
 * Domain Set View for displaying both the sets of one-to-one and pool activities
 *
 * @module SourceTrak
 * @class Views.ActivityList
 * @extends Ibp.Backbone.View
 */
Ibp.Sourcetrak.Views.ActivityList = Ibp.Backbone.View.extend({
    name: 'activityList',
    template: Ibp.Sourcetrak.Templates.activity_list,

    /**
     * The views object retains references to children views
     *
     * @property views
     * @type Object
     */
    views: null,

    /**
     * A list of references to our child activity views
     *
     * @property activityViews
     * @type Object
     */
    activityViews: null,

    /**
     * A state model object we use for tracking filter activity
     *
     * @property state
     * @type Model
     */
    state: null,

    /**
     * The activities being displayed
     *
     * @property paginatedActivities
     * @type Collection
     */
    paginatedActivities: null,

    /**
     * String storing the DOM element that this view will populate
     *
     * @property $container
     * @type jQuery Object
     */
    $container: null,

    /**
     * Human-readable string describing the type of activities in this list
     *
     * @property label
     * @type String
     */
    label: null,

    /**
     * Namespace string used for marking the class/id properties of DOM elements in this template
     *
     * @property namespace
     * @type String
     */
    namespace: null,

    /**
     * String selector for this view's search box DOM element
     * Initialized using the this.namespace
     *
     * @property searchBoxEl
     * @type String
     */
    searchBoxEl: null,

    /**
     * String selector for this view's filter alert DOM element
     *
     * @property filterAlertEl
     * @type String
     */
    filterAlertEl: '.js-filter-alert',

    /**
     * Holds a jQuery reference to the filter DOM element
     *
     * @property $filterAlert
     * @type jQuery Object
     */
    $filterAlert: null,

    /**
     * String selector for this view's activities list
     *
     * @property activitiesEl
     * @type String
     */
    activitiesEl: '.js-activities',

    /**
     * Holds a jQuery reference to the activities list container
     *
     * @property $activities
     * @type jQuery Object
     */
    $activities: null,

    /**
     * String selector for the toolbar link to add a new activity
     *
     * @property addActivityBtnEl
     * @type String
     */
    addActivityBtnEl: '.js-add-activity-btn',

    /**
      * jQuery object for the toolbar link to add a new activity
      *
      * @property $addActivityBtn
      * @type jQuery Object
      */
    $addActivityBtn: null,

    /**
     * String selector for the toolbar link to add a new activity
     *
     * @property addActivityCardEl
     * @type String
     */
    addActivityCardEl: '.js-add-activity-card',

    /**
      * jQuery object for the toolbar link to add a new activity
      *
      * @property $addActivityCard
      * @type jQuery Object
      */
    $addActivityCard: null,

    /**
     * String selector for container holding number of activities
     * displayed out of total activities
     *
     * @property activitiesCountEl
     * @type String
     */
    activitiesCountEl: '.js-activities-count',

    /**
     * Holds a jQuery reference to activitiesCountEl
     *
     * @property $activityCount
     * @type jQuery Object
     */
    $activitiesCount: null,

    /**
     * String selector for the paginator buttons and count
     *
     * @property paginatorControlsEl
     * @type String
     */
    paginatorControlsEl: '.js-paginator-controls',

    /**
     * Paginator controls jQuery object
     *
     * @property $paginatorControls
     * @type jQuery Object
     */
    $paginatorControls: null,

    /**
     * DOM Event handlers
     *
     * @property events
     * @type Object
     */
    events: {
        'click .js-filter-alert-dismiss': 'handleDismissFilterAlert',
        'click .js-add-activity': 'handleAddActivityClick'
    },

    /**
     * Init
     *
     * @method init
     * @param options {Object}
     */
    init: function (options) {
        _.bindAll(this);

        // Pull a number of options into local properties
        this.$container = options.$container;
        this.domainSet  = options.domainSet;
        this.namespace  = options.namespace;
        this.label      = options.label;
        this.isOto      = options.isOto;

        // use the label to create a unique class for the searchbox el
        this.searchBoxEl = '.js-' + this.namespace + '-search-box';

        this.views = {};
        this.activityViews = [];

        // create a collection of activities that can be paginated
        this.paginatedActivities = new Ibp.Sourcetrak.Collections.PaginatedActivities(
            this.getActivities()
        );

        // prime the paginator state
        this.paginatedActivities.setPageState();

        this.state = new Ibp.Carbon.Models.State({
            query: '',
            showCreateCard: false,
            showAddButtons: this.getButtonsState(),
            showControls: this.shouldShowControls(),
            renderActivities: true
        });

        this.listenTo(this.state, 'change', this.handleViewStateChange);
        this.listenTo(this.paginatedActivities.ClientPagination.state, 'change', this.handlePageStateChange);

        // handle adding and removing activities
        this.listenTo(Ibp.Sourcetrak.Data.Activities, 'add remove', this.handleActivitiesChange);

        // allow users to create activities once a new domain set has been saved
        if (this.domainSet.isNew()) {
            this.listenToOnce(this.domainSet, 'sync', this.handleCreateDomainSet);
        }

        // render this view
        this.render();
    },

    /**
     * Render
     *
     * @method render
     * @return this view
     */
    render: function () {
        var data = {
            label: this.label,
            namespace: this.namespace
        };

        this.$el = $(this.template(data));

        this.$container.append(this.$el);

        this.$filterAlert       = this.$(this.filterAlertEl);
        this.$activities        = this.$(this.activitiesEl);
        this.$addActivityBtn    = this.$(this.addActivityBtnEl);
        this.$addActivityCard   = this.$(this.addActivityCardEl);
        this.$activitiesCount   = this.$(this.activitiesCountEl);
        this.$paginatorControls = this.$(this.paginatorControlsEl);

        this.renderSearchbox();

        this.renderFilterAlert();

        this.renderActivities();

        this.renderAddActivityButtons();

        return this;
    },

    /**
     * Renders the search box for this activity list
     *
     * @method renderSearchbox
     */
    renderSearchbox: function () {
        this.views.searchBox = new Ibp.Carbon.Views.SearchBox({
            el: this.searchBoxEl,
            placeholderText: 'Filter ' + this.label
        });

        this.listenTo(this.views.searchBox, 'search', this.handleSearch);
    },

    /**
     * Shows or hides the query alert box, with updated search text.
     *
     * @method renderFilterAlert
     * @param query {String} The value of the search box
     */
    renderFilterAlert: function () {
        var query = this.state.get('query'),
            hideFilter = !Boolean(query);

        // if we have a query, show the filter alert
        this.$filterAlert.toggleClass('hidden', hideFilter);

        this.$filterAlert.find('.value').html(query);
    },

    /**
     * Renders the activities in the activities list container based on the page and filtering
     * Also handles rendering the paginator controls and current display count
     *
     * @method renderActivities
     */
    renderActivities: function () {
        var activities = this.paginatedActivities.getPage(),
            viewState  = {renderActivities: false};

        // just start from scratch for filters
        if (this.wasQueryChanged()) {
            _.invoke(this.activityViews, 'unrender');

            this.activityViews = _.map(activities, function (activity) {
                var view = new Ibp.Sourcetrak.Views.Activity({
                        model: activity,
                        currentCard: activity.isNew() ? 'create' : null
                    });

                this.$activities.append(view.$el);

                return view;
            }.bind(this));
        }
        // attempt to preserve state for non-filter operations
        else {
            var activityIds = _.pluck(activities, 'id'),
                viewIds     = _.chain(this.activityViews).pluck('model').pluck('id').value(),
                addedActivityIds   = _.difference(activityIds, viewIds),
                removedActivityIds = _.difference(viewIds, activityIds);

            // unrender removed views
            if (removedActivityIds.length) {
                var removed = [];

                _.each(this.activityViews, function (view) {
                    if (_.contains(removedActivityIds, view.model.id)) {
                        view.unrender();

                        removed.push(view);
                    }
                }.bind(this));

                this.activityViews = _.difference(this.activityViews, removed);
            }

            // create and render new views
            if (addedActivityIds.length) {
                _.each(addedActivityIds, function (id) {
                    var activity = _.findWhere(activities, {id: id}),
                        view     = new Ibp.Sourcetrak.Views.Activity({
                            model: activity,
                            currentCard: activity.isNew() ? 'create' : null
                        });

                    // handle the special case of adding a new activity at the top of the list
                    if (this.state.get('showCreateCard') && activity.isNew()) {
                        this.activityViews.unshift(view);

                        this.$activities.prepend(view.$el);
                    }
                    else {
                        this.activityViews.push(view);

                        this.$activities.append(view.$el);
                    }
                }.bind(this));
            }
        }

        this.renderPaginatorControls();

        this.renderActivitiesCount();

        // silently reset the view state
        this.state.set(viewState, {silent: true});
    },

    /**
     * Hides and shows the button and card for adding a new activity
     *
     * @method renderAddActivityButtons
     */
    renderAddActivityButtons: function () {
        switch (this.state.get('showAddButtons')) {
            case 'none':
                this.$addActivityBtn.hide();
                this.$addActivityCard.hide();
                break;
            case 'btn':
                this.$addActivityBtn.show();
                this.$addActivityCard.hide();
                break;
            case 'both':
                this.$addActivityBtn.show();
                this.$addActivityCard.show();
        }
    },

    /**
     * Renders the paginator buttons for switching pages
     *
     * @method renderPaginatorControls
     */
    renderPaginatorControls: function () {
        this.resetPaginatorControls();

        // don't show the controls
        if (!this.state.get('showControls')) {
            return;
        }

        this.views.paginatorControls = new Ibp.Sourcetrak.Views.listControls({
            $container: this.$paginatorControls,
            pageState:  this.paginatedActivities.getPageState()
        });

        this.listenTo(this.views.paginatorControls, 'pageClick', this.handlePageClick);
    },

    /**
     * Renders the count of what the paginator is currently displaying
     * This is shown both above and below the paginated list of activities
     *
     * @method renderActivitiesCount
     */
    renderActivitiesCount: function () {
        // clear the old count
        this.$activitiesCount.html('');

        // don't show the count
        if (!this.state.get('showControls')) {
            return;
        }

        var info  = this.paginatedActivities.getPageState(),
            total = info.totalItems,
            range = info.pageRange,
            displayed = range.end - range.start + 1,
            countMessage = 'Displaying ';

        // if we're displaying all of the items or just one item, don't show a range
        if (displayed === total || range.end - range.start === 0) {
            countMessage += range.end + ' of ' + total;
        }
        else {
            countMessage += range.start + ' - ' + range.end + ' of ' + total;
        }

        this.$activitiesCount.html(countMessage);
    },

    /**
     * Removes and cleans up the paginator controls so they can be re-rendered
     *
     * @method resetPaginatorControls
     */
    resetPaginatorControls: function () {
        var view = this.views.paginatorControls;

        if (view) {
            view.unrender();

            this.stopListening(view);

            delete this.views.paginatorControls;
        }
    },

    /**
     * Clears the edit element and removes listeners
     *
     * @method onUnrender
     */
    onUnrender: function() {
        // clean up any new activities
        _.each(this.paginatedActivities.models, function (model) {
            if (model.isNew()) {
                model.trigger('remove', model);
            }
        });

        _.invoke(this.views, 'unrender');

        _.invoke(this.activityViews, 'unrender');
    },

    /**
     * Handles the click event on either add activity element (button or card)
     * Creates a new activity model and trigger re-render
     * If the create is canceled, the model is removed
     *
     * @method handleAddActivityClick
     */
    handleAddActivityClick: function () {
        var attrs = {
                domain_set_id: this.domainSet.id,
                one_to_one: this.isOto
            },
            activity = Ibp.Sourcetrak.Data.Activities.getNew(attrs);

        // when create activity is canceled, remove it from the paginator, triggering a re-render
        this.listenTo(activity, 'remove', function (model) {
            this.paginatedActivities.remove(model);
        }.bind(this));

        // add it to the paginator collection and trigger re-render
        this.paginatedActivities.add(activity);
    },

    /**
     * Re-renders activities whenever the collection changes
     * Will automatically move back to the first page and expand a new activity when added
     *
     * @method handleActivitiesChange
     * @param {Model} model Unused
     * @param {Collection} collection Unused
     * @param {Object} event Information about the event
     */
    handleActivitiesChange: function (model, collection, event) {
        var newActivities = this.getActivities(),
            viewState    = {
                showControls:   this.shouldShowControls(),
                showCreateCard: this.shouldShowCreateCard(),
                showAddButtons: this.getButtonsState()
            },
            resetOptions = {
                models: newActivities
            };

        // handle adding a new activity if this was an add event
        if (event.add) {
            // move back to the first page and clear the query
            resetOptions.hard = true;
        }

        // silently update view state
        this.state.set(viewState, {silent: true});

        // update the paginated collection and trigger re-render
        this.paginatedActivities.resetPageState(resetOptions);
    },

    /**
     * Handles changes to the search box by setting the query state for the view and the activities
     *
     * @method handleSearch
     * @param query {String} The value of the search box
     */
    handleSearch: function (query) {
        this.paginatedActivities.setQuery(query);
    },

    /**
     * Switches the page when a user clicks a paginator button
     *
     * @method paginatedActivities
     * @param page {String} The page to change to
     */
    handlePageClick: function (page) {
        this.paginatedActivities.setCurrentPage(page);
    },

    /**
     * Handles an updated query state
     *
     * @method handlePageStateChange
     * @param {Model} model The paginator state model
     */
    handlePageStateChange: function (model) {
        var state = {
            renderActivities: true,
            query: model.get('query'),
            showCreateCard: this.shouldShowCreateCard()
        };

        this.state.set(state);
    },

    /**
     * Handles an updated query state
     *
     * @method handleViewStateChange
     */
    handleViewStateChange: function () {
        this.renderAddActivityButtons();

        // clear the search box if the paginator has been reset
        if (!this.state.get('query')) {
            this.views.searchBox.reset();
        }

        this.renderFilterAlert();

        this.renderActivities();
    },

    /**
     * Clears the activity list filter
     *
     * @method handleDismissFilterAlert
     */
    handleDismissFilterAlert: function () {
        this.paginatedActivities.setQuery('');

        this.views.searchBox.reset();
    },

    /**
     * Show the add activity row once a domain set has been created
     *
     * @method handleCreateDomainSet
     */
    handleCreateDomainSet: function () {
        this.state.set('showAddButtons', this.getButtonsState());
    },

    /**
     * Determines if we should render the create card
     * The create card should be shown if the paginator contains an unsaved activity
     *
     * @method shouldShowCreateCard
     */
    shouldShowCreateCard: function () {
        return _.any(this.paginatedActivities.models, function (model) {
            return model.isNew();
        });
    },

    /**
     * Determines if we should render the paginator controls
     * If the total activities only require a single page, the controls should be hidden
     * since they wouldn't actually do anything.
     * If the list has been filtered down to a single page, we'll still show the controls,
     * since they'll still be useful once the query has been cleared.
     *
     * @method shouldShowControls
     */
    shouldShowControls: function () {
        return this.getActivities().length > this.paginatedActivities.ClientPagination.state.get('pageSize');
    },

    /**
     * Determines which add activity buttons should be shown
     * No buttons should show until the domain set has been created on the server.
     * The add activity card should only show for domain sets with no activities and if the create card is not showing
     * The add activity button should always be shown for created domain sets.
     *
     * @method getButtonsState
     */
    getButtonsState: function () {
        var state;

        if (this.domainSet.isNew()) {
            state = 'none';
        }
        // check shouldShowCreateCard instead of state so we can use this function to initialize state
        else if (!this.getActivities().length && !this.shouldShowCreateCard()) {
            state = 'both';
        }
        else {
            state = 'btn';
        }

        return state;
    },

    /**
     * Determines if we're currently filtering or have just cleared a filter
     *
     * @method wasQueryChanged
     */
    wasQueryChanged: function () {
        var oldQuery = this.state.previous('query') || '',
            newQuery = this.state.get('query');

        return newQuery !== oldQuery;
    },

    /**
     * Returns the activities for this domain set filtered by one to one
     *
     * @method getActivities
     * @return {Array} The list of activities
     */
    getActivities: function () {
        return this.domainSet.getActivities({one_to_one: this.isOto});
    }
});

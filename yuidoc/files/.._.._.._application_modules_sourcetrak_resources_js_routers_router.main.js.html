Ibp.Sourcetrak.Routers.Main = Ibp.Backbone.Router.extend({
    name: 'main',

    /**
     * Model to hold router state
     *
     * @property state
     * @type Model
     */
    state: null,

    /**
     * Routes defines the url fragments that will be matched as routes for the app
     *
     * @property routes
     * @type Object
     */
    routes: {
        '': 'dashboard',
        'create': 'create',
        'snippet': 'snippet',
        ':id': 'edit',
        '(:base/)pools(/)': 'pools',
        '(:base/)pools/create/:type(/)': 'wizard',
        '(:base/)pools/:pool(/)': 'poolsEdit',
        '(:base/)pools/:pool/edit(/)': 'wizard'
    },

    /**
     * Modal Routes defines the modal routes and a string that will be present in Modal Routes
     *
     * @property modalRoutes
     * @type Object
     */
    modalRoutes: {
        'pools': 'pools',
        'snippet': 'snippet'
    },

    /**
     * Eventhub subscriptions handlers
     *
     * @property subscriptions
     */
    subscriptions: {
        'navigate': 'handleNavigation',
        'modal:change': 'handleModalChange',
        'newDomainSet': 'handleNewDomainSet'
    },

    /**
     * Init - set route listeners
     *
     * @method init
     */
    init: function () {
        // set initial state
        this.state = new Ibp.Carbon.Models.State({
            isModalRendered: false
        });

        this.on('route', this.handleRouteChange);
    },

    /**
     * Handle routing to the Dashboard
     *
     * @method dashboard
     */
    dashboard: function () {
        Ibp.Sourcetrak.App.renderDashboard();
    },

    /**
     * Handle routing to the create Domain Set page
     *
     * @method create
     */
    create: function () {
        Ibp.Sourcetrak.App.renderEditDomainSet(null);
    },

    /**
     * Handle routing to the edit Domain Set page
     *
     * @method edit
     */
    edit: function (domainSetId) {
        if (domainSetId === 'pools' || domainSetId === 'pools/') {
            this.pools(domainSetId, null);
        }
        else {
            Ibp.Sourcetrak.App.renderEditDomainSet(domainSetId);
        }
    },

    /**
     * Handle routing to the Pools Modal
     *
     * @method pools
     * @param base {String} The base route
     */
    pools: function (base) {
        var data = {
            viewType: 'dashboard'
        };

        this.renderBaseRoute(base);

        this.renderPoolRoute(data);
    },

    /**
     * Handle routing to the Pools Edit Page
     *
     * @method poolsEdit
     * @param base {String} The base route
     * @param pool {String} The pool id
     */
    poolsEdit: function (base, pool) {
        var data = {
            viewType: 'edit',
            model: Ibp.Sourcetrak.Data.Pools.get(pool)
        };

        this.renderBaseRoute(base);

        this.renderPoolRoute(data);
    },

    /**
     * Handle routing to the Pools Modal
     *
     * @method wizard
     * @param base {String} The base route
     * @param type {String|Number} The wizard type (pool|oneToOne|main) for create, or pool id for edit
     */
    wizard: function (base, type) {
        var data = {
            viewType: 'wizard',
            wizardType: type
        };

        // get the model if we're using the wizard to edit a pool
        if (parseInt(type, 10)) {
            data.model = Ibp.Sourcetrak.Data.Pools.get(type);
            data.wizardType = 'edit';
        }

        this.renderBaseRoute(base);

        this.renderPoolRoute(data);
    },

    /**
     * Handle routing to the Snippet Modal
     *
     * @method snippet
     */
    snippet: function () {
        this.dashboard();

        this.publish('modal:render', {name: this.modalRoutes.snippet});
    },

    /**
     * Handle routing changes and determine if a modalUnrender event needs to be emitted
     *
     * @method handleRouteChange
     */
    handleRouteChange: function () {
        var cf = this.getCurrentFragments();

        // handle navigating to a non modal page from a modal page
        if (!this.isModalRoute(cf.current) && this.state.get('isModalRendered')) {
            this.publish('modal:unrender');
        }
    },

    /**
     * Handle modal changes and determine if a navigation event needs to be triggered
     *
     * @method handleModalChange
     */
    handleModalChange: function (isRendered) {
        var cf = this.getCurrentFragments();

        // track the render state of the modals
        this.state.set({'isModalRendered': isRendered});

        // handle navigating to a non modal page from a modal page
        if (this.isModalRoute(cf.current) && !isRendered) {
            this.navigate(cf.base, {trigger: true});
        }
    },

    /**
     * Determine if a given route/fragment is a Modal Route
     *
     * @method isModalRoute
     * @return Boolean
     */
    isModalRoute: function(route) {
        var isModal = _.find(this.modalRoutes, function (modalRoute) {
            return (route.indexOf(modalRoute) > -1) ? true : false;
        });

        return (isModal === undefined) ? false : true;
    },

    /**
     * Handle navigate events published on the event hub
     *
     * @method handleNavigation
     */
    handleNavigation: function (fragment) {
        var options = {
                trigger: true
            },
            cf = this.getCurrentFragments(),
            route;

        fragment = typeof fragment !== 'string' ? fragment.toString() : fragment;

        if (this.contains(fragment, 'pools')) {
            if (this.contains(cf.current, 'pools')) {
                route = this.addTrailingSlash(cf.base) + fragment + '/';
            }
            else {
                route = this.addTrailingSlash(cf.current) + fragment + '/';
            }

            this.navigate(route, options);
        }
        else {
            this.navigate(fragment, options);
        }
    },

    handleNewDomainSet: function (model) {
        this.transferToEdit(model);
    },

    /**
     * Contains returns true if a string contains a given substring
     *
     * @method contains
     */
    contains: function(str, substr) {
        return str.indexOf(substr) > -1;
    },

    /**
     * Adds a trailing slash to a given string
     *
     * @method addTrailingSlash
     */
    addTrailingSlash: function(str) {
        return (str.charAt(str.length-1) !== '/') ? str + '/' : str;
    },

    /**
     * Get Current URL Fragment data
     *
     * @method getCurrentFragments
     */
    getCurrentFragments: function () {
        var currentFragement    = Backbone.history.getFragment(),
            currentFragments    = currentFragement.split('/'),
            currentBaseFragment = this.isModalRoute(currentFragments[0]) ? '/' : currentFragments[0];

        return {
            base: currentBaseFragment,
            current: currentFragement,
            fragments: currentFragments
        };
    },

    /**
     * Render the base view for sourcetrak
     *
     * @method renderBaseRoute
     * @param base {String} The base route (create|pools|domain set id)
     */
    renderBaseRoute: function (base) {
        // match that baseToken
        if (base === null || base === 'pools') {
            this.dashboard();
        }
        else if (!isNaN(base)) {
            this.edit(base);
        }
        else if (base === 'create') {
            this.create();
        }
    },

    /**
     * Render the pool modal view
     *
     * @method renderPoolRoute
     * @param data {Object} The data to pass into the view
     */
    renderPoolRoute: function (data) {
        this.publish('modal:render', {
            name: this.modalRoutes.pools,
            data: data
        });
    },

    transferToEdit: function (model) {
        this.navigate('#' + model.id);
    }
});

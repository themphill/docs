/**
 * Mixin for Collections that adds client-side pagination with filtering
 *
 * @module Carbon
 * @class Mixins.ClientPagination
 */
Ibp.Carbon.Mixins.ClientPagination = {
    ClientPagination: null,

    /**
     * Initializes the paginator state
     *
     * @method init
     */
    init: function () {
        var state = {
                query: '',
                firstPage:    1,
                lastPage:     1,
                currentPage:  1,
                pageSize:     10,
                totalItems:   0,
                displayRange: 3,
                pageRange:    {
                    start: 0,
                    end:   0
                }
            };

        this.ClientPagination = this.ClientPagination || {};

        // set the dafault paginator state, respecting anything set in the base class
        state = _.deepExtend(state, this.ClientPagination.defaults || {});

        // create paginator state model
        this.ClientPagination.state = new Ibp.Carbon.Models.State(state);

        // set the function used to filter the models
        this.search = this.search || this.where;

        // re-initialize page state when the models change
        this.listenTo(this, 'add remove reset', this.handleCollectionChange);
    },

    /**
     * Reset page state when the collection changes
     *
     * @private
     * @method handleCollectionChange
     */
    handleCollectionChange: function () {
        this.setPageState();
    },

    /**
     * Applies the semantic api and keeps the page number in bounds
     *
     * @private
     * @method formatPageNumber
     */
    formatPageNumber: function (pageNumber) {
        var num;

        switch (pageNumber) {
            case 'first':
                num = this.ClientPagination.state.get('firstPage');

                break;
            case 'last':
                num = this.ClientPagination.state.get('lastPage');

                break;
            case 'next':
                num = this.ClientPagination.state.get('currentPage') + 1;

                break;
            case 'previous':
                num = this.ClientPagination.state.get('currentPage') - 1;

                break;
            default:
                // default to the first page if we don't get an actual number
                num = parseInt(pageNumber, 10) || this.ClientPagination.state.get('firstPage');
        }

        // make sure we stay in bounds
        if (num < this.ClientPagination.state.get('firstPage')) {
            num = this.ClientPagination.state.get('firstPage');
        }

        if (num > this.ClientPagination.state.get('lastPage')) {
            num = this.ClientPagination.state.get('lastPage');
        }

        return num;
    },

    /**
     * Generates new page state from a page number and/or query
     * This is the only place where paginator state is actually mutated
     *
     * @private
     * @method setPageState
     * @param attrs {Object} The state attributes and values to set
     */
    setPageState: function (attrs) {
        attrs || (attrs = {});

        var query       = _.has(attrs, 'query')
                        ? attrs.query
                        : this.ClientPagination.state.get('query'),
            currentPage = _.has(attrs, 'currentPage')
                        ? attrs.currentPage
                        : this.ClientPagination.state.get('currentPage'),
            pageSize = this.ClientPagination.state.get('pageSize'),
            startIndex,
            endIndex,
            models,
            page;

        // pageSize * currentPage is the index of first model of the page AFTER currentPage
        // subtracting 1 sets us at the index of the first model of currentPage
        startIndex = (pageSize * (currentPage - 1));

        endIndex = startIndex + pageSize;

        // get the list of models to paginate
        models = this.search(query);

        // get the page requested
        page = models.slice(startIndex, endIndex);

        // cache last page, which is at least 1
        attrs.lastPage = Math.ceil(models.length/pageSize) || 1;

        // cache the total and the range of models in this page to display to the user, e.g. 4 - 8 of 12
        attrs.totalItems = models.length;

        attrs.pageRange = {
            start: startIndex + 1,
            end: startIndex + page.length
        };

        this.ClientPagination.state.set(attrs);
    },

    /**
     * Wrapper for Collection.reset()
     * Optionally allows for a hard reset, which clears the query and moves back to the first page
     *
     * @method resetPageState
     * @param models {Array} The new list of models to be paginated
     * @param options {Object}
     * @param options.models {Array} Models to reset the collection with
     * @param options.hard {Boolean} Flag to perform a hard reset
     */
    resetPageState: function (options) {
        var state = this.ClientPagination.state;

        options = options || {};

        if (options.hard === true) {
            this.ClientPagination.state.set({
                currentPage: this.ClientPagination.state.get('firstPage'),
                query: ''
            }, {silent: true});
        }

        // resetting the collection will trigger a state change
        if (options.models) {
            this.reset(options.models);
        }
        // otherwise, we can do it manually
        else {
            state.trigger('change', state);
        }
    },

    /**
     * Sets the current page via page number or semantic term
     *
     * @method setCurrentPage
     * @param pageNUmber {String|Number} The page number or term to set as the current page
     */
    setCurrentPage: function (pageNumber) {
        var page = this.formatPageNumber(pageNumber);

        this.setPageState({currentPage: page});
    },

    /**
     * Sets the current query used to filter the collection
     * Filtering always takes you back to first page
     *
     * @method setQuery
     * @param query {Mixed} The term used to filter the collection. Defer to the filter method for the data type.
     */
    setQuery: function (query) {
        var attrs = {
            query: query,
            currentPage: this.formatPageNumber('first')
        };

        this.setPageState(attrs);
    },

    /**
     * Accessor method for current page state
     * Can be used to render the paginator controls
     *
     * @method getPageState
     */
    getPageState: function () {
        return this.ClientPagination.state.toJSON();
    },

    /**
     * Returns the page generated from the paginator state
     *
     * @method getPage
     */
    getPage: function () {
        var query       = this.ClientPagination.state.get('query'),
            currentPage = this.ClientPagination.state.get('currentPage'),
            pageSize    = this.ClientPagination.state.get('pageSize'),
            startIndex  = (pageSize * (currentPage - 1)),
            endIndex    = startIndex + pageSize;

        return this.search(query).slice(startIndex, endIndex);
    }
};

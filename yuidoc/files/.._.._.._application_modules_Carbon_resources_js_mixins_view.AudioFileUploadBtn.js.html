/**
 * Mixin for Views that handles all of the audio file buttons on the page
 *
 * @module Carbon
 * @class Mixins.AudioFileUploadBtn
 */
Ibp.Carbon.Mixins.AudioFileUploadBtn = {
    /**
     * AudioFileUploadBtn Namespace
     *
     * @property audioFileUploadBtnEl Audio File Upload Button class selector
     * @property $audioFileUploadBtns Audio File Upload Buttons jQuery Object
     * @property audioBtnCollection Collection to hold audio file upload models
     * @type Object
     */
    AudioFileUploadBtn: {
        audioFileUploadBtnEl: '.audio-file-upload-btn',
        $audioFileUploadBtns: null,
        audioBtnCollection: null
    },

    /**
     * Render
     *
     * @method render
     * @return this view
     */
    render: function () {
        this.delegateAudioBtns();
    },

    /**
     * Delegate Audio Buttons
     *
     * @method delegateAudioBtns
     */
    delegateAudioBtns: function() {
        // Find all the upload buttons
        this.AudioFileUploadBtn.$audioFileUploadBtns = $(this.AudioFileUploadBtn.audioFileUploadBtnEl);

        // Create a collection to store data about each upload object
        this.AudioFileUploadBtn.audioBtnCollection   = new Ibp.Backbone.Collection();

        // Gather relevant information about each audio component we find
        _.each(this.AudioFileUploadBtn.$audioFileUploadBtns, function (btn) {
            var $btn = $(btn),
                id   = $btn.data('audio-btn-id'),
                name = $btn.data('audio-btn-name');

            this.AudioFileUploadBtn.audioBtnCollection.add({
                id: id,
                name: name,
                file: null,
                iframeId: 'audio-iframe-' + id,
                formId:   'audio-form-'   + id,
                inputId:  'audio-file-browser-input-' + id
            });
        }.bind(this));

        // After discovering our audio components, bind change event listeners to
        // each components filebrowser input
        this.bindInputs();
    },

    /**
     * Bind Inputs
     *
     * @method bindInputs
     */
    bindInputs: function () {
        var e = {
            'change .audio-file-browser-input': 'handleAudioFileChange'
        };

        if (this.events) {
            _.defaults(this.events, e);
        }
        else {
            this.events = e;
        }
    },

    /**
     * Handle Audio File Change
     *
     * @method handleAudioFileChange
     */
    handleAudioFileChange: function (e) {
        // Match the event with a model representing the audio object we want to work with
        var audioBtn = this.AudioFileUploadBtn.audioBtnCollection.findWhere({'inputId': $(e.target).attr('id')});

        // If no model found, abort. It is mostly likely being handled by a different view that
        // is currently in play and contains the same mixin
        if (audioBtn === undefined) {
            return;
        }

        // Once we are sure it is us, setup the initial references to DOM elements and get the
        // file name of the file attempting to be uploaded
        var $iFrame  = $('#' + audioBtn.get('iframeId')),
            $form    = $('#' + audioBtn.get('formId')),
            file     = $('#' + audioBtn.get('inputId')).val(),
            response;

        // This preventDefualt line is used in IVR and I dont want to remove cause I know it serves
        // some edge case purpose.
        e.preventDefault();

        // If there is a actual value (they didnt just hit cancel which triggers a change in some browsers)
        // then we set about initiating an upload.
        if (file.length > 0) {
            // Save the base filename (no extension) to the audio model
            // This can be used by the parent view to match the selection of audio files in a list
            // to the file that was uploaded
            audioBtn.set('file', file.split('.')[0]);

            // Bind listener to the form contained in our audio upload component. The following submit
            // will target this iframe and we will listen to the namespaced load event. We make the
            // assumption that on load, our audio has been imported and a parent view can query the
            // servery for a new list of audio files. Our uploaded file should be in that list.
            $iFrame.on('load.audio', function(){
                // build a response using the current audio button and the server's response
                response = {
                    audioBtn: audioBtn,
                    response: Ibp.Utils.parseIframeBody(audioBtn.get('iframeId'))
                };

                $iFrame.off('load.audio');
                this.publish('audio:imported', response);
            }.bind(this));

            // Submit audio upload form (which contains the file browser input). This will target the
            // iFrame located within the audio component using the random id assigned by the
            // handlebars helper. It is here in this iFrame that the file upload takes place, and we
            // will listen for a load event.
            $form.submit();
        }
    }
};

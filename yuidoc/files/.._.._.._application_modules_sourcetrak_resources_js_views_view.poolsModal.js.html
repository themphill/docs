Ibp.Sourcetrak.Views.PoolsModal = Ibp.Carbon.Views.Modal2.extend({
    name: 'poolsModal',
    template: Ibp.Sourcetrak.Templates.modal_pools,

    /**
     * The views object retains references to children views
     *
     * @property views
     * @type Object
     */
    views: null,

    /**
     * Flag for allowing the modal to be closed using escape key
     *
     * @property keyboard
     * @type Boolean
     */
    keyboard: false,

    /**
     * Mapping of Pool Modal states to their views
     *
     * @property poolViews
     * @type Object
     */
    poolViews: {
        'dashboard': Ibp.Sourcetrak.Views.PoolsDashboard,
        'edit': Ibp.Sourcetrak.Views.EditPool,
        'wizard': Ibp.Sourcetrak.Views.Wizard
    },

    /**
     * Selector for the back button
     *
     * @property backButtonEl
     * @type String
     */
    backButtonEl: '.js-modal-back',

    /**
     * jQuery reference for the back button
     *
     * @property $backButton
     * @type jQuery Object
     */
    $backButton: null,

    /**
     * DOM Event handlers
     *
     * @property modalEvents
     */
    modalEvents:  {
        'click .js-modal-back': 'handleBack'
    },

    /**
     * Inits state and listens for refresh events
     *
     * @method onInit
     */
    onInit : function () {
        this.state.set({
            currentViewType: '',
            showBackButton: false,
            hideModalHeader: false
        });

        this.views = {};

        this.listenTo(this, 'refresh', this.handleRefresh);
        this.listenTo(this.state, 'change:hideModalHeader', this.handleModalHeaderChange);
    },

    /**
     * Renders the pool view
     *
     * @method onRender
     */
    onRender: function (options) {
        this.$backButton = this.$(this.backButtonEl);

        this.renderPoolView(options);
    },

    /**
     * Unrenders child views
     *
     * @method onUnrender
     */
    onUnrender: function () {
        _.invoke(this.views, 'remove');
    },

    /**
     * Decides which view to render based on the route data
     *
     * @method renderPoolView
     * @params data {Object} The data to render the view with
     */
    renderPoolView: function (data) {
        var newType = data.viewType,
            oldType = this.state.get('currentViewType');

        // initialize the new pool view (render happens later)
        switch (newType) {
            case 'dashboard':
                this.renderPoolsDashboard(data);
                break;
            case 'edit':
                this.renderEditPool(data);
                break;
            case 'wizard':
                this.renderWizard(data);
                break;
        }

        // hide old view
        if (this.views[oldType] && oldType !== newType) {
            // prevent animation in some cases
            if (oldType === 'wizard' || (oldType === 'dashboard' && newType === 'wizard')) {
                this.views[oldType].unrender();
            }
            else {
                this.views[oldType].hide();
            }
        }

        // render without animation if the modal is just opening
        if (!oldType || newType === 'wizard') {
            this.views[newType].render();
        }
        // animate render if this is a transition
        else {
            this.views[newType].show();
        }

        // update backdrop settings
        this.updateBackdrop();

        // show or hide back button
        this.$backButton.toggle(this.state.get('showBackButton'));

        this.state.set({
            currentViewType: newType,
            hideModalHeader: (newType === 'wizard')
        });
    },

    /**
     * Handle hideModalHeader state change
     *
     * @method handleModalHeaderChange
     */
    handleModalHeaderChange: function () {
        this.toggleHeader(!this.state.get('hideModalHeader'));
    },

    /**
     * Toggle the Pools Modal Header
     *
     * @method toggleHeader
     * @param show {Boolean} Optional. If supplied indicates show or hide.
     */
    toggleHeader: function (show) {
        var $header = this.$('.js-pools-modal-header');

        if (show !== undefined) {
            $header.toggle(show);
        }
        else {
            $header.toggle();
        }
    },

    /**
     * Render Pools Dashboard
     * then shows the modal.
     *
     * @method renderPoolsDashboard
     */
    renderPoolsDashboard: function () {
        if (!this.views.dashboard) {
            this.views.dashboard = new Ibp.Sourcetrak.Views.PoolsDashboard();
        }

        this.state.set({
            showBackButton: false,
            backdropActive: true
        });
    },

    /**
     * Render Pools Edit page
     *
     * @method renderEditPool
     */
    renderEditPool: function (data) {
        if (!this.views.edit) {
            this.views.edit = new Ibp.Sourcetrak.Views.EditPool({
                model: data.model
            });
        }
        else if (this.views.edit.model.id !== data.model.id) {
            this.views.edit.remove();

            this.views.edit = new Ibp.Sourcetrak.Views.EditPool({
                model: data.model
            });
        }

        this.state.set({
            showBackButton: true,
            backdropActive: true
        });
    },

    /**
     * Render Pool creation wizard
     *
     * @method renderWizard
     */
    renderWizard: function (data) {
        var isOto = data.wizardType.toLowerCase() === 'onetoone' ? true : false,
            attrs = {
                one_to_one: isOto
            };

        this.views.wizard = new Ibp.Sourcetrak.Views.Wizard({
            type: data.wizardType,
            model: data.model || Ibp.Sourcetrak.Data.Pools.getNew(attrs)
        });

        this.state.set({
            showBackButton: false,
            backdropActive: false
        });
    },

    /**
     * Render sub view
     *
     * @method handleRefresh
     * @param data {Object} View data
     */
    handleRefresh: function (data) {
        this.renderPoolView(data);
    },

    /**
     * Closes the Pools modal
     *
     * @method handleClose
     */
    handleClose: function () {
        this.cancel();
    },

    /**
     * Navigates back to the dashboard view
     *
     * @method handleBack
     */
    handleBack: function () {
        this.publish('navigate', 'pools');
    }
});

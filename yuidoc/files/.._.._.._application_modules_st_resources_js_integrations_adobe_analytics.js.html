/**
 * This is the integration object to be used as an example for creating future integrations.
 * New integrations may copy this, but only need to change the name,
 * the fingerprint function,
 * and the data object sent in the fetch function.
 *
 * Properties and methods marked Final should NOT be modified.
 *
 * The YUIDoc syntax reference can be found at http://yui.github.io/yuidoc/syntax/index.html
 *
 * @module Integration
 */


/**
 * The Adobe Analytics integration
 * @class aa
 */
(function (window) {
    /**
     * The integration "class" object
     * Final
     */
    var integration = {};

    /**
     * The name of this integration
     * This name will be matched against a list of integrations from the server,
     * so it must be the same in both places
     *
     * @private
     * @property name
     * @type String
     */
    var name = 'aa';

    /**
     * Whether or not this integration's data has been found and registered to be sent to the server
     * Final
     *
     * @private
     * @property found
     * @type Boolean
     * @default false
     */
    var found = false;

    /**
     * Getter for this integration's name
     * Final
     *
     * @method name
     * @return name
     */
    integration.name = function () {
        return name;
    };

    /**
     * Getter for this integration's found status
     * Final
     *
     * @method found
     * @return found
     */
    integration.found = function () {
        return found;
    };

    /**
     * Checks to see if the integration data is present, then finds and dispatches that data
     * The integration's data is stored both in cookies and also in the client code
     *
     * The nature of the data is unique to each integration, but should always be wrapped in an object.
     * The backend integration code is responsible for parsing that data.
     * The wrapping if() block should not be modified.
     *
     * @method name
     * @param {Function} done Must be passed to dispatchData()
     * @return undefined
     */
    integration.fetch = function (done) {
        if (fingerprint()) {
            // take the info we need from the adobe tracking cookie
            var data = {
                visitorId: findPriorityId()
            };

            // let st know we've found the data and pass it along
            dispatchData(data, done);
        }
    };

    /**
     * Determines if the data needed for this integration is present on the page
     *
     * @private
     * @method fingerprint
     * @return {Boolean}
     */
    function fingerprint () {
        // check to see if the adobe analytics client code has loaded and produced a valid visitor ID
        return findPriorityId() !== null;
    }

    /**
     * Checks for the existence of a number of IDs, taking the most relevant one if it exists
     *
     * @private
     * @method findPriorityId
     * @return String|Null
     */
    function findPriorityId () {
        var result = null,
            // this object is provided by adobe and contains methods for extracting IDs
            visitorService = window.visitor,
            // this object is sometimes used by adobe clients to specify custom IDs
            visitorObject = window.s;

        if (visitorService
            && visitorService.getMarketingCloudVisitorID
            && visitorService.getMarketingCloudVisitorID().length > 0) {
            // the marketing cloud service id is the highest priority option
            result = visitorService.getMarketingCloudVisitorID();
        } else if (visitorObject && visitorObject.visitorID) {
            // a client-made custom visitor ID is the middle priority options
            result = visitorObject.visitorID;
        } else if (visitorObject
            && visitorObject.c_r
            && visitorObject.c_r('s_vi').length > 0) {
            // the classic visitor ID stored in the session cookie is the lowest priority option
            result = visitorObject.c_r('s_vi');
        }

        return result;
    }

    /**
     * Calls the done function to register this integration's data to be sent to the server
     * and marks this integration as found
     * Final
     *
     * @private
     * @method dispatchData
     * @param {Object} data   The integration's data
     * @param {Function} done The callback function to set the integration's data
     * @return undefined
     */
    function dispatchData (data, done) {
        // prepare the data for this integration
        var response = {
            n: name,
            d: data
        };

        // mark this integration as found
        found = true;

        // set this data on the integrationData object
        done(response);
    }

    // add this integration to SourceTrak
    window._st.integrations.push(integration);
})(window);

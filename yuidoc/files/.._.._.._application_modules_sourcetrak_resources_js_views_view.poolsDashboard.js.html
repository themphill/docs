/**
 * Pools managment dashboard
 * Loads list of pools/one-to-ones, a search box to filter/search that list, and the pool management toolbar
 *
 * @module Sourcetrak
 * @class Views.PoolsDashboard
 */
Ibp.Sourcetrak.Views.PoolsDashboard = Ibp.Carbon.Views.ModalPageView.extend({
    name: 'PoolsDashboard',
    el: '#pools-dashboard',
    template: Ibp.Sourcetrak.Templates.pools_dashboard,

    /**
    * Model to hold view state
    *
    * @property state
    * @type Model
    */
    state: null,

    /**
    * The views object retains references to children views
    *
    * @property views
    * @property views.poolsDashboardToolbar {Object} Pools Dashboard Toolbar
    * @property views.poolsList {Object} Pools List
    * @type Object
    */
    views: {},

    /**
    * Container element for the searchbox/filter in the modal's header
    *
    * @property searchboxEl
    * @type String
    */
    searchboxEl: '#pools-search-box',

    /**
    * Init notifications and render
    *
    * @method init
    */
    init: function () {

    },

    /**
    * Assigns $el and renders pools dashboard toolbar, the searchbox, and the pools list
    *
    * @method render
    * @return this view
    */
    render: function () {
        this.$el.removeClass('hidden');
        this.$el.html(this.template());

        this.renderPoolsDashboardToolbar();
        this.renderPoolsSearchbox();
        this.renderPoolsList();

        this.$('i[title]').tooltip({placement: 'bottom'});

        return this;
    },

    /**
    * Renders the Manage Pools Modal Toolbar
    *
    * @method renderPoolsDashboardToolbar
    */
    renderPoolsDashboardToolbar: function () {
        this.views.poolsDashboardToolbar = new Ibp.Sourcetrak.Views.PoolsDashboardToolbar();
    },

    /**
    * Renders the Manage Pools Modal Filter/Searchbox with assigned options
    *
    * @method renderPoolsSearchbox
    */
    renderPoolsSearchbox: function () {
        var options = {
            el: this.searchboxEl,
            placeholderText: 'Search By Description, Area Code, or Phone Number',
            eventName: 'filterPoolsList'
        };

        this.views.poolsSearchBox = new Ibp.Carbon.Views.SearchBox(options);
    },

    /**
    * Instantiates poolsList view to render the list of pools/one-to-ones
    *
    * @method renderPoolsList
    */
    renderPoolsList: function () {
        this.views.poolsList = new Ibp.Sourcetrak.Views.PoolsList({
            el: this.$('#pools-list'),
            collection: Ibp.Sourcetrak.Data.Pools
        });
    }
});

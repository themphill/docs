/**
 * Create New Activity Card View
 *
 * @module SourceTrak
 * @class Views.ActivityCreate
 * @extends Ibp.Backbone.View
 */
Ibp.Sourcetrak.Views.ActivityCreate = Ibp.Backbone.View.extend({
    mixins: ['Autoset', 'HighlightInvalid'],
    name: 'ActivityCreate',
    template: Ibp.Sourcetrak.Templates.activity_edit,

    /**
     * Backbone Model to hold view state
     *
     * @property state
     * @type Model
     */
    state: null,

    /**
     * @property cardType
     * @type String
     */
    cardType: 'create',

    /**
     * Advanced options button selector
     *
     * @property advancedButtonEl
     * @type String
     */
    advancedButtonEl: '.js-activity-advanced',

    /**
     * Reference to the advanced button
     *
     * @property $advancedButton
     * @type jQuery Object
     */
    $advancedButton: null,

    /**
     * Selector for the keyword input
     *
     * @property keywordEl
     * @type String
     */
    keywordEl: '.js-keyword',

    /**
     * Reference to the keyword input
     *
     * @property $keyword
     * @type jQuery Object
     */
    $keyword: null,

    /**
     * Selector for the keyword input label
     *
     * @property createButtonEl
     * @type String
     */
    keywordLabelEl: '.js-keyword-label',

    /**
     * Reference to the keyword input label
     *
     * @property $createButton
     * @type jQuery Object
     */
    $keywordLabel: null,

    /**
     * Disabled class for the campaign label
     *
     * @property labelDisabledClass
     * @type String
     */
    labelDisabledClass: 'disabled',

    /**
     * Class name to add to create button to disable it
     *
     * @property btnDisabledClass
     * @type String
     */
    btnDisabledClass: 'btn-disabled',

    /**
     * Listens for:
     * any change in input fields,
     * clicks on the cancel and create buttons
     *
     * @property events
     * @type String
     */
    events: {
        'click .js-activity-cancel': 'handleCancel',
        'click .js-activity-advanced': 'handleAdvanced',
        'click .js-activity-create': 'handleCreate'
    },

    /**
     * Calls render
     *
     * @method init
     */
    init: function (options) {
        this.$container = options.$container;

        this.state = new Ibp.Carbon.Models.State({
            optionsEnabled: !this.isReferral(),
            edit: !this.model.isNew()
        });

        // update the pools list when the pools collection changes
        this.listenTo(Ibp.Sourcetrak.Data.Pools, 'add change:description', this.render);

        // update the model's pool when an activity changes pools
        this.listenTo(Ibp.Sourcetrak.Data.Activities, 'change:pool_id', this.handleOtoPoolChange);

        // update the model's pool when the pools collection changes
        this.listenTo(Ibp.Sourcetrak.Data.Pools, 'remove', this.handlePoolRemoved);

        // update the keyword field whenever the activity type changes
        this.listenTo(this.model, 'change:activity_type', this.handleTypeChange);

        // listen for this model's creation event
        this.listenToOnce(this.model, 'sync', this.handleSave);

        this.render();
    },

    /**
     * Renders view and gets reference to the create button
     *
     * @method render
     * @return this view
     */
    render: function () {
        var activityType = this.model.getActivityType() || {},
            data = {
                // omit the domain set activity type
                activityTypes:    this.model.activityTypes.slice(1),
                poolDescriptions: this.getPoolDescriptions(),
                keywordLabel:     activityType.label,
                keywordAttr:      activityType.attribute,
                keywordValue:     this.model.get('keyword') || this.model.get('url_path'),
                optionsActive:    this.state.get('optionsEnabled'),
                edit:             this.state.get('edit'),
                otoFlag:          this.model.get('one_to_one')
            };

        _.extend(data, this.model.toJSON());

        // create activity body
        this.$el = $(this.template(data));

        // render
        this.$container.html(this.$el);

        // rebind events after rerendering keyword label changes
        this.delegateEvents();

        this.$advancedButton = this.$(this.advancedButtonEl);
        this.$keyword = this.$(this.keywordEl);
        this.$keywordLabel = this.$(this.keywordLabelEl);

        return this;
    },

    /**
     * Handler for the cancel button
     *
     * @method handleCancel
     */
    handleCancel: function () {
        this.cancelCreate();
    },

    /**
     * Handler for the advanced options button
     *
     * @method handleAdvanced
     */
    handleAdvanced: function () {
        if (this.$advancedButton.hasClass(this.btnDisabledClass)) {
            return;
        }

        this.showAdvanced();
    },

    /**
     * Handler for the create button
     * Only triggers if the button is enabled
     *
     * @method handleCreate
     */
    handleCreate: function () {
        if (this.model.isValid()) {
            this.createActivity();
        }
    },

    /**
     * Handler for activity type changes
     * Rerenders the card to reflect changes in keyword label and options being toggled
     *
     * @method handleTypeChange
     */
    handleTypeChange: function () {
        this.checkActivityType();

        this.render();
    },

    /**
     * Handler for the model creation event
     * Flips the card to display
     *
     * @method handleSave
     */
    handleSave: function () {
        this.trigger('flip', {card: this.cardType, event: 'create'});
    },

    /**
     * Resets the activity's pool id when another activity takes its pool
     * This is to ensure one to one pools only ever have one activity
     *
     * @method handleOtoPoolChange
     * @param pool {Model} The pool that was deleted
     */
    handleOtoPoolChange: function (pool) {
        if (this.model.get('one_to_one') && this.model.get('pool_id') === pool.id) {
            this.model.set('pool_id', null);
        }

        this.render();
    },

    /**
     * Resets the activity's pool id when that pool is deleted
     *
     * @method handlePoolRemoved
     * @param pool {Model} The pool that was deleted
     */
    handlePoolRemoved: function (pool) {
        if (this.model.get('pool_id') && this.model.get('pool_id') === pool.id) {
            this.model.set('pool_id', null);
        }

        this.render();
    },

    /**
     * Cancels create and flips back to the add card
     *
     * @method cancelCreate
     */
    cancelCreate: function () {
        // dismiss validation error notification
        if (this.model.validationError) {
            this.publish('notify:dismiss');
        }

        this.model.trigger('remove', this.model);
    },

    /**
     * Show advanced options
     *
     * @method showAdvanced
     */
    showAdvanced: function () {
        this.trigger('flip', {card: this.cardType, event: 'advanced'});
    },

    /**
     * Creates new activity and flips to the display card
     *
     * @method createActivity
     */
    createActivity: function () {
        this.model.trigger('create', this.model, this.model.attributes);
    },

    /**
     * Sets the options enabled state based on the activity type
     * Referrals have the options disabled
     *
     * Because the card rerenders whenever the activity type changes, there is no logic
     * for toggling the options. It all happens in the template via classes/attributes
     *
     * @method checkActivityType
     */
    checkActivityType: function () {
        this.state.set('optionsEnabled', !this.isReferral());
    },

    /**
     * Returns a list of all the pool descriptions by pool id
     * Used to populate the pools select list
     *
     * @method getPoolDescriptions
     * @return Array
     */
    getPoolDescriptions: function () {
        var activityOto = this.model.get('one_to_one'),
            pools = [];

        _.each(Ibp.Sourcetrak.Data.Pools.models, function (pool) {
            // filter out pools that don't match the activity's one_to_one setting
            if (pool.get('one_to_one') !== activityOto) {
                return;
            }

            // filter out one_to_one pools that already have an activity
            if (activityOto && pool.getActivities().length) {
                return;
            }

            pools.push({
                id: pool.get('id'),
                description: pool.get('description')
            });
        });

        return pools;
    },

    /**
     * Is the current model or passed id a Referral type
     *
     * @method isReferral
     * @param id {String|Number} The activity id to check
     * @return Boolean
     */
    isReferral: function (id) {
        id = id ? parseInt(id, 10) : this.model.get('activity_type');

        return id === 5;
    }
});

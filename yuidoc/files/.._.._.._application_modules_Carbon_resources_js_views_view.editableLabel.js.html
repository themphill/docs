Ibp.Carbon.Views.EditableLabel = Ibp.Backbone.View.extend({
    name: 'editableLabel',
    template: Ibp.Carbon.Templates.editable_label,

    /**
     * jQuery reference for the label container
     *
     * @property $container
     * @type jQuery Object
     */
    $container: null,

    /**
     * The name of the model attribute to use as the label
     *
     * @property labelAttribute
     * @type String
     */
    labelAttribute: null,

    /**
     * The length at which to truncate the label for truncatedLabel
     *
     * @property truncatedAt
     * @type Number
     */
    truncatedAt: null,

    /**
     * Selector for the label wrapper
     *
     * @property labelTextEl
     * @type String
     */
    labelTextEl: '.js-label-wrapper',

    /**
     * jQuery reference for the label wrapper
     *
     * @property $label
     * @type jQuery Object
     */
    $labelText: null,

    /**
     * Selector for the label input
     *
     * @property labelInputEl
     * @type String
     */
    labelInputEl: '.js-label-input',

    /**
     * jQuery reference for the label input
     *
     * @property $labelInput
     * @type jQuery Object
     */
    $labelInput: null,

    /**
     * Flag for applying the phone number input mask or not
     *
     * @property maskPhoneNumber
     * @type Boolean
     */
    maskPhoneNumber: false,

    /**
     * The default length to truncate at
     *
     * @property truncatedAt
     * @type Number
     */
    DEFAULT_TRUNCATE: 45,

    /**
     * DOM Event handlers
     *
     * @property events
     */
    events:  {
        'click .js-label-name, .js-label-wrapper': 'showLabelInput',
        'blur  .js-label-input':   'handleBlur',
        'keyup .js-label-input':   'handleKeyboardInput'
    },

    /**
     * Init
     *
     * @method init
     */
    init: function (options) {
        _.bindAll(this);

        this.$container       = options.$container;
        this.labelName        = options.labelName;
        this.labelAttribute   = options.labelAttribute;
        this.labelPlaceholder = options.labelPlaceholder || 'Enter value';
        this.truncateAt       = options.truncateAt || this.DEFAULT_TRUNCATE;
        this.maskPhoneNumber  = options.maskPhoneNumber === undefined ? false : options.maskPhoneNumber;

        this.listenTo(this.model, 'sync', this.handleChange);

        this.onInit();

        this.render();

        // attach the view to the DOM
        this.renderToContainer();
    },

    onInit: function () {},

    /**
     * Render
     *
     * @method render
     * @return this view
     */
    render: function () {
        var data = {
            label:            this.getLabel(),
            labelName:        this.labelName,
            truncatedLabel:   this.escapeSpaces(this.getTruncatedLabel()),
            labelPlaceholder: this.escapeSpaces(this.labelPlaceholder),
            namespace:        this.model.name.toLowerCase(),
            attribute:        this.labelAttribute,
            phoneNumber:      this.maskPhoneNumber
        };

        this.$el.html(this.template(data));

        this.$labelText  = this.$(this.labelTextEl);
        this.$labelInput = this.$(this.labelInputEl);

        // TODO: need a solution for listening to CSS events, so we can fix the width when the element becomes visible
        // anti-jank hack part 1: round out the label width
        // jquery automatically rounds pixels from floats to ints (i.e. 100.125 > 100)
        if (this.$el.css('width') !== '0px') {
            this.$labelText.width(this.$labelText.width());
        }

        // if we're supposed to, apply the phone number mask to this input
        if (this.maskPhoneNumber) {
            Ibp.Utils.PhoneNumber({
                'el': this.$labelInput,
                'addListener': true,
                'regionCode': 'US',
                'maxLength': 14
            });
        }

        return this;
    },

    /**
     * Inserts this view into the DOM
     *
     * @method renderToContainer
     */
    renderToContainer: function () {
        this.$container.html(this.$el);
    },

    /**
     * Handles blur event
     * Rerenders if blur is triggered without any change in the input
     * Patched to fire a change event in IE for masked phone numbers
     *
     * @method handleBlur
     */
    handleBlur: function () {
        if (Ibp.Utils.isIE() && this.maskPhoneNumber) {
            this.triggerChange();
        }

        if (!this.model.isSaving && !this.model.validationError) {
            this.render();
        }
    },

    /**
     * Handles key press events in the label input
     * Rerenders the view if 'escape' is pressed while editing
     *
     * @method handleKeyboardInput
     */
    handleKeyboardInput: function (e) {
        var keyCode = e.keyCode || e.which;

        // for escape, rerender to hide and reset the input
        if (keyCode === 27) {
            e.stopPropagation();

            // reset label value so we don't accidentally trigger a change event
            this.$labelInput.val(this.getLabel());

            this.render();
        }
    },

    /**
     * Handles the change event for the model's label attribute
     * Rerenders when the model label attribute changes
     *
     * @method handleChange
     */
    handleChange: function () {
        this.render();
    },

    /**
     * Shows the edit label input
     *
     * @method showLabelInput
     */
    showLabelInput: function () {
        // anti-jank hack part 2: set input width
        // since there's no way to do 'width:auto' for <input>. cuz that wouldn't be useful at all
        this.$labelInput.width(this.$labelText.width());

        this.$labelText.addClass('hidden');
        this.$labelInput.removeClass('hidden');

        this.focusAtEnd(this.$labelInput);

        if (this.model.isNew()) {
            this.$labelInput.select();
        }
    },

    /**
     * Gets the value of the model's label attribute
     *
     * @method getLabel
     * @return String
     */
    getLabel: function () {
        var label = this.model.get(this.labelAttribute) || '';

        // if this input is controlling a phone number, the label should be formatted
        if (this.maskPhoneNumber) {
            label = Ibp.Utils.PhoneNumber().formatNumber(String(label)).number;
        }

        return label;
    },

    /**
     * Gets the truncated value of the model's label attribute
     *
     * @method getTruncatedLabel
     * @return String
     */
    getTruncatedLabel: function () {
        return Ibp.Utils.truncate(this.getLabel(), this.truncateAt);
    },

    /**
     * Shows the input and selets the text
     *
     * @method focusInput
     */
    focusInput: function () {
        this.showLabelInput();

        this.$labelInput.select();
    },

    /**
     * Sets the caret to the end of the input so users can more easily resume typing
     *
     * @method focusAtEnd
     */
    focusAtEnd: function ($input) {
        var length = $input.val().length;

        $input.focus();

        $input[0].setSelectionRange(length, length);
    },

    /**
     * Replaces breaking spaces with non-breaking spaces
     *
     * @method escapeSpaces
     * @return String
     */
    escapeSpaces: function (str) {
        return str ? str.replace(/\s/g, '&nbsp;') : str;
    },

    /**
     * Fixes an IE bug where the change event isn't bubbled in certain situations (http://bugs.jquery.com/ticket/10818)
     * Tracking the input state for changes just for IE isn't worth the payoff, so this is going to fire on every blur
     *
     * @method triggerChange
     */
    triggerChange: function () {
        this.$labelInput.trigger('change');
    }
});

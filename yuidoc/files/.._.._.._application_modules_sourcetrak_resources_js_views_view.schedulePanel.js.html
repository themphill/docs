/**
 * Routing View
 *
 * @module Sourcetrak
 * @class Views.Wizard
 */
Ibp.Sourcetrak.Views.SchedulePanel = Ibp.Backbone.View.extend({
    name: 'SchedulePanel',
    template: Ibp.Sourcetrak.Templates.schedule_panel,

    /**
     * Pattern Choices DOM object
     *
     * @type jQuery Object
     */
    $scheduleChoices: null,

    /**
     * Pattern Choices CSS selector
     *
     * @type String
     */
    scheduleChoicesEl: '.js-schedule-choices',

    /**
     * Modes DOM object
     *
     * @type jQuery Object
     */
    $modes: null,

    /**
     * modes CSS selector
     *
     * @type String
     */
    modesEl: '.js-mode',

    /**
     * Model to hold view state
     *
     * @property state
     * @type Model
     */
    state: null,

    /**
     * Default No Schedule In Use array
     *
     * @type array
     */
    defaultNoSchedule: [{
        id: 0,
        name: 'No Schedule'
    }],

    /**
     * Define the Dom Events to bind to in delegateEvents
     *
     * @type Object
     */
    events: {
        'change .js-schedule-choices': 'handleScheduleChange',
        'click .js-mode': 'handleModeChange'
    },

    /**
     * Define the channels to subscribe
     *
     * @type Object
     */
    subscriptions: {
        'routingState': 'handleRoutingStateChange'
    },


    /**
     * Init state and render
     *
     * @method init
     */
    init: function (options) {
        this.$container = options.container;

        // set initial state
        this.state = new Ibp.Carbon.Models.State(options.routingState);

        this.listenTo(this.state, 'change:activePatternId', this.handlePatternChange);
        this.listenTo(this.state, 'change:activeScheduleId', this.render);

        this.render();
    },

    /**
     * Render
     *
     * @method render
     * @return this view
     */
    render: function () {
        var data = {
            schedules: Ibp.Sourcetrak.Data.Schedules.toJSON(),
            modes: this.getModes(this.state.get('activeScheduleId')),
            activeScheduleId: this.state.get('activeScheduleId'),
            activeModeId: this.state.get('activeModeId')
        };

        this.$el.html(this.template(data));

        this.$scheduleChoices = this.$(this.scheduleChoicesEl);

        this.$modes = this.$(this.modesEl);

        if (Ibp.Utils.isIE()) {
            this.$el.addClass('no-touch');
            this.$('select').selecter({cover: true});
        }

        this.delegateEvents();

        return this;
    },

    /**
     * getModes
     *
     * @method getModes
     */
    getModes: function (scheduleId) {
        var schedule = null,
            modes    = this.defaultNoSchedule;

        if (scheduleId > 0) {
            schedule = Ibp.Sourcetrak.Data.Schedules.get(scheduleId);

            if (schedule) {
                modes = schedule.getModes().toJSON();
            }
        }

        return modes;
    },

    /**
     * handleRoutingStateChange
     *
     * @method handleRoutingStateChange
     */
    handleRoutingStateChange: function (routingState) {
        this.state.set(routingState);
    },

    /**
     * handlePatternChange
     *
     * @method handlePatternChange
     */
    handlePatternChange: function () {
        this.render();
    },

    /**
     * handleScheduleChange
     *
     * @method handleScheduleChange
     */
    handleScheduleChange: function () {
        var scheduleId = parseInt(this.$scheduleChoices.val(), 10),
            schedule   = Ibp.Sourcetrak.Data.Schedules.get(scheduleId),
            modeId     = schedule ? schedule.getModes().at(0).id : 0,
            routingState = {
                activeScheduleId: scheduleId,
                activeModeId: modeId
            };

        if (scheduleId === 0) {
            this.model
                .getRouting()
                .getRouteByPattern(this.state.get('activePatternId'))
                .removeSchedule();
        }
        else {
            this.model
                .getRouting()
                .getRouteByPattern(this.state.get('activePatternId'))
                .setSchedule(schedule);
        }

        this.state.set(routingState);

        this.publish('routingState', routingState);

        this.render();
    },

    /**
     * handleModeChange
     *
     * @method handleModeChange
     */
    handleModeChange: function (e) {
        var $mode = $(e.target),
            modeId = parseInt($mode.data('mode-id'), 10),
            routingState = {
                activeModeId: modeId
            };

        this.state.set(routingState);

        this.publish('routingState', routingState);

        this.$modes.removeClass('active');
        $mode.addClass('active');
    },

    /**
     * onUnrender
     *
     * @method onUnrender
     */
    onUnrender: function() {
        this.undelegateSubscriptions();
    }

});

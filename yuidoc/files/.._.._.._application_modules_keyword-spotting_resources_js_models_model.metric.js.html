Ibp.KeywordSpotting.Models.Metric = Ibp.Backbone.Model.extend({
    name: 'metric',
    selected: false,
    isSaving: false,

    /**
     * validators is an object consisting of keys that map to arrays of validator objects
     * the validators object is a whitelist. attributes not listed will not be validated.
     *
     * validator: {
     *     key: [ { validate: string|function , message: string }, ... ]
     * }
     *
     * key: the name of a DeepModel property as a string
     * validate: the space-separated validation method and arguments OR custom validation function
     * message: string displayed to user if the validation fails
     */
    validators: {
        'name': [
            { validate: 'required', message: 'Name cannot be blank.' }
        ],
        'keywords.X._body.search_string': [
            { validate: 'required', message: 'Keyword cannot be blank.' }
        ],
        'keywords.X._body.weight': [
            { validate: 'num -10000 10000', message: 'Weight can only contain numeric values between -10,000 and 10,000.' },
            { validate: 'required', message: 'Weight cannot be blank.' }
        ]
    },

    names: {
        // Settings
        'name': 'Metric Name',
        'keyword.search_string': 'Keyword',
        'keyword.speaker': 'Speaker',
        'keyword.weight': 'Weight'
    },

    defaults: {
        keywords: []
    },

    init: function(options) {
        options = options || {};

        // set isSaving on request start and end
        this.listenTo(this, 'request', this.startSaving);
        this.listenTo(this, 'sync', this.endSaving);
    },

    startSaving: function() {
        this.isSaving = true;
    },

    endSaving: function() {
        this.isSaving = false;
    },

    // Orgnic URL enpoint
    url: function() {
        var url = BootstrappedData.apiUrlBase + 'keyword-spotting/metric';
        return this.isNew() ? url : url + '/' + this.id;
    },

    getMinMax: function() {
        var min = 0,
            max = 0,
            keywords = this.get('keywords');

        _.each(keywords, function(keyword) {
            var weight = Math.round(keyword._body.weight);

            if (weight > 0) {
                max += weight;
            } else if (weight < 0) {
                min += weight;
            }
        });

        return {
            min: min,
            max: max
        }
    },

    toggleSelected: function() {
        return this.setSelected(!this.selected);
    },

    setSelected: function(selection) {
        this.selected = selection;
        this.trigger('selected', this.selected);
    },

    isSelected: function() {
        return this.selected;
    },

    addNewKeyword: function() {
        // avoid race conditions and internal state conflicts by waiting for existing save to finish
        if (this.isSaving) {
            this.listenToOnce(this, 'saved', this.addNewKeyword);
            return;
        }

        var _this = this,
            keyword = {
                "_body": {
                    "search_string": "New Keyword",
                    "speaker": "2", // defaults to "Caller"
                    "weight": "0"
                }
            },
            data = {
                "keywords": [
                    keyword
                ]
            };

        this.save(data, {
            patch: true,
            silent: true,
            success: function() {
                _this.trigger('saved');
                _this.publish('notify', { type: 'save:success', msg: 'added new keyword' });
                _this.publish('newKeywordAdded', _this.getLastKeyword());
            },
            error: function() {
                _this.isSaving = false;
                _this.publish('notify', { type: 'save:error', msg: JSON.parse(jqXHR.responseText).message });
            }
        });
    },

    deleteKeyword: function(kId) {
        if (this.isSaving) {
            this.listenToOnce(this, 'saved', function() {
                this.deleteKeyword(kId);
            });
            return;
        }

        var _this = this,
            data = {},
            keyword = this.getKeyword(kId),
            deletedKeyword;

        this.splice('keywords', this.getIndex('keywords', kId), {
            silent: true
        });

        deletedKeyword = _.omit(keyword, "_body");
        data.keywords = [deletedKeyword];

        this.save(data, {
            patch: true,
            silent: true,
            success: function() {
                _this.trigger('saved');
                _this.publish('notify', { type: 'save:success', msg: 'deleted keyword' });
                _this.publish('keywordDeleted', keyword);
            },
            error: function() {
                metric.isSaving = false;
                _this.publish('notify', { type: 'save:error', msg: JSON.parse(jqXHR.responseText).message });
            }
        });
    },

    getLastKeyword: function() {
        var keywords = this.get('keywords'),
            kLength = (keywords !== null && keywords !== undefined)
                      ? keywords.length
                      : 0,
            keyword = kLength > 0 ? _.max(keywords, function(keyword) {
                return keyword._id;
            }) : undefined;

        return keyword;
    },

    getIndex: function(key, id) {
        var index = -1,
            attribute = this.get(key),
            l = attribute.length,
            i = 0;

        for (i; i < l; i++) {
            // Use == for ids that come in as strings
            if (attribute[i]._id == id) {
                index = i;
                break;
            }
        }

        return index;
    },

    getKeyword: function(kId) {
        return _.findWhere(this.get('keywords'), {"_id": kId});
    },

    splice: function(key, index, options) {
        var path = key.split('.'),
            pathLength = path.length,
            attribute = this.attributes,
            i = 0;

        options || (options = {});

        for (i; i < pathLength; i++) {
            attribute = attribute[path[i]];
        }

        attribute.splice(index, 1);

        if (!options.silent) {
            this.trigger('change', this, options);
        }

        return this;
    },

    /**
     * Set attribute(s) on keywords nested in the model
     *
     * @param options
     */
    setKeywordAttribute: function(options) {
        // avoid race conditions and internal state conflicts by waiting for existing save to finish
        if (this.isSaving) {
            this.listenToOnce(this, 'saved', function() {
                this.setKeywordAttribute(options);
            });
            return;
        }

        var _this = this,
            keywordId = options.keywordId,
            attribute = options.attribute,
            value = options.value,
            keywordIndex = this.getIndex('keywords', keywordId),
            deepKeywordObj = {},
            keyword = {
                "_id": keywordId,
                "_body": {}
            },
            data = {
                "keywords": []
            },
            keywordPath = 'keywords.' + keywordIndex + '._body',
            fullAttributePath = keywordPath + '.' + attribute;

        // only set if value is different
        if (this.get(fullAttributePath) !== value) {

            // set single attribute and push onto data array
            keyword._body[attribute] = value;
            data.keywords.push(keyword);

            // update the keyword attribute on the model. we do this silently
            // since the 'changeKeyword' event below will do the actual saving
            deepKeywordObj[fullAttributePath] = value;
            this.set(deepKeywordObj, {
                silent: true,
                validate: false
            });

            // set attribute name for notification
            options.attributeName = this.getAttributeName('keyword.' + attribute);

            if (!options.silent) {
                _this.trigger('changeKeyword', this, data, this.getKeyword(keywordId), options);
            }
        }
    },

    getAttributeName: function(key) {
        if (!_.isString(key)) {
            key = _.keys(this.flatten(key))[0];
        }

        key = key ? key : 'name';

        return this.names[key] || null;
    }
});

/**
 * Mixin for Views - when an edit fails validation,
 * the corresponding input field will be highlighted with a red border
 *
 * Elements with data-validate="<attribute name>" will be highlighted on invalid and take precedence over activeInput
 *
 * @module Carbon
 * @class Mixins.HighlightInvalid
 */
Ibp.Carbon.Mixins.HighlightInvalid = {
    /**
     * HighlightInvalid Namespace
     *
     * @property inputSelector The selector for text inputs
     * @property $activeInput The input last changed by the user to be highlight on 'invalid'
     * @property errorClass The CSS class that contains the highlight rules
     * @type Object
     */
    HighlightInvalid: {
        inputSelector: 'input[type="text"]',
        $activeInput: null,
        errorClass: 'validation-error'
    },

    /**
     * Set up events
     *
     * @method init
     */
    init: function () {
        this.bindInputs();

        this.listenTo(this.model, 'invalid', this.handleInvalid);
    },

    /**
     * Listen to text inputs for changes
     *
     * @method bindInputs
     */
    bindInputs: function () {
        var keyupEvent = 'keyup ' + this.HighlightInvalid.inputSelector,
            e = {};

        e[keyupEvent] = 'handleKeyup';

        if (this.events) {
            _.defaults(this.events, e);
        }
        else {
            this.events = e;
        }
    },

    /**
     * Text input 'keyup' handler
     *
     * @method handleKeyup
     * @param e {Object} The event object from the element that triggered the event
     */
    handleKeyup: function (e) {
        this.setActiveInput(e.target);
    },

    /**
     * Model 'invalid' event handler
     *
     * @method handleInvalid
     * @param model {Model} The model that triggered the 'invalid' event
     */
    handleInvalid: function (model) {
        this.resetHighlights();

        this.highlightError(model);
    },

    /**
     * Sets the last text input that was typed in as the input to highlight on the next 'invalid' event
     *
     * @method setActiveInput
     * @param target {DOM Element} The input DOM element to set as the active input
     */
    setActiveInput: function (target) {
        this.HighlightInvalid.$activeInput = $(target);
    },

    /**
     * Highlights the text input linked to the invalid attribute
     * Sets a listener to remove the highlight when the user changes their input
     *
     * @method highlightError
     * @param model {Model} The model that triggered the invalid event
     */
    highlightError: function (model) {
        var attributes = _.pluck(model.validationError, 'key'),
            $inputs    = this.findAttributeInputs(attributes) || this.HighlightInvalid.$activeInput;

        if ($inputs) {
            $inputs.addClass(this.HighlightInvalid.errorClass);

            // respond to user input once to remove the highlight
            $inputs.one('change', this.removeHighlight.bind(this));
        }
    },

    /**
     * Remove the highlight from an input
     *
     * @method removeHighlight
     * @param e {Object} The event object from the element that triggered the event
     */
    removeHighlight: function (e) {
        $(e.target).removeClass(this.HighlightInvalid.errorClass);
    },

    /**
     * Removes the highlights on every highlighted element
     *
     * @method resetHighlights
     */
    resetHighlights: function () {
        var selector = '.' + this.HighlightInvalid.errorClass;

        this.$(selector).removeClass(this.HighlightInvalid.errorClass);
    },

    /**
     * Takes a list of model attributes a looks for the input elements associated with them via data-validate
     *
     * @method findAttributeInputs
     * @param attributes {Object} The list of model attributes whose inputs we're searching for
     */
    findAttributeInputs: function (attributes) {
        var _this = this;

        // this insane cluster of functions takes each attribute key, finds the input
        // element associated with it, removes any empty search results, and
        // turns it into an array we can call jQuery functions on
        var inputs = _.compact($.map(attributes, function (attribute) {
            return $(_this.$('[data-validate="' + attribute + '"]')).toArray();
        }));

        return inputs.length ? $(inputs) : null;
    }
};

/**
* Graph View
*
* @module Sourcetrak
* @class Views.Graph
*/

Ibp.Sourcetrak.Views.Graph = Ibp.Backbone.View.extend({
    name: 'Graph',
    template: Ibp.Sourcetrak.Templates.graph,

    /**
     * Sets Charts container and renders view
     *
     * @method init
     * @param options {Object} Config object
     */
    init: function (options) {
        this.$container = options.$container;
        this.data = this.formatData(options.data);
        this.label = options.label;
        this.colors = options.colors;

        // render app header and container to page
        this.render();
    },

    /**
     * Renders graph Item
     *
     * @method render
     * @return this view
     */
    render: function () {
        this.$el = $(this.template({label: this.label}));
        this.$container.append(this.$el);
        options = {
            $graphContainer: this.$('.js-highchart'),
            colors: this.colors,
            dataSet: this.data,
            seriesName: this.label
        };

        this.instantiateGraph(options);

        return this;
    },

    /**
    * A single method that allows us to only specify unique properties of a new graph, while only listing out 'default'
    * properties once (in the method definition), instead with each new instantiation.
    *
    * @method renderCharts
    * @param options {Object}
    * @param options.$graphContainer {jQuery Object} container for graph
    * @param options.colors {Object}
    * @param options.colors.tooltipColor {String} border color for tooltip and hover color for data point
    * @param options.colors.seriesColor {String} line color and area fill color
    * @param options.colors.markerColor {String} color of a data point
    * @param options.dataSet {Array} server data reformatted for highcharts (each element is a point on the chart)
    * @param options.seriesName {String} unit name for tooltip
    */
    instantiateGraph: function(options) {
        var backgroundType = this.createGraphBackground(),
            newGraph = options.$graphContainer.highcharts({
            chart: {
                type: 'area',
                animation: false,
                showAxes: false,
                plotBackgroundColor: backgroundType,
                spacingRight: 0,
                spacingLeft: 0,
                marginLeft: -9,
                marginRight: -9,
                marginBottom: 0,
                marginTop: 0
            },
            plotOptions: {
                series: {
                    animation: false
                }
            },
            credits: {
                enabled: false
            },
            tooltip: {
                backgroundColor: '#fff',
                borderColor: options.colors.tooltipColor,
                borderRadius: 3,
                borderWidth: 3,
                formatter: function() {
                    var labelNoun = this.point.y === 1
                                        ? this.series.name
                                        : this.series.name + 's',
                        label     = this.point.name,
                        total     = this.point.y,
                        sharedCss = 'font-family: helvetica-neue, helvetica, arial; font-weight: bold; text-transform: uppercase; font-size: 11px; letter-spacing: 0.75px;',
                        labelCss  = 'style="' + sharedCss + 'color: #999;"',
                        pointCss  = 'style="' + sharedCss + 'color: #082336;"';

                return '<span ' + labelCss + '>' +  label + '</span><br><span ' + pointCss + '>' + total + ' ' + labelNoun + '</span>';
                }

            },
            legend: {
                enabled: false
            },
            title: {
                text: null
            },
            xAxis: {
                gridLineColor: 'transparent',
                startOnTick: false
            },
            yAxis: {
                gridLineColor: 'transparent'
            },
            series: [{
                name: options.seriesName,
                animation: false,
                data: options.dataSet,
                fillColor: options.colors.seriesColor,
                lineColor: options.colors.seriesColor,
                marker: {
                    fillColor: '#fff',
                    lineColor: options.colors.markerColor,
                    lineWidth: 3,
                    states: {
                        hover: {
                            lineColor: options.colors.tooltipColor
                        }
                    }
                }
            }]
        });

        return newGraph;
    },

    /**
     * Returns the value for this graph's backgroundColor. If the
     * graph has no data, this method adds a class to the view's $el
     * so a background image for empty sets can be properly positioned
     * based on the series type.
     *
     * @method createGraphBackground
     * @return {String} backgroundType
     */
    createGraphBackground: function () {
        var backgroundType;

        if (!this.data) {
            backgroundType = 'transparent';
            this.$el.addClass('empty ' + this.label.toLowerCase() + 's');
        }
        else {
            backgroundType = '#ffffff';
        }

        return backgroundType;
    },

    /**
     * Reformats data from server into highchart format if
     * data exists, otherwise it returns the dataset created
     * by calling the function to create an empty data set,
     * compileEmptyDataset()
     *
     * @method formatData
     * @param data {Array} server data (or 0 if no data found)
     */
    formatData: function (data) {
        var graphData;

        if (data && data.length) {
            graphData = _.map(data, function(d) {
                return {name: moment(d.date, "YYYY-MM-DD").format("MM-DD-YYYY"), y: d.total};
            });
        }

        return graphData;
    },

    /**
     * Destroys popover view and self
     *
     * @method unrender
     */
    unrender: function () {
        this.remove();
    }
});

Ibp.Sourcetrak.Views.DomainSetToolbar = Ibp.Carbon.Views.Toolbar.extend({
    name: 'DomainSetToolbar',
    el: "#domain-set-toolbar",

    /**
    * Element containing the Advanced Optiosn button, which
    * needs to be disabled if the domain set model is new
    *
    * @property advancedOptionsBtnEl
    * @type String
    */
    advancedOptionsBtnEl: '#DomainSetToolbar-advancedOptions',

    /**
    * jQuery object for advancedOptionsBtnEl
    *
    * @property $advancedOptionsBtn
    * @type jQuery Object
    */
    $advancedOptionsBtn: null,

    /**
    * Element containing the Delete button, which
    * needs to be disabled if the domain set model is new
    *
    * @property testBtnEl
    * @type String
    */
    testBtnEl: '#DomainSetToolbar-test',

    /**
    * jQuery object for testBtnEl
    *
    * @property $testBtn
    * @type jQuery Object
    */
    $testBtn: null,

    /**
    * Element containing the Delete button, which
    * needs to be disabled if the domain set model is new
    *
    * @property deleteBtnEl
    * @type String
    */
    deleteBtnEl: '#DomainSetToolbar-delete',

    /**
    * jQuery object for deleteBtnEl
    *
    * @property $deleteBtn
    * @type jQuery Object
    */
    $deleteBtn: null,

    /**
     * Array describing the buttons in the toolbar
     *
     * @property buttongroups
     * @type Array
     */
    buttongroups: [

        {
            buttons:[{
                action: 'back',
                icon: 'icon-arrow-left3',
                label: '',
                title: 'Back to SourceTrak Dashboard',
                disabled: {
                    create: false,
                    locked: false
                }
            }]
        },
        {
            buttons:[{
                action: 'advancedOptions',
                label: 'Advanced Options',
                title: 'Configure Advanced Options',
                disabled: {
                    create: true,
                    locked: true
                }
            }]
        },
        {
            buttons: [
            {
                action: 'test',
                label: 'Test',
                title: 'Test your SourceTrak Activites',
                disabled: {
                    create: true,
                    locked: true
                }
            }//,
            // {
                // action: 'undo',
                // label: 'Undo',
                // title: 'Undo Something',
                // disabled: {
                //     create: true,
                //     locked: true
                // }
            // }
            ]
        },
        {
            buttons: [
            {
                action: 'delete',
                label: 'Delete',
                title: 'Delete This Domain Set',
                disabled: {
                    create: true,
                    locked: true
                }
            }]
        }
    ],

    /**
     * List of button actions to be executed on click
     *
     * @method executeAction
     */
    executeAction: function (action) {
        switch (action) {
            case 'advancedOptions':
                this.getAdvancedOptionsModal();
                break;
            case 'back' :
                this.backToDashboard();
                break;
            case 'test' :
                this.testDomainSet();
                break;
        //  case 'undo' :
            //   this.undoPreviousSave();
            //   break;
            case 'delete' :
                this.deleteDomainSet();
                break;
            default:
                // do nothing
        }
    },

    /**
     * Supplement/customization to the toolbar view's default render method
     *
     * @method onRender
     */
    onRender: function () {
        this.$advancedOptionsBtn = $(this.advancedOptionsBtnEl);
        this.$testBtn = $(this.testBtnEl);
        this.$deleteBtn = $(this.deleteBtnEl);

        if (this.model.isNew()) {
            this.disableButtonsForNewModel();

            // Once the domain set model exists, adjust functionality
            this.listenToOnce(this.model, 'sync', this.handleSync);
        }
    },

    /**
     * Calls functions dependent on the occurance of a sync event
     *
     * @method handleSync
     */
    handleSync: function () {
        this.enableButtonsForSavedModel();
    },

    /**
     * Add a 'disabled' class to the button(s) that
     * can't function until the model has been saved.
     *
     * @method disableButtonsForNewModel
     */
    disableButtonsForNewModel: function () {
        this.$advancedOptionsBtn.addClass('disabled');
        this.$testBtn.addClass('disabled');
        this.$deleteBtn.addClass('disabled');
    },

    /**
     * Remove 'disabled' class to the buttons now that the model is saved and
     *
     * @method enableButtonsForSavedModel
     */
    enableButtonsForSavedModel: function () {
        this.$advancedOptionsBtn.removeClass('disabled');
        this.$testBtn.removeClass('disabled');
        this.$deleteBtn.removeClass('disabled');
    },


    /**
     * Back To Dashboard
     *
     * @method back
     */
    backToDashboard: function () {
        this.publish('navigate', '/');
    },

    /**
     * Test Domain Set
     *
     * @method back
     */
    testDomainSet: function () {
        this.publish('getTestModal');
    },

    // /**
    //  * Undo Previous Save
    //  *
    //  * @method back
    //  */
    // undoPreviousSave: function () {
    //     console.log('undo');
    // },

    /**
     * Tell editdomainset.js view to render the  advanced options modal
     *
     * @method back
     */
    getAdvancedOptionsModal: function () {
        this.publish('getAdvancedOptionsModal');
    },

    /**
     * Delete Domain Set
     *
     * @method back
     */
    deleteDomainSet: function (){
        this.publish('deleteDomainSetModal');
    }
});

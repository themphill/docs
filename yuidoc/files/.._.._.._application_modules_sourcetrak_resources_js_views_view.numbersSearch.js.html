Ibp.Sourcetrak.Views.NumbersSearch = Ibp.Backbone.View.extend({
    name: 'numbersSearch',
    template: Ibp.Sourcetrak.Templates.numbers_search,
    selectOptionsTemplate: Ibp.Sourcetrak.Templates.select_options,

    /**
     * The views object retains references to children views
     *
     * @property views
     * @type Object
     */
    views: null,

    /**
     * The element the view will render itself in
     *
     * @property $container
     * @type jQuery Object
     */
    $container: null,

    /**
     * Selector for the states select div
     *
     * @property statesSelectEl
     * @type String
     */
    statesSelectEl: '.js-state-list',

    /**
     * Container for the states div
     *
     * @property $states
     * @type jQuery Object
     */
    $states: null,

    /**
     * Selector for the toll free area codes select
     *
     * @property tollfreeCodesSelectEl
     * @type String
     */
    tollfreeCodesSelectEl: '.js-toll-free',

    /**
     * Container for the states area codes select
     *
     * @property $tollfreeCodes
     * @type jQuery Object
     */
    $tollfreeCodes: null,

    /**
     * Selector for the area codes select div
     *
     * @property areaCodesEl
     * @type String
     */
    areaCodesEl: '.js-state-area-codes',

    /**
     * Container for the state area codes div
     *
     * @property $areaCodes
     * @type jQuery Object
     */
    $areaCodes: null,

    /**
     * Selector for the area code text input element
     *
     * @property areaCodeEl
     * @type String
     */
    areaCodeEl: '.js-area-code',

    /**
     * Container for the area code text input
     *
     * @property $areaCode
     * @type jQuery Object
     */
    $areaCode: null,

    /**
     * Selector for the exchange text input element
     *
     * @property exchangeEl
     * @type String
     */
    exchangeEl: '.js-exchange',

    /**
     * Container for the exchange text input
     *
     * @property $exchange
     * @type jQuery Object
     */
    $exchange: null,

    /**
     * Selector for the submit button
     *
     * @property submitEl
     * @type String
     */
    submitEl: '.js-search-submit',

    /**
     * Container for the submit object
     *
     * @property $submit
     * @type jQuery Object
     */
    $submit: null,

    /**
     * Number type that is being searched for
     *
     * @property numberType
     * @type String
     */
    numberType: null,

    /**
     * Type of the object being configured (pool, one to one, or mail number)
     *
     * @property objectType
     * @type String
     */
    objectType: null,

    /**
     * Url for accessing the RPC endpoint of Organic
     *
     * @property rpcUrl
     * @type String
     */
    rpcUrl: '/v1/rpc',

    /**
     * The CSS class that contains the highlight rules
     *
     * @property errorClass
     * @type String
     */
    errorClass: 'validation-error',

    subscriptions: {
        'beginSearch': 'handleBeginSearch',
        'endSearch': 'handleEndSearch'
    },

    events: {
        'change .js-state-list': 'handleStateChange',
        'focus .js-area-code': 'clearStateSearch',
        'focus .js-exchange': 'clearStateSearch',
        'click .js-search-submit': 'handleSubmit'
    },

    /**
     * Init state and render
     *
     * @method init
     */
    init: function (options) {
        this.$container = options.$container;

        // store which type of number we're selecting (local or tollfree)
        this.numberType = options.numberType;

        // store what type of object we're configuring (pool or one-to-one)
        this.objectType = options.objectType;

        this.views = {};

        this.render();
    },

    /**
     * Render
     *
     * @method render
     * @return this view
     */
    render: function () {
        this.$el.html(this.template({
            numberType: this.numberType,
            tollFreeAreaCodes: BootstrappedData.tollfreeAreaCodes,
            objectType: this.objectType
        }));

        this.$container.html(this.$el);

        this.renderLocalSearch();

        // store a reference to the submit button
        this.$submit = this.$(this.submitEl);

        return this;
    },

    /**
     * Render items specific to the local search state of this view
     *
     * @method renderLocalSearch
     */
    renderLocalSearch: function () {
        // store the area code text input for future reference
        this.$areaCode = this.$(this.areaCodeEl);

        // store the exchange text input for future reference
        this.$exchange = this.$(this.exchangeEl);

        // store the area codes div for future reference
        this.$areaCodes = this.$(this.areaCodesEl);

        // store the states select for future reference
        this.$states = this.$(this.statesSelectEl);

        this.$tollfreeCodes = this.$(this.tollfreeCodesSelectEl);

        // render the state-related dropdown options
        this.renderAreaCodeOptions();
    },

    /**
     * Take an array of options and use them in the select options template
     *
     * @method renderAreaCodeOptions
     * @param array choices
     */
    renderAreaCodeOptions: function (choices) {
        // pass the select options template's HTML into our area codes select element
        this.$areaCodes.html(this.selectOptionsTemplate({
            choices: choices || []
        }));
    },

    /**
     * React to a change in the state dropdown by fetching related area codes and rendering them
     *
     * @method handleStateChange
     * @param Event e
     */
    handleStateChange: function (e) {
        var state = $(e.target).val();

        // Clear out area code and exchange values
        this.$areaCode.val('');
        this.$exchange.val('');

        this.fetchAreaCodesByState(state);
    },

    // TODO: Validate input manually
    /**
     * React to focus within the area-code or exchange input
     *
     * @method clearStateSearch
     */
    clearStateSearch: function () {
        this.$states.val(0);
        this.$areaCodes.html('');
    },

    /**
     * Determine what information is being submitted and publish the search term
     *
     * @method handleSubmit
     */
    handleSubmit: function () {
        var searchTerm = this.getSearchTerm();

        if (searchTerm) {
            this.publish('numberSearch', searchTerm);
        }

    },

    /**
     * Callback for the area code api request
     *
     * @method handleFetchAreaCodes
     * @param res {Array} The api response
     */
    handleFetchAreaCodes: function (res) {
        this.renderAreaCodeOptions(res);
    },

    /**
     * Handler for the event that fires just before the area code api request
     *
     * @method handleBeginSearch
     */
    handleBeginSearch: function () {
        this.disableSearch();
    },

    /**
     * Handler for the event that fires just after the area code api request
     *
     * @method handleEndSearch
     */
    handleEndSearch: function () {
        this.enableSearch();
    },

    /**
     * Makes an api request to get all the area codes for a given state
     *
     * @method fetchAreaCodesByState
     * @param state {String} the state abbreviation to search for
     */
    fetchAreaCodesByState: function (state) {
        var url  = this.rpcUrl,
            data = {
                action: 'areaCodesByState',
                state: state
            },
            success = function (res) {
                this.handleFetchAreaCodes(res);
            }.bind(this);

        return $.getJSON(url, data, success);
    },

    /**
     * Enables the area code search button
     *
     * @method enableSearch
     */
    enableSearch: function () {
        this.$submit.prop('disabled', false);
    },

    /**
     * Disables the area code search button
     *
     * @method disableSearch
     */
    disableSearch: function () {
        this.$submit.prop('disabled', true);
    },

    /**
     * Figures out which input contains the search term, validates it, and returns the value
     *
     * @method getSearchTerm
     * @return String
     */
    getSearchTerm: function () {
        /* jshint indent:false */
        var searchTerm;

        switch (this.numberType) {
            case 'canadian':
                var inputVal    = this.$areaCode.val(),
                    exchangeVal = this.$exchange.val(),
                    selectVal   = this.$areaCodes.val();

                // handle local search by state
                if (selectVal) {
                    searchTerm = selectVal;
                }
                // handle local search by user input
                else {
                    if (_.contains(BootstrappedData.canAreaCodes, inputVal)) {
                        searchTerm = inputVal;
                    }
                    else {
                        this.highlightError(this.$areaCode);

                        return;
                    }

                    // use the exchange if it's valid
                    if (exchangeVal) {
                        if (_.contains(BootstrappedData.exchanges, exchangeVal)) {
                            searchTerm += exchangeVal;
                        }
                        else {
                            this.highlightError(this.$exchange);

                            return;
                        }
                    }
                }

                break;
            case 'local':
                var inputVal    = this.$areaCode.val(),
                    exchangeVal = this.$exchange.val(),
                    selectVal   = this.$areaCodes.val();

                // handle local search by state
                if (selectVal) {
                    searchTerm = selectVal;
                }
                // handle local search by user input
                else {
                    if (_.contains(BootstrappedData.localAreaCodes, inputVal)) {
                        searchTerm = inputVal;
                    }
                    else {
                        this.highlightError(this.$areaCode);

                        return;
                    }

                    // use the exchange if it's valid
                    if (exchangeVal) {
                        if (_.contains(BootstrappedData.exchanges, exchangeVal)) {
                            searchTerm += exchangeVal;
                        }
                        else {
                            this.highlightError(this.$exchange);

                            return;
                        }
                    }
                }

                break;
            case 'tollfree':
                searchTerm = this.$tollfreeCodes.val();

                break;
            case 'true800':
                searchTerm = '800';
                break;
        }

        return searchTerm;
    },

    /**
     * Highlights the given text input
     *
     * @method highlightError
     * @param $input {jQuery Object} THe input to highlight
     */
    highlightError: function ($input) {
        $input.addClass(this.errorClass);

        this.publish('notify', {type: 'error', msg: 'Enter a valid area code.'});

        // respond to user input once to remove the highlight
        $input.one('change', this.removeHighlight.bind(this));
    },

    /**
     * Remove the highlight from an input
     *
     * @method removeHighlight
     * @param e {Object} The event object from the element that triggered the event
     */
    removeHighlight: function (e) {
        $(e.target).removeClass(this.errorClass);

        this.publish('dismissNotification');
    }
});

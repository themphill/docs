Ibp.KeywordSpotting.Views.KeywordTableRow = Ibp.Backbone.View.extend({
    name:       'keywordTableRow',
    template:   Ibp.KeywordSpotting.Templates.keyword_table_row,
    $container: null,
    keyword:    null,
    position:   'append',

    events: {
        'click .keyword-delete i':      'showDeleteModal',
        'change input':                 'autoSave',
        'change select':                'autoSave',
        'change .keyword-weight input': 'styleWeightInput'
    },

    subscriptions: {},

    init: function(options) {
        this.$container = options.container;
        this.keyword = options.keyword;
        this.position = options.position ? options.position : 'append';

        this.subscriptions['selectKeywordInput:' + this.keyword._id] = 'focusAndSelect';

        this.render();
    },

    render: function() {
        var templateData = {
            keyword:     this.keyword._body,
            speakerList: {
                '1': 'Agent',
                '2': 'Caller',
                '3': 'Either Caller or Agent'
            }
        };

        // render row
        this.$el = $(this.template(templateData));

        this.styleWeightInput();

        // append/prepend based on options
        if (this.position == 'prepend') {
            this.$container.prepend(this.$el);
        }
        else {
            this.$container.append(this.$el);
        }
    },

    /**
     * Focus and select keyword text
     */
    focusAndSelect: function() {
        this.$el.find('input[name=search_string]').focus().select();
    },

    autoSave: function(e) {
        var $target = $(e.target),
            attribute = $target.attr('name'),
            value = $target.attr('value'),
            save = $target.data('autosave');

        this.publish('keyword:autoSave', $target);

        // determine if autosave is disabled for this element
        save = (save || save === undefined) ? true : false;

        // trim strings
        if (_.isString(value)) {
            value = value.trim();
        }

        // for checkboxes, assumes boolean value
        if (e.target.type === "checkbox") {
            value = (value == "1" || value == "true") ? true : false;
            value = e.target.checked ? value : !value;
        }

        // typecast true/false radio button values
        if (e.target.type === "radio" && (value === "false" || value === "true")) {
            value = value === "true" ? true : false;
        }

        // if autosave enabled and the property exists, save the change
        if (save) {
            // clear pending changes
            $target.removeData('original-length');
            $target.removeData('original');

            // setting the attribute will trigger the change event which triggers the actual save to server
            this.model.setKeywordAttribute({
                keywordId: this.keyword._id,
                attribute: attribute,
                value:     value
            });
        }
    },

    showDeleteModal: function() {
        var _this = this,
            keyword = this.keyword._body.search_string,
            options = {
                el:       '#modal',
                data:     {keyword: keyword},
                template: Ibp.KeywordSpotting.Templates.modal_keyword_delete,
                // Context for 'this' is the modal view, not this view
                handler:  function() {
                    _this.model.deleteKeyword(_this.keyword._id);
                    this.unrender();
                }
            },
            modal = new Ibp.Carbon.Views.Modal(options);

        modal.render();
    },

    styleWeightInput: function() {
        var $target = this.$el.find('.keyword-weight input'),
            value = $target.attr('value'),
            classname = 'neutral';

        if (value == 0) {
            classname = 'neutral';
        } else if (value > 0) {
            classname = 'positive';
        } else if (value < 0) {
            classname = 'negative';
        }

        $target.removeClass('neutral positive negative').addClass(classname);
    }
});

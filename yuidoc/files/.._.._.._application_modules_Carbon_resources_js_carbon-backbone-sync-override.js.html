/**
 *  Carbon Backbone,js JSONP support
 *  Add JSONP, OAuth2, and API Status Code support to Backbone.sync
 */
Backbone.syncOriginal = Backbone.sync;

Backbone.sync = function(method, model, options) {
    var token = (BootstrappedData && BootstrappedData.oauthToken) ? BootstrappedData.oauthToken : false,
        authHeader = token === false ? '' : "Bearer " + token,
        corsSupport = true,
        params = null,

        // set original error and sucess methods, otherwise set generic
        success = $.isFunction(options.success)
                    ? options.success
                    : function (data, textStatus, jqXHR) {},
        error  = $.isFunction(options.error)
                    ? options.error
                    : function (jqXHR, textStatus, errorThrown) {},

        // change from PUT to PATCH when Organic Supports Patch
        methods = {
            'read': 'GET',
            'create': 'POST',
            'update': 'PUT',
            'delete': 'DELETE',
            'patch': 'PUT'
        };

    // update all outgoing sync requests to contain the version hash to avoid midair collisions
    if (method === "update" || method === "patch") {
        options.attrs.version_hash = model.get('version_hash');
    }
    params = options.data;

    // allow cookies to be sent with request
    if (token === false) {
        if (options.xhrFields) {
           options.xhrFields.withCredentials = true;
        } else {
            options.xhrFields = {withCredentials: true};
        }
    }

    // force PATCH requests to follow our PUT implementation of PATCH
    if (method === 'patch') {
        method = 'update';
    }

    var isRestException = function(response) {
        var isException = false;
        if (response !== null
                && response.hasOwnProperty('status')
                && response.hasOwnProperty('message')
                && response.hasOwnProperty('code')) {
            isException = true;
        }
        return isException;
    };

    if (!corsSupport) {
        options.timeout = 5000;
        options.dataType = "jsonp";
        options.data = "_method=" + methods[method];

        if (token !== false) {
            options.data += "&access_token=" + token;
        }

        // $.ajax data object
        if(params !== undefined){
            options.data += "&" + $.param(params);
        }

        // model changed attributes
        if(options.attrs !== undefined){
            options.data += "&" + $.param(options.attrs);
        }
    } else {
        options.beforeSend = function(xhr) {
            xhr.setRequestHeader("X-Requested-With", "XMLHttpRequest");
            if (token !== false) {
                xhr.setRequestHeader("Authorization", authHeader);
            }
        };
    }

    options.success = function(data, textStatus, jqXHR) {
        // jsonp check for rest exception
        if (!corsSupport && isRestException(data)) {
            if (data.status === 401) {
                window.location.href = "/logout.php";
            } else {
                // error, double check required inputs and retry
                // ToDo: notify user
                error(jqXHR, data.status, data.message);
            }
        } else {
            success(data, textStatus, jqXHR);
        }
    };

    options.error = function(jqXHR, textStatus, errorThrown) {
        // authentication error
        if (jqXHR.status === 401) {
            window.location.href = "/logout.php";
            return;
        }

        // server error
        if (jqXHR.status === 400 || jqXHR.status === 500) {
            try {
                var response = $.parseJSON(jqXHR.responseText);

                // api error
                if (isRestException(response)) {
                    // TODO: Do call the error callback - might break IVR?
                    // don't call error callback, let app handle it
                    Ibp.EventHub.trigger('apiError', response, model);

                    return;
                }
            } catch (e) {
                // malformed json string exception thrown
                error(jqXHR, textStatus, errorThrown);

                return;
            }
        }

        // parse error
        if (jqXHR.status === 0) {
            // ToDo: notify user
        }

        error(jqXHR, textStatus, errorThrown);
    };

    return Backbone.syncOriginal(method, model, options);
};

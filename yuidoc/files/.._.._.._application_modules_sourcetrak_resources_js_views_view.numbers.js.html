/**
 * Manage Numbers View
 *
 * @module Sourcetrak
 * @class Views.Numbers
 */
Ibp.Sourcetrak.Views.Numbers = Ibp.Carbon.Views.ModalPageView.extend({
    mixins: ['Autoset', 'HighlightInvalid'],
    name: 'numbers',
    template: Ibp.Sourcetrak.Templates.numbers,

    /**
     * Model to hold view state
     *
     * @property state
     * @type Model
     */
    state: null,

    /**
     * The views object retains references to children views
     *
     * @property views
     * @type Object
     */
    views: null,

    /**
    * The selector for the warning container (warning for when
    * a user tries to change between toll free and local)
    *
    * @property warningEl
    * @type String
    */
    warningEl: '.js-wizard-overlay',

    /**
    * Reference to warning
    *
    * @property $warning
    * @type jQuery Object
    */
    $warning: null,

    /**
     * Selector for the description container
     *
     * @property descriptionEl
     * @type String
     */
    descriptionEl: '.js-pool-description',

    /**
     * Reference to descriptionel
     *
     * @property $description
     * @type jQuery Object
     */
    $description: null,

    /**
     * Selector for the search container
     *
     * @property searchEl
     * @type String
     */
    searchEl: '#numbers-search',

    /**
     * Selector for the summary container
     *
     * @property summaryEl
     * @type String
     */
    summaryEl: '#numbers-summary',

    /**
     * Selector for the addon error
     *
     * @property addonErrorEl
     * @type String
     */
    addonErrorEl: '.js-addon-error',

    /**
     * Reference to addon error
     *
     * @property $addonError
     * @type jQuery Object
     */
    $addonError: null,

    /**
     * Selector for the number type radio buttons
     *
     * @property typeEl
     * @type String
     */
    typeEl: '.js-number-type',

    /**
     * Reference to the number type radio buttons
     *
     * @property $type
     * @type jQuery Object
     */
    $type: null,

    /**
     * Data store from the wizard with data from other step views
     *
     * @property wizardData
     * @type Object
     */
    wizardData: null,

    /**
     * default to displaying the local number search, for new pools/OTOs
     *
     * @property DEFAULT_NUMBER_TYPE
     * @type String
     */
    DEFAULT_NUMBER_TYPE: 'local',

    /**
     * Max character length before a model name needs to be truncated
     * within this view.
     *
     * @property MODEL_NAME_MAX_CHARS
     * @type Number
     */
    MODEL_NAME_MAX_CHARS: 114,

    subscriptions: {
        'beginSearch': 'handleBeginSearch',
        'endSearch': 'handleEndSearch',
        'noSlots': 'handleNoSlots'
    },

    events: {
        'change input[name="number-type"]': 'handleTypeRadio',
        'focus .js-pool-description': 'handleDescriptionFocus',
        'click .js-change-continue': 'continueChange',
        'click .js-change-cancel': 'cancelChange'
    },

    /**
     * Init state and render
     *
     * @method init
     */
    init: function (options) {
        _.bindAll(this);

        this.publish('beginLoadStep');

        this.$container = options.$container;
        this.wizardData = options.wizardData;

        this.views = {};

        // set initial state
        this.state = new Ibp.Carbon.Models.State({
            type: options.type,
            numberType: this.getNumberType(),
            newNumbersType: null,
            edit: options.edit,
            isLocation: options.type === 'location'
        });

        // set the one to one flag on the model depending on the type of the wizard
        if (this.model.isNew()) {
            this.model.one_to_one = (options.type === 'pool') ? false : true;
        }

        this.listenTo(this.state, 'change:numberType', this.handleNumberTypeChange);

        this.render();

        this.publish('endLoadStep');
    },

    /**
     * Render
     *
     * @method render
     * @return this view
     */
    render: function () {
        var data = {
            numberType: this.state.get('numberType'),
            description: Ibp.Utils.truncate((this.model.get('description') || this.model.get('name')), this.MODEL_NAME_MAX_CHARS),
            isLocation: this.state.get('isLocation')
        };

        this.$el.html(this.template(data));

        this.$container.html(this.$el);

        this.$addonError = this.$(this.addonErrorEl);

        this.$warning = this.$(this.warningEl);

        this.$description = this.$(this.descriptionEl);

        this.$type = this.$(this.typeEl);

        // render the search view
        this.renderSearch();

        // render the summary template
        this.renderSummary();

        this.delegateEvents();

        return this;
    },

    /**
     * Clears the default pool name in the input field on focus
     *
     * @method handleDescriptionFocus
     */
    handleDescriptionFocus: function () {
        // Only clear the field contents if the user hasn't put different text in
        if (this.model.isNew() && this.$description.val() === this.model.defaults.description) {
            this.$description.val('');
        }
    },

    /**
     * Unrender child views when view is being removed
     *
     * @method onUnrender
     */
    onUnrender: function () {
        _.invoke(this.views, 'unrender');
    },

    /**
     * Renders the summary view into the main numbers template
     *
     * @method renderSummary
     * @param Object options
     */
    renderSummary: function (options) {
        options = options || {};

        // if this view currently exists, remove it
        if (this.views.summary) {
            this.views.summary.unrender();
        }

        this.hideAddonError();

        this.views.summary = new Ibp.Sourcetrak.Views.NumbersSummary({
            $container: this.$(this.summaryEl),
            type: this.state.get('type'),
            numberType: this.state.get('numberType'),
            model: this.model,
            reset: options.reset || false,
            wizardData: this.wizardData,
            edit: this.state.get('edit')
        });
    },

    /**
     * Render the search view into the main numbers template
     *
     * @method renderSearch
     */
    renderSearch: function () {
        // if this view currently exists, remove it
        if (this.views.search) {
            this.stopListening(this.views.search);
            this.views.search.remove();
        }

        // render the search template
        this.views.search = new Ibp.Sourcetrak.Views.NumbersSearch({
            $container: this.$(this.searchEl),
            numberType: this.state.get('numberType'),
            objectType: this.state.get('type'),
            model: this.model
        });
    },

    /**
     * React to a new radio button being selected
     *
     * @method handleTypeRadio
     * @param Event e
     */
    handleTypeRadio: function (e) {
        var type = $(e.target).val();

        // show a warning if changing the type results in numbers being removed
        if (this.model.getTotalNumbers()) {
            // store user's choice before warning so we can continue set action if warning accepted
            this.state.set('newNumbersType', type);

            this.renderWarning();
        }
        else {
            this.state.set('numberType', type);
        }
    },

    /**
     * Renders an overlay that warns the user of repercussions for changing number types
     *
     * @method renderWarning
     */
    renderWarning: function () {
        this.$warning.removeClass('hidden');
    },

    /**
     * Continues with the set for the numbers type change, using the stored user choice,
     * and calls function to remove the warning overlay.
     *
     * @method continueChange
     */
    continueChange: function () {
        this.removeAllAreaCodes();

        this.state.set({
            numberType: this.state.get('newNumbersType'),
            newNumbersType: null
        });

        this.unrenderWarning();
    },

    /**
     * Does not follow through with numbers type change (does nothing
     * other than call function to remove the warning overlay).
     *
     * @method cancelChange
     */
    cancelChange: function () {
        var selector = 'input[value=' + this.state.previous('numberType') + ']';

        this.$(selector).attr('checked', 'checked');
        this.state.set('newNumbersType', null);

        this.unrenderWarning();
    },

    /**
     * Removes the warning overlay
     *
     * @method unrenderWarning
     */
    unrenderWarning: function () {
        this.$warning.addClass('hidden');
    },

    /**
     * Updates the view when the number type is changed between local, toll free and registered
     *
     * @method handleNumberTypeChange
     */
    handleNumberTypeChange: function () {
        this.renderSearch();

        this.renderSummary();
    },

    /**
     * Handler for when an area code search begins
     *
     * @method handleBeginSearch
     */
    handleBeginSearch: function () {
        this.hideAddonError();

        this.disableType();
    },

    /**
     * Handler for when an area code search ends
     *
     * @method handleEndSearch
     */
    handleEndSearch: function () {
        this.enableType();
    },

    /**
     * Hides the error message for number slots
     *
     * @method handleNoSlots
     */
    handleNoSlots: function () {
        this.showAddonError();
    },

    /**
     * Get the number type from the model or use default
     *
     * @method getNumberType
     * @return String
     */
    getNumberType: function () {
        return this.model.getNumberType() || this.DEFAULT_NUMBER_TYPE;
    },

    /**
     * Shows the number slots error
     *
     * @method showAddonError
     */
    showAddonError: function () {
        this.$addonError.show();
    },

    /**
     * Hide the number slots error
     *
     * @method hideAddonError
     */
    hideAddonError: function () {
        this.$addonError.hide();
    },

    /**
     * Enables the type radio buttons
     *
     * @method enableType
     */
    enableType: function () {
        this.$type.prop('disabled', false);
    },

    /**
     * Disables the type radio buttons
     *
     * @method disableType
     */
    disableType: function () {
        this.$type.prop('disabled', true);
    },

    /**
     * Removes all the area codes from a pool
     *
     * @method removeAllAreaCodes
     */
    removeAllAreaCodes: function () {
        // for new pools, remove area codes completely
        if (this.model.isNew()) {
            this.model.set('area_codes', null);

            this.model.getAreaCodes().resetSubModels();
        }
        // for existing pools, mark the area codes for removal by the server
        else {
            _.each(this.model.getAreaCodes().models, function (areaCode) {
                areaCode.set('quantity', 0);
            });
        }

        // reset the addons we need to order
        this.wizardData.addons = {};
    }
});

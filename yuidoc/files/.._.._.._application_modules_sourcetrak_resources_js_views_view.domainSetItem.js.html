/**
 * Domain Set Item View
 *
 * @module Sourcetrak
 * @class Views.DomainSetItem
 */
(function () {
    function getDomainsLength (domains) {
        return domains.join('').length;
    }

    Ibp.Sourcetrak.Views.DomainSetItem = Ibp.Backbone.View.extend({
        name: 'DomainSetItem',
        template: Ibp.Sourcetrak.Templates.domain_set_item,

        /**
         * Backbone Model to hold view state
         *
         * @property state
         * @type Model
         */
        state: null,

        /**
         * Object to hold child views
         *
         * @property views
         * @type Object
         */
        views: null,

        /**
         * Reference to the Domain Popover view
         *
         * @property popoverView
         * @type View
         */
        popoverView: null,

        /**
         * jQuery reference for the graphs container
         *
         * @property $graphs
         * @type jQuery Object
         */
        $graphs: null,

        /**
         * jQuery reference for the graphs toggle button
         *
         * @property $toggle
         * @type jQuery Object
         */
        $toggle: null,

        /**
         * Selector for the element that triggers the domains popover
         *
         * @property domainsEl
         * @type String
         */
        domainsEl: '.js-domains',

        /**
         * Selector for the graphs container
         *
         * @property graphsEl
         * @type String
         */
        graphsEl: '.domainSet-graphs',

        /**
         * Selector for the button that shows/hides the graphs container
         *
         * @property toggleEl
         * @type String
         */
        toggleEl: '.js-graphs-toggle',

        /**
         * Class for the toggle button when the graphs are expanded
         *
         * @property expandedClass
         * @type String
         */
        expandedClass: 'icon-angle-up',

        /**
         * Class for the toggle button when the graphs are collapsed
         *
         * @property collapsedClass
         * @type String
         */
        collapsedClass: 'icon-angle-down',

        /**
         * Max number of characters that can fit in the view before the domains popover is displayed
         *
         * @property MAX_CHARS
         * @type Number
         */
        MAX_CHARS: 40,

        /**
         * DOM Event handlers
         *
         * @property events
         */
        events: {
            'click .js-domainSet-name': 'nameHandler',
            'click .js-graphs-toggle': 'toggleHandler'
        },

        /**
         * Sets DomainSets container and renders view
         *
         * @method init
         * @param options {Object} Config object
         */
        init: function (options) {
            // set initial state
            this.state = new Ibp.Carbon.Models.State({
                // whether the view is expanded or collapsed
                expanded: options.expanded,
                // if we need to show the domains popover
                truncateDomains: this.getDomains().join('').length > this.MAX_CHARS,
                // have the graph views been rendered in this item row
                renderedGraphs: false
            });

            this.views = {};

            this.listenTo(this.state, 'change:expanded', this.toggleGraphs);

            this.render();
        },

        /**
         * Renders DomainSet Item
         * Actually attaching this view to the DOM is handled by Views.Dashboard
         *
         * @method render
         * @return this view
         */
        render: function () {
            var toggleBtnClass = this.state.get('expanded') ? this.expandedClass : this.collapsedClass,
                data = {
                    name: this.model.get('name'),
                    domains: this.getDomainsString(),
                    calls: this.model.getStats('total_calls_30_days'),
                    visits: this.model.getStats('total_visits_30_days')
                };

            // create fragment
            this.$el.html(this.template(data));

            // store references to the graphs and toggle button
            this.$graphs = this.$(this.graphsEl);
            this.$toggle = this.$(this.toggleEl);

            // show or hide graphs
            if (!this.state.get('expanded')) {
                this.$graphs.addClass('hidden');
            } else {
                this.renderItemGraphs();
            }

            // show the appropriate arrow for the toggle button
            this.$toggle.addClass(toggleBtnClass);

            // show domains popover
            if (this.state.get('truncateDomains')) {
                this.views.popover = new Ibp.Sourcetrak.Views.DomainsPopover({
                    $container: this.$(this.domainsEl),
                    domains: this.getDomains()
                });
            }

            return this;
        },

        /**
         * Render the two graph views inside of the domain set list item
         *
         * @method renderItemGraphs
         */
        renderItemGraphs: function () {
            this.views.visitsGraph = new Ibp.Sourcetrak.Views.Graph({
                label: 'Visit',
                $container: this.$graphs,
                data: this.model.getStats('daily_visits_30_days'),
                colors: {
                    tooltipColor: '#27ae60',
                    seriesColor: '#97e6b8',
                    markerColor: '#88d3a7'
                }
            });

            this.views.callsGraph = new Ibp.Sourcetrak.Views.Graph({
                label: 'Call',
                $container: this.$graphs,
                data: this.model.getStats('daily_calls_30_days'),
                colors: {
                    tooltipColor: '#1377b6',
                    seriesColor: '#3498da',
                    markerColor: '#227eba'
                }
            });

            this.state.set('renderedGraphs', true);
        },

        /**
         * Click handler for the Domain Set Name
         *
         * @method toggleHandler
         */
        nameHandler: function () {
            this.publish('navigate', '#' + this.model.id);
        },

        /**
         * Click handler for the expand/collapse button
         *
         * @method toggleHandler
         */
        toggleHandler: function () {
            this.toggleExpandedState();
        },

        /**
         * Switches the 'expanded' view state property between true and false
         *
         * @method toggleExpandedState
         */
        toggleExpandedState: function () {
            this.state.toggle('expanded');
        },

        /**
         * 'Expanded' view state event listener
         * Animates the graph expand/collapse and changes the button arrow icon
         *
         * @method toggleGraphs
         */
        toggleGraphs: function () {
            if (!this.state.get('renderedGraphs')) {
                this.renderItemGraphs();
            }

            // show/hide the graphs
            this.$graphs.slideToggle({
                duration: 350,
                easing: 'linear',
                always: function () {
                    // toggle the arrow icon
                    this.$toggle.toggleClass(this.expandedClass);
                    this.$toggle.toggleClass(this.collapsedClass);
                }.bind(this)
            });
        },

        /**
         * Gets the array of this domain set's domains
         *
         * @method getDomains
         * @return domains {Array}
         */
        getDomains: function () {
            return this.model.getDomains().pluck('domain');
        },

        /**
         * Gets the max number of domains to show in the row
         * The rest will be shown in the popover
         *
         * @method getTruncatedDomains
         * @return domains {Array}
         */
        getTruncatedDomains: function () {
            var domains = _.clone(this.getDomains()),
                domainsLength,
                remainingChars,
                lastDomain,
                lastDomainLength;

            // if we're still over the limit without the last domain,
            // then we can just drop it
            while (getDomainsLength(domains.slice(0, -1)) > this.MAX_CHARS) {
                domains.pop();
            }

            // set values with the whittled down array
            domainsLength    = getDomainsLength(domains);
            remainingChars   = domainsLength - this.MAX_CHARS;
            lastDomain       = _.last(domains);
            lastDomainLength = _.last(domains).length;

            // truncate the last item
            domains[domains.length - 1] = lastDomain.slice(0, lastDomainLength - remainingChars);

            return domains;
        },

        /**
         * Gets the domains as a concatenated string
         *
         * @method getDomainsString
         * @return domains {String}
         */
        getDomainsString: function () {
            var domains = this.state.get('truncateDomains')
                        ? this.getTruncatedDomains()
                        : this.getDomains();

            return domains.join('&nbsp;&nbsp;');
        },

        /**
         * Destroys popover view and self
         *
         * @method unrender
         */
        unrender: function () {
            _.invoke(this.views, 'unrender');

            this.views.length = 0;

            this.remove();
        }
    });
})();

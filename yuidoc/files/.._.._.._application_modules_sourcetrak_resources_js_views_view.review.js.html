/**
 * Review Number Purchase View
 *
 * @module Sourcetrak
 * @class Views.Wizard
 */
Ibp.Sourcetrak.Views.Review = Ibp.Carbon.Views.ModalPageView.extend({
    name: 'review',
    template: Ibp.Sourcetrak.Templates.review,

    /**
     * String selector for loading overlay
     *
     * @property loadingEl
     * @type string
     */
    loadingEl: '.js-review-receipt-overlay',

    /**
     * Model to hold view state
     *
     * @property state
     * @type Model
     */
    state: null,

    /**
     * The views object retains references to children views
     *
     * @property views
     * @type Object
     */
    views: {},

    events: {
        'click .js-dirty-numbers-ok': 'dismissNotice'
    },

    /**
     * Init state and render
     *
     * @method init
     */
    init: function (options) {
        this.publish('beginLoadStep');

        this.$container = options.$container;
        this.wizardData = options.wizardData;

        // set initial state
        this.state = new Ibp.Carbon.Models.State({
            showRouting: true,
            edit: options.edit
        });

        this.name = this.model.get('description');
        this.addons = this.wizardData.addons;
        this.areaCodes = this.model.getAreaCodes();
        this.routing = this.model.get('routing');
        this.dirtyNumbers = false;

        this.calculateTotalNumbers();
        this.setupAddonCosts();
        this.calculateBillingTotals();

        this.render();

        this.publish('endLoadStep');
    },

    /**
     * handles hiding the dirty number notice overlay
     *
     * @method dismissNotice
     */
    dismissNotice: function () {
        this.dirtyNumbers = false;
        this.render();
    },

    /**
     * calculates the total amount of numbers from the areaCodes object
     *
     * @method calculateTotalNumbers
     */
    calculateTotalNumbers: function () {
        this.totalNumbers = 0;
        _.each(this.areaCodes.models, function(areaCode) {
            var code = _.findWhere(this.wizardData.currentNumbers, {area_code: areaCode.get('area_code')});
                originalQuantity = code ? code.quantity : 0,
                quantity = areaCode.get('quantity'),
                newQuantity = (quantity - originalQuantity) > 0 ? (quantity - originalQuantity) : 0,
                available = this.wizardData.available.hasOwnProperty(areaCode.get('area_code'))
                            ? parseInt(this.wizardData.available[areaCode.get('area_code')].ifbyphone, 10)
                            : 0,
                unconfigured = this.wizardData.unconfigured.hasOwnProperty(areaCode.get('area_code'))
                            ? parseInt(this.wizardData.unconfigured[areaCode.get('area_code')], 10)
                            : 0,
                totalAvailable = available + unconfigured;

            this.totalNumbers += quantity;

            // Check quantity vs available
            if (totalAvailable > 0 && totalAvailable < newQuantity) {
                this.dirtyNumbers = true;
            }
        }.bind(this));
    },

    /**
     * retrieves billing information from wizardData and calculates totals
     *
     * @method calculateBillingTotals
     */
    calculateBillingTotals: function () {
        this.proratedFee = 0;
        this.regulatoryFee = 0;
        this.totalFee = 0;

        // Grab billingInfo from wizardData
        this.billingType = this.wizardData.billingInfo.type;
        this.regulatoryFeePercent = this.wizardData.billingInfo.regulatoryFeePercent;
        this.prorateFactor = this.wizardData.billingInfo.prorateFactor;

        // Calculate totals
        this.proratedFee = this.recurringCharge * parseFloat(this.prorateFactor);
        this.regulatoryFee = (this.proratedFee + this.setupFee) * parseFloat(this.regulatoryFeePercent);
        this.totalFee = this.proratedFee + this.setupFee + this.regulatoryFee;
        this.recurringCharge = this.recurringCharge + (this.recurringCharge * this.regulatoryFeePercent);
    },

    /**
     * Gets addon costs from wizardData and associates them with the addon being purchased
     *
     * @method setupAddonCosts
     */
    setupAddonCosts: function () {
        this.addonQuantity = 0;
        this.addonType = null;
        this.addonCost = 0;
        this.addonSetupCost = 0;
        this.recurringCharge = 0;
        this.setupFee = 0;

        _.each(this.addons, function (quantity, type) {
            if (quantity > 0) {
                this.addonType = type;
            }
        }.bind(this));

        if (this.addonType) {
            this.addonQuantity = this.addons[this.addonType];
            this.addonCost = this.wizardData.addonInfo[this.addonType].purchasePrice;
            this.addonSetupCost = this.wizardData.addonInfo[this.addonType].nonRecurringCharge;
            this.recurringCharge += this.addonQuantity * parseFloat(this.addonCost);
            this.setupFee += this.addonQuantity * parseFloat(this.addonSetupCost);
        }
    },

    /**
     * Render
     *
     * @method render
     * @return this view
     */
    render: function () {
        var data = {
            name: this.name,
            monthlyFee: this.proratedFee,
            regulatoryFee: this.regulatoryFee,
            totalFee: this.totalFee,
            proratedFee: this.proratedFee,
            setupFee: this.setupFee,
            recurringCharge: this.recurringCharge,
            showRouting: this.state.get('showRouting'),
            mainNumber: this.mainNumber,
            areaCodes: this.areaCodes.toJSON(),
            totalNumbers: this.totalNumbers,
            summary: this.summary,
            routingSummary: this.model.getRoutingSummary(),
            addonCost: this.addonCost,
            addonSetupCost: this.addonSetupCost,
            addonQuantity : this.addonQuantity,
            addonType: this.addonType === 'true800' ? 'true 800' : this.addonType,
            dirtyNumbers: this.dirtyNumbers,
            edit: this.state.get('edit')
        };

        this.$el = $(this.template(data));

        this.$container.html(this.$el);

        return this;
    }
});

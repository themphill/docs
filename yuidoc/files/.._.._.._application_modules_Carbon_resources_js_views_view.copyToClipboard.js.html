/**
 * Copy To Clipboard View
 * Wrapper module for ZeroClipboard
 * ZeroClipboard API https://github.com/zeroclipboard/zeroclipboard/blob/v2.1.5/docs/api/ZeroClipboard.md
 *
 * Basic view that allows you to easily attach copy-to-clipboard functionality to any element
 * Views that extend this class must define the 'template' property
 * The template should contain the element that will be the copy button
 * The copy button element must have an id and the property 'data-clipboard-target'
 * The value of 'data-clipboard-target' must be the id of the element that contains the copy text
 * On successful copy, a confirmation tooltip will briefly appear. The location is configurable
 *
 * Uses FlashDetect for graceful degradation
 * For unsupported broswers, this view will abort rendering and destroy itself
 *
 * To prevent slowing page loads, this view will load the ZeroClipboard library asynchronously upon instantiation
 * The view will then render once the library has loaded
 *
 * initClipboard can be overridden to add functionality
 *
 * For an implementation example, see sourcetrak/resources/js/views/view.CopyToClipboard.js
 * and sourcetrak/resources/js/templates/copy_to_clipboard.hbs
 *
 * @module Carbon
 * @class Views.CopyToClipboard
 * @requires Ibp.Utils.FlashDetect, ZeroClipboard
 */
Ibp.Carbon.Views.CopyToClipboard = Ibp.Backbone.View.extend({
    name: 'copyToClipboard',

    /**
     * Checks for support, loads lib, then renders and binds handler
     *
     * @method init
     * @param options {Object}
     * @param options.$container {jQuery Object} The element to render the copy button into
     * @param options.copyEl {String} The element id of the copy button
     * @param options.tooltipLoc {String} The position of the confirmation Bootstrap tooltip
     */
    init : function (options) {
        _.bindAll(this);

        this.$container = options.$container;
        this.copyEl = options.copyEl;
        this.copySelector = '#' + this.copyEl;
        this.tooltipLoc = options.tooltipLoc;

        this.load();
    },

    /**
     * Renders copy button
     *
     * @method render
     * @return this view
     */
    render: function () {
        this.$el = $(this.template({}));

        this.$container.append(this.$el);

        // get reference to the copy button
        this.$copy = this.$container.find(this.copySelector);

        return this;
    },

    /**
     * Loads ZeroClipboard lib
     * Will either use/fetch the lib or abort this view
     *
     * @method load
     */
    load: function () {
        var _this = this;

        // TODO: Currently disabled for all IE. Need to figure out no-conflict
        // check flash and browser support, otherwise just destroy this view
        if (Ibp.Utils.FlashDetect.versionAtLeast(11) && !Ibp.Utils.isIE()) {
            // fetch the lib if we don't already have it
            if (window.ZeroClipboard) {
                this.initClipboard();
            } else {
                $.ajax({
                    url: '/assets/js/lib/zeroclipboard/ZeroClipboard.js',
                    dataType: 'script',
                    cache: true,
                    success: function() {
                        _this.initClipboard();
                    }
                });
            }
        } else {
            this.remove();
        }
    },

    /**
     * Creates an instance of the clipboard and binds the copy handler
     *
     * @method setup
     */
    setup: function () {
        var _this = this;

        // set the correct path for the swf file and instantiate the clipboard
        ZeroClipboard.config({swfPath: '/assets/js/lib/zeroclipboard/ZeroClipboard.swf'});
        this.clipboard = new ZeroClipboard(document.getElementById(this.copyEl));

        // display confirmation tooltip on copy
        this.clipboard.on("copy", function () {
            // create tooltip
            _this.$copy.tooltip({
                placement: _this.tooltipLoc,
                title: 'Copied!',
                trigger: 'manual'
            });

            _this.$copy.tooltip('show');

            // fade out, then remove
            _this.$container.find('.tooltip')
                .fadeOut({
                    duration: 2000,
                    easing: 'easeInQuint',
                    always: function () {
                        _this.$copy.tooltip('destroy');
                    }
                });
        });

        // check IE and run no conflict script for modals
        if (Ibp.Utils.isIE()) {
            this.ieNoConflict();
        }
    },

    /**
     * Renders and initializes the clipboard view
     * Child views can override this method to add functionality
     *
     * @method initClipboard
     */
    initClipboard: function () {
        this.render();

        this.setup();
    },

    /**
     * Destroys clipboard and removes this view
     *
     * @method unrender
     */
    unrender: function () {
        if (this.clipboard) {
            this.clipboard.destroy();
        }

        this.remove();
    },

    // TODO: make this actually work so we can enable copy for IE
    /**
     * Patches Bootstrap modal to fix conflict with IE when using clipboard
     *
     * @method ieNoConflict
     */
    ieNoConflict: function () {
        (function($) {
            var that = this,
                zcClass = ZeroClipboard.config('containerClass');

            // Bootstrap 2.3.2
            var proto = $.fn.modal.Constructor.prototype;
            proto.enforceFocus = function() {
                $(document).on('focusin.modal', function (e) {
                    if (that.$element[0] !== e.target
                        && !that.$element.has(e.target).length
                        // Adding this final condition check is the only real change
                        && !$(e.target).closest(zcClass).length) {
                        that.$element.focus();
                    }
                });
            };
        })(window.jQuery);
    }
});
